175    public void reloadSettings() {
176        TracerUrlCheckBox.setSelected(m_customTracerUrl);
177        TracerUrlValueField.setText(m_customTracerUrlText);
178
179        TracerAdjustCheckBox.setSelected(m_tracerAdjustPosition);
180        TracerAdjustLatSpinner.setValue(new Integer(m_tracerAdjustPositionLatVal));
181        TracerAdjustLonSpinner.setValue(new Integer(m_tracerAdjustPositionLonVal));
182        TracerLatSignComboBox.setSelectedIndex(m_tracerAdjustPositionLatSign);
183        TracerLonSignComboBox.setSelectedIndex(m_tracerAdjustPositionLonSign);
184
185        RuianUrlCheckBox.setSelected(m_customRuianUrl);
186        RuianUrlValueField.setText(m_customRuianUrlText);
187
188        RuianAdjustCheckBox.setSelected(m_ruianAdjustPosition);
189        RuianAdjustLatSpinner.setValue(new Integer(m_ruianAdjustPositionLatVal));
190        RuianAdjustLonSpinner.setValue(new Integer(m_ruianAdjustPositionLonVal));
191        RuianLatSignComboBox.setSelectedIndex(m_ruianAdjustPositionLatSign);
192        RuianLonSignComboBox.setSelectedIndex(m_ruianAdjustPositionLonSign);
193    }

646    private void setLatLonAdjust() {
647      // Tracer
648      if (m_tracerAdjustPosition) {
649        m_tracerAdjustPositionLat = (double) m_tracerAdjustPositionLatVal / 10000000.0d;
650        m_tracerAdjustPositionLon = (double) m_tracerAdjustPositionLonVal / 10000000.0d;
651        if (m_tracerAdjustPositionLatSign == 1)
652          m_tracerAdjustPositionLat *= -1;
653        if (m_tracerAdjustPositionLonSign == 1)
654          m_tracerAdjustPositionLon *= -1;
655      } else {
656        m_tracerAdjustPositionLat = 0;
657        m_tracerAdjustPositionLon = 0;
658      }
659      // RUIAN
660      if (m_ruianAdjustPosition) {
661        m_ruianAdjustPositionLat = (double) m_ruianAdjustPositionLatVal / 10000000.0d;
662        m_ruianAdjustPositionLon = (double) m_ruianAdjustPositionLonVal / 10000000.0d;
663        if (m_ruianAdjustPositionLatSign == 1)
664          m_ruianAdjustPositionLat *= -1;
665        if (m_ruianAdjustPositionLonSign == 1)
666          m_ruianAdjustPositionLon *= -1;
667      } else {
668        m_ruianAdjustPositionLat = 0;
669        m_ruianAdjustPositionLon = 0;
670      }
671    }

708    private void EnabledModulesListValueChanged(javax.swing.event.ListSelectionEvent evt) {
709      int cnt = EnabledModulesList.getSelectedIndices().length;
710      switch (cnt) {
711        case 0: ModulesLeftButton.setEnabled(false);
712                ModulesUpButton.setEnabled(false);
713                ModulesDownButton.setEnabled(false);
714                break;
715        case 1: ModulesLeftButton.setEnabled(true);
716                ModulesUpButton.setEnabled(true);
717                ModulesDownButton.setEnabled(true);
718                break;
719       default: ModulesLeftButton.setEnabled(true);
720                ModulesUpButton.setEnabled(false);
721                ModulesDownButton.setEnabled(false);
722                break;
723      }
724
725      if (cnt == 1) {
726        int selIndex = EnabledModulesList.getSelectedIndex();
727        if (selIndex == 0) {
728          ModulesUpButton.setEnabled(false);
729        } else if (selIndex == m_modulesEnabledModel.getSize() - 1) {
730          ModulesDownButton.setEnabled(false);
731        }
732      }
733    }

745    /**
746     *  Sets Available modules
747     *  @param modules - List of strings to add
748     */
749    @SuppressWarnings("unchecked")
750    public void setAvailableModules(List<String> modules) {
751
752      // Clean up Enabled modules - remove not existing modules
753      for (int i = 0; i< m_modulesEnabledModel.getSize(); i++) {
754        if (modules.indexOf(m_modulesEnabledModel.get(i)) < 0) {
755          m_modulesEnabledModel.remove(i);
756        }
757      }
758
759      // Add remaining modules as Available
760      m_modulesAvailableModel.clear();
761      for (int i = 0; i < modules.size(); i++) {
762        String m = modules.get(i);
763        if (m_modulesEnabledModel.indexOf(m) < 0)
764          m_modulesAvailableModel.addElement(m);
765      }
766    }

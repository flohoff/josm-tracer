142    private ArrayList<LatLon> parseGeometry (String geometry) {
143
144        ArrayList<LatLon> arrList = new ArrayList<>();
145        LatLon prevCoor = null;
146
147        String[] coorVal = geometry.split(" ");
148        for (int i = 0; i < coorVal.length; i = i + 2) {
149            String x = coorVal[i];
150            String y = coorVal[i+1];
151            krovak k = new krovak();
152            LatLon ll = k.krovak2LatLon(x, y);
153
154            // Sometimes, after rouding, two nodes could have the same LatLon coordinates
155            // Skip duplicated coordinate
156            if (prevCoor == null || !ll.equalsEpsilon(prevCoor)) {
157                arrList.add(ll);
158                prevCoor = ll;
159            }
160        }
161        return arrList;
162    }

164    /**
165    * Parse given XML string and fill variables with LPIS data
166    * There are two modes:
167    *   - basic - get LPIS ID and geometry
168    *   - extra - get type (landuse) of the element
169    *  @param action - basic or extra
170    *  @param xmlStr - data for parsing
171     * @throws javax.xml.parsers.ParserConfigurationException Parser Configuration Exception
172     * @throws org.xml.sax.SAXException Incorrect xml structure
173     * @throws java.io.IOException Input/Output issue
174     * @throws javax.xml.xpath.XPathExpressionException Wrong XPath Expression
175    *
176    */
177    public void parseXML (String xmlStr) throws ParserConfigurationException, SAXException, IOException, XPathExpressionException {
178
179        System.out.println("");
180        System.out.println("parseXML() - Start");
181
182        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
183        DocumentBuilder builder = factory.newDocumentBuilder();
184
185        Document doc = builder.parse(new ByteArrayInputStream(xmlStr.getBytes("UTF-8")));
186        doc.getDocumentElement().normalize();
187
188        XPath xPath =  XPathFactory.newInstance().newXPath();
189        init();
190        String expID = "//*[name()='ms:LPIS_DPB_UCINNE'][1]/*[name()='ms:id']";
191        String expOuter = "//*[name()='ms:LPIS_DPB_UCINNE'][1]//*[name()='gml:exterior']//*[name()='gml:posList']";
192        String expInner = "//*[name()='ms:LPIS_DPB_UCINNE'][1]//*[name()='gml:interior']//*[name()='gml:posList']";
193        String expUsage = "//*[name()='ms:LPIS_DPB_UCINNE'][1]/*[name()='ms:kultura']";
194
195        NodeList nodeList;
196
197        System.out.println("parseXML(basic) - expID: " + expID);
198        nodeList = (NodeList) xPath.compile(expID).evaluate(doc, XPathConstants.NODESET);
199        if (nodeList.getLength() > 0) {
200            m_lpis_id = Long.parseLong(nodeList.item(0).getFirstChild().getNodeValue());
201        } else {
202            return;
203        }
204
205        System.out.println("parseXML(basic) - m_lpis_id: " + m_lpis_id);
206
207        System.out.println("parseXML(basic) - expOuter: " + expOuter);
208        nodeList = (NodeList) xPath.compile(expOuter).evaluate(doc, XPathConstants.NODESET);
209        String outer = nodeList.item(0).getFirstChild().getNodeValue();
210        System.out.println("parseXML(basic) - outer: " + outer);
211        List<LatLon> way = parseGeometry(outer);
212        System.out.println("parseXML(basic) - outer list: " + way);
213        super.setOuter(way);
214
215        System.out.println("parseXML(basic) - expInner: " + expInner);
216        nodeList = (NodeList) xPath.compile(expInner).evaluate(doc, XPathConstants.NODESET);
217        for (int i = 0; i < nodeList.getLength(); i++) {
218            String inner = nodeList.item(i).getFirstChild().getNodeValue();
219            System.out.println("Inner("+i+": "+ inner);
220            super.addInner(parseGeometry(inner));
221        }
222        List<List<LatLon>> inner_ways = super.getInners();
223        for (int i = 0; i < inner_ways.size(); i++) {
224            System.out.println("parseXML(basic) - Inner("+i+"): " + inner_ways.get(i));
225        }
226
227        nodeList = (NodeList) xPath.compile(expUsage).evaluate(doc, XPathConstants.NODESET);
228        if (nodeList != null && nodeList.getLength() > 0 && nodeList.item(0).hasChildNodes()) {
229            m_usage = nodeList.item(0).getFirstChild().getNodeValue();
230            mapToOsm();
231        }
232        System.out.println("parseXML(extra) - m_usage: " + m_usage);
233
234        System.out.println("parseXML() - End");
235    }

271    static List<LpisRecord> parseBasicXML(String content, double adjlat, double adjlon) throws ParserConfigurationException, SAXException, IOException, XPathExpressionException {
272        DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
273        DocumentBuilder docBuilder = docBuilderFactory.newDocumentBuilder();
274        Document doc = docBuilder.parse (new ByteArrayInputStream(content.getBytes("utf-8")));
275        doc.getDocumentElement().normalize();
276
277        XPath xPath =  XPathFactory.newInstance().newXPath();
278        String expID = "//*[name()='ms:LPIS_DPB_UCINNE']/*[name()='ms:id']";
279
280        System.out.println("parseXML(basic) - expID: " + expID);
281        NodeList expids = (NodeList) xPath.compile(expID).evaluate(doc, XPathConstants.NODESET);
282
283        List<LpisRecord> list = new ArrayList<> (expids.getLength());
284
285        for (int i = 0; i < expids.getLength(); i++) {
286            final long exp_id = Long.parseLong(expids.item(i).getFirstChild().getNodeValue());
287
288            LpisRecord lpis = new LpisRecord (adjlat, adjlon);
289            lpis.m_lpis_id = exp_id;
290
291            String expOuter = "//*[name()='ms:LPIS_DPB_UCINNE' and @*='LPIS_DPB_UCINNE." + Long.toString(exp_id) + "']//*[name()='gml:exterior']//*[name()='gml:posList']";
292            NodeList outernl = (NodeList) xPath.compile(expOuter).evaluate(doc, XPathConstants.NODESET);
293            if (outernl.getLength() > 0) {
294                String outer = outernl.item(0).getFirstChild().getNodeValue();
295                List<LatLon> way = lpis.parseGeometry(outer);
296                lpis.setOuter(way);
297            } else {
298                continue;
299            }
300
301            String expInner = "//*[name()='ms:LPIS_DPB_UCINNE' and @*='LPIS_DPB_UCINNE." + Long.toString(exp_id) + "']//*[name()='gml:interior']//*[name()='gml:posList']";
302            NodeList innersnl = (NodeList) xPath.compile(expInner).evaluate(doc, XPathConstants.NODESET);
303
304            for (int j = 0; j < innersnl.getLength(); j++) {
305                String inner = innersnl.item(j).getFirstChild().getNodeValue();
306                lpis.addInner(lpis.parseGeometry(inner));
307            }
308
309            String expUsage = "//*[name()='ms:LPIS_DPB_UCINNE'] and @*='LPIS_DPB_UCINNE." + Long.toString(exp_id) + "']/*[name()='ms:kultura']";
310            NodeList usagenl = (NodeList) xPath.compile(expUsage).evaluate(doc, XPathConstants.NODESET);
311            if (usagenl != null && usagenl.getLength() > 0 && usagenl.item(0).hasChildNodes()) {
312                lpis.m_usage = usagenl.item(0).getFirstChild().getNodeValue();
313                lpis.mapToOsm();
314            }
315
316            list.add (lpis);
317        }
318
319        return list;
320    }

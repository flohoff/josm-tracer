020package org.openstreetmap.josm.plugins.tracer.modules.lpis;
021
022 /**
023  * Class to store krovak XY coordinates
024  */
025public class xyCoor {
026  private Double m_x;
027  private Double m_y;
028
029/**
030  * Constructor
031  */
032  public xyCoor () {
033    m_x = 0.;
034    m_y = 0.;
035  }
036
037/**
038  * Constructor
039  * @param x - the X coordinate
040  * @param y - the Y coordinate
041  */
042  public xyCoor (Double x, Double y) {
043    m_x = x;
044    m_y = y;
045  }
046
047/**
048  * Set X and Y coordinates
049  * @param x - the X coordinate
050  * @param y - the Y coordinate
051  */
052  public void set (Double x, Double y) {
053    m_x = x;
054    m_y = y;
055  }
056
057/**
058  * Return X coordinate
059  * @return The X coordinate
060  */
061  public Double x () {
062    return m_x;
063  }
064
065/**
066  * Return Y coordinate
067  * @return The Y coordinate
068  */
069  public Double y () {
070    return m_y;
071  }
072}

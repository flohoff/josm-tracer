222    /**
223     * Parse given JSON string and fill variables with RUIAN data
224     *
225     * @param jsonStr JSON string with RUIAN data
226     */
227    public void parseJSON(String jsonStr) {
228
229        init();
230
231        JsonObject obj;
232        try (JsonReader jsonReader = Json.createReader(new ByteArrayInputStream(jsonStr.getBytes()))) {
233            obj = jsonReader.readObject();
234        }
235
236        m_source = obj.getString("source");
237        if (m_source.length() == 0)
238            m_source = "cuzk:ruian";
239
240        String keys = retrieveJsonString(obj, "keys");
241        if (keys != null) {
242            String[] kv = keys.replace("\"", "").replace(",{", "").replace("{", "").replace("}}", "}").split("}");
243            System.out.println("keys: " + Arrays.toString(kv));
244            for (int i = 0; i < kv.length; i++) {
245                System.out.println("key[" + i + "]: " + kv[i]);
246                String[] x = kv[i].split(",");
247                m_keys.put(x[0], x[1]);
248            }
249            m_keys.put("source", m_source);
250        }
251
252        JsonArray arr = retrieveJsonArray(obj, "geometry");
253        if (arr != null && arr.size() > 0) {
254            List<LatLon> way = new ArrayList<>(arr.size());
255            for (int i = 0; i < arr.size(); i++) {
256                System.out.println("i=" + i);
257                JsonArray node = arr.getJsonArray(i);
258
259                LatLon coor = new LatLon(
260                        node.getJsonNumber(1).doubleValue(),
261                        node.getJsonNumber(0).doubleValue()
262                );
263                System.out.println("coor: " + coor.toString());
264                way.add(coor);
265            }
266            super.setOuter(way);
267        }
268    }

280    /**

287    }

289    /**

297
298    /**
299     * Return whether traced object is building (has building key)
300     *
301     * @return True/False - is building or not
302     */
303    public boolean isBuilding() {
304        return m_keys.containsKey("building");
305    }
306
307    /**
308     * Returns whether traced object is a land (has landuse, natural or leisure
309     * key)
310     *
311     * @return True/False - is land or not
312     */
313    public boolean isLand() {
314        return m_keys.containsKey("landuse")
315                || m_keys.containsKey("natural")
316                || m_keys.containsKey("leisure");
317    }
318
319    /**
320     * Returns whether traced object is a garden (has leisure=garden key)
321     *
322     * @return True/False - is garden or not
323     */
324    public boolean isGarden() {
325        return m_keys.containsKey("leisure") && m_keys.get("leisure").equals("garden");
326    }
327
328    @Override
329    public boolean hasData() {
330        return super.hasOuter() && !m_keys.isEmpty();
331    }

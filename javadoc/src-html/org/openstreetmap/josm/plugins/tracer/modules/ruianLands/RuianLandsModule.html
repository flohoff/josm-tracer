171        @Override
172        protected TracerRecord downloadRecord(LatLon pos) throws Exception {
173            TracerPreferences pref = TracerPreferences.getInstance();
174            String sUrl = RuianLandsUrl;
175            if (pref.isCustomRuainUrlEnabled())
176              sUrl = pref.getCustomRuainUrl();
177
178            // Get coordinate corrections
179            double adjlat = 0, adjlon = 0;
180            if (pref.isRuianAdjustPositionEnabled()) {
181              adjlat = pref.getRuianAdjustPositionLat();
182              adjlon = pref.getRuianAdjustPositionLon();
183            }
184
185            RuianLandsServer server = new RuianLandsServer();
186            return server.trace(m_pos, sUrl, adjlat, adjlon);
187        }

189        @Override
190        protected EdObject createTracedPolygonImpl(WayEditor editor) {
191
192            System.out.println("  RUIAN keys: " + record().getKeys());
193
194            Match clipWayMatch;
195            Match mergeWayMatch;
196            Match retraceAreaMatch;
197
198            // Determine type of the objects
199            if (record().isBuilding()) {
200              clipWayMatch = m_clipBuildingWayMatch;
201              mergeWayMatch = m_mergeBuildingWayMatch;
202              retraceAreaMatch = m_retraceBuildingAreaMatch;
203            } else if (record().isGarden()) {
204              clipWayMatch = m_clipGardenWayMatch;
205              mergeWayMatch = m_mergeLanduseWayMatch;
206              retraceAreaMatch = m_retraceLandsAreaMatch;
207            } else {
208              clipWayMatch = m_clipLanduseWayMatch;
209              mergeWayMatch = m_mergeLanduseWayMatch;
210              retraceAreaMatch = m_retraceLandsAreaMatch;
211            }
212
213            // Look for object to retrace
214            EdObject retrace_object = null;
215            if (m_performRetrace) {
216                Pair<EdObject, Boolean> repl = getObjectToRetrace(editor, m_pos, retraceAreaMatch);
217                retrace_object = repl.a;
218                boolean ambiguous_retrace = repl.b;
219
220                if (ambiguous_retrace) {
221                    postTraceNotifications().add(tr("Multiple existing Ruian polygons found, retrace is not possible."));
222                    return null;
223                }
224            }
225
226            // Only update tags, do not change geometry of the object
227            if (m_updateTagsOnly) {
228
229                // To update tags only we need an existing object to update
230                if (retrace_object == null) {
231                    postTraceNotifications().add(tr("No existing Ruian polygon found, tags only update is not possible."));
232                    return null;
233                }
234
235                // TODO - allows update of Multipolygons tags
236                if (!retrace_object.isWay() || retrace_object.hasReferrers()) {
237                    postTraceNotifications().add(tr("Multipolygon retrace is not supported yet."));
238                    return null;
239                }
240
241                // Tag object
242                tagTracedObject(retrace_object);
243
244                return retrace_object;
245            }
246
247            // Update object geometry as well
248            // Create traced object
249            Pair<EdWay, EdMultipolygon> trobj = this.createTracedEdObject(editor);
250            if (trobj == null)
251                return null;
252            EdWay outer_way = trobj.a;
253            EdMultipolygon multipolygon = trobj.b;
254
255            // Retrace simple ways - just use the old way
256            if (retrace_object != null) {
257                if ((multipolygon != null) || !retrace_object.isWay() || retrace_object.hasReferrers()) {
258                    postTraceNotifications().add(tr("Multipolygon retrace is not supported yet."));
259                    return null;
260                }
261                EdWay retrace_way = (EdWay)retrace_object;
262                retrace_way.setNodes(outer_way.getNodes());
263                outer_way = retrace_way;
264            }
265
266            // Everything is inside DataSource bounds?
267            if (!checkInsideDataSourceBounds(multipolygon == null ? outer_way : multipolygon, retrace_object)) {
268                wayIsOutsideDownloadedAreaDialog();
269                return null;
270            }
271
272            // Tag object
273            tagTracedObject(multipolygon == null ? outer_way : multipolygon);
274
275            // Connect to touching nodes of near building polygons
276            connectExistingTouchingNodes(multipolygon == null ? outer_way : multipolygon);
277
278            // Clip other areas
279            if (m_performClipping) {
280                AreaPredicate filter = new AreaPredicate (clipWayMatch);
281                if (m_invertClip) {
282                    ClipObjectArea clip = new ClipObjectArea (editor, m_clipSettings, postTraceNotifications ());
283                    EdObject obj = clip.clipObject (multipolygon == null ? outer_way : multipolygon, filter, m_pos);
284                    outer_way = obj.getFirstOuterWay();
285                    multipolygon = obj.isMultipolygon() ? (EdMultipolygon)obj : null;
286                }
287                else {
288                    // #### Now, it clips using only the outer way. Consider if multipolygon clip is necessary/useful.
289                    ClipAreas clip = new ClipAreas(editor, m_clipSettings, postTraceNotifications());
290                    clip.clipAreas(outer_way, filter);
291                }
292            }
293
294            // Merge duplicate ways
295            // (Note that outer way can be merged to another way too, so we must watch it.
296            // Otherwise, outer_way variable would refer to an unused way.)
297            if (m_performWayMerging) {
298                AreaPredicate merge_filter = new AreaPredicate (mergeWayMatch);
299                MergeIdenticalWays merger = new MergeIdenticalWays(editor, merge_filter);
300                outer_way = merger.mergeWays(editor.getModifiedWays(), true, outer_way);
301            }
302
303            return multipolygon == null ? outer_way : multipolygon;
304        }

306        private void tagTracedObject (EdObject obj) {
307
308            Map <String, String> map = obj.getKeys();
309
310            Map <String, String> new_keys = new HashMap <> (record().getKeys());
311            for (Map.Entry<String, String> new_key: new_keys.entrySet()) {
312                map.put(new_key.getKey(), new_key.getValue());
313            }
314            // #### delete any existing retraced tags??
315            obj.setKeys(map);
316        }

331            // Prepare outer way nodes

333            List<LatLon> outer = record().getOuter();

335            for (int i = 0; i < outer.size() - 1; i++) {
336                EdNode node = editor.newNode(outer.get(i));

382        private Pair<EdObject, Boolean> getObjectToRetrace(WayEditor editor, LatLon pos, Match retraceAreaMatch) {
383            AreaPredicate filter = new AreaPredicate(retraceAreaMatch);
384            Set<EdObject> areas = editor.useNonEditedAreasContainingPoint(pos, filter);
385
386            String ruianref = Long.toString(record().getLandID());
387
388            // restrict to RUIAN areas only, yet ... #### improve in the future
389            boolean multiple_areas = false;
390            EdObject building_area = null;
391            for (EdObject area: areas) {
392                  // Retrace all building if possible
393//                 String source = area.get("source");
394//                 if (source == null || !source.equals("cuzk:ruian"))
395//                     continue;
396
397                if (area.isWay())
398                    System.out.println("Retrace candidate EdWay: " + Long.toString(area.getUniqueId()));
399                else if (area.isMultipolygon())
400                    System.out.println("Retrace candidate EdMultipolygon: " + Long.toString(area.getUniqueId()));
401
402                String ref = area.get("ref:ruian:building");
403                if (ref != null && ref.equals(ruianref)) // exact match ;)
404                    return new Pair<>(area, false);
405
406                if (building_area == null)
407                    building_area = area;
408                else
409                    multiple_areas = true;
410            }
411
412            if (multiple_areas) {
413                return new Pair<>(null, true);
414            }
415
416            if (building_area != null) {
417                return new Pair<>(building_area, false);
418            }
419
420            return new Pair<>(null, false);
421        }

423        private void connectExistingTouchingNodes(EdObject obj) {
424            // Setup filters - include building nodes only, exclude all nodes of the object itself
425            IEdNodePredicate area_filter = new AreaBoundaryWayNodePredicate(m_reuseExistingBuildingNodeMatch);
426            IEdNodePredicate exclude_my_nodes = new ExcludeEdNodesPredicate(obj);
427            IEdNodePredicate filter = new EdNodeLogicalAndPredicate (exclude_my_nodes, area_filter);
428
429            obj.connectExistingTouchingNodes(m_tolerance, filter);
430        }
431
432        private boolean checkInsideDataSourceBounds(EdObject new_object, EdObject retrace_object) {
433            LatLonSize bounds_oversize = LatLonSize.get(new_object.getBBox(), oversizeInDataBoundsMeters);
434            if (retrace_object != null && !retrace_object.isInsideDataSourceBounds(bounds_oversize))
435                return false;
436            return new_object.isInsideDataSourceBounds(bounds_oversize);
437        }

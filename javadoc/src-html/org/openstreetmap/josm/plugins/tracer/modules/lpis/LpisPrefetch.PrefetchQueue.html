020package org.openstreetmap.josm.plugins.tracer.modules.lpis;
021
022import java.util.ArrayList;
023import java.util.HashSet;
024import java.util.Set;
025import java.util.concurrent.ExecutorService;
026import java.util.concurrent.Executors;
027import org.openstreetmap.josm.data.coor.LatLon;
028import org.openstreetmap.josm.data.osm.BBox;
029import org.openstreetmap.josm.plugins.tracer.QuadCache;
030import org.openstreetmap.josm.plugins.tracer.connectways.LatLonSize;
031
032public class LpisPrefetch {
033
034    private static final ExecutorService m_prefetchExecutor = Executors.newSingleThreadExecutor();
035
036    private final LatLonSize m_quadSize;
037    private final LpisServer m_lpisServer;
038    private final Set<QuadCache.QuadIndex> m_prefetchedTiles = new HashSet<> ();
039
040    private final Object m_lock = new Object ();
041    private PrefetchQueue <QuadCache.QuadIndex> m_prefetchQueue = null;  // m_lock, null means prefetching task is not running
042
043    public LpisPrefetch (LatLonSize quad_size, LpisServer server) {
044        m_quadSize = quad_size;
045        m_lpisServer = server;
046    }
047
048    public void schedulePrefetch (LatLon pos) {
049        QuadCache.QuadIndex qi = QuadCache.QuadIndex.latLonToQuadIndex(m_quadSize, pos.lat(), pos.lon());
050        QuadCache.QuadIndex[] list = new QuadCache.QuadIndex[9];
051        int index = 0;
052        list[index++] = new QuadCache.QuadIndex (qi.iLat() - 1, qi.iLon() - 1);
053        list[index++] = new QuadCache.QuadIndex (qi.iLat() - 1, qi.iLon() + 1);
054        list[index++] = new QuadCache.QuadIndex (qi.iLat() + 1, qi.iLon() - 1);
055        list[index++] = new QuadCache.QuadIndex (qi.iLat() + 1, qi.iLon() + 1);
056        list[index++] = new QuadCache.QuadIndex (qi.iLat(), qi.iLon() - 1);
057        list[index++] = new QuadCache.QuadIndex (qi.iLat(), qi.iLon() + 1);
058        list[index++] = new QuadCache.QuadIndex (qi.iLat() - 1, qi.iLon());
059        list[index++] = new QuadCache.QuadIndex (qi.iLat() + 1, qi.iLon());
060        list[index++] = qi;
061        schedulePrefetchTiles (list);
062    }
063
064    private void schedulePrefetchTiles (QuadCache.QuadIndex[] list) {
065        synchronized (m_lock) {
066
067            PrefetchQueue <QuadCache.QuadIndex> new_queue = null;
068
069            for (QuadCache.QuadIndex qi : list) {
070
071                // already prefetched?
072                if (m_prefetchedTiles.contains(qi)) {
073                    System.out.println ("prefetch: already prefetched: " + qi.toString());
074                    continue;
075                }
076
077                // prefetch task is running, add to existing prefetch queue
078                if (m_prefetchQueue != null) {
079                    System.out.println ("prefetch: adding to running queue: " + qi.toString());
080                    m_prefetchQueue.add (qi);
081                    continue;
082                }
083
084                // no prefetch task running, prepare add to new queue
085                if (new_queue == null)
086                    new_queue = new PrefetchQueue<> ();
087                System.out.println ("prefetch: scheduling for new prefetch batch: " + qi.toString());
088                new_queue.add (qi);
089            }
090
091            // launch prefetch task if there're tiles to prefetch
092            if (new_queue != null) {
093                m_prefetchQueue = new_queue;
094                m_prefetchExecutor.submit(new Runnable () {
095                    @Override
096                    public void run() {
097                        prefetchTask ();
098                    }
099                });
100            }
101        }
102    }
103
104    private void prefetchTask () {
105
106        QuadCache.QuadIndex qi = null;
107        boolean succeeded = false;
108
109        System.out.println ("prefetch: starting prefetch task");
110
111        while (true) {
112
113            synchronized (m_lock) {
114
115                // handle previously downloaded tile
116                if (qi != null) {
117                    m_prefetchQueue.remove (qi);
118                    if (succeeded) {
119                        m_prefetchedTiles.add (qi);
120                    }
121                }
122
123                // get a non-prefetched tile from queue
124                while (true) {
125                    if (m_prefetchQueue.isEmpty()) {
126                        System.out.println ("prefetch: queue drained, leaving prefetch task");
127                        m_prefetchQueue = null;
128                        return;
129                    }
130                    QuadCache.QuadIndex aqi = m_prefetchQueue.peek();
131                    if (m_prefetchedTiles.contains(aqi)) {
132                        System.out.println ("prefetch: queued tile already prefetched: " + aqi.toString());
133                        m_prefetchQueue.remove(aqi);
134                        continue;
135                    }
136                    qi = aqi;
137                    break;
138                }
139            }
140
141            succeeded = downloadLpisTile (qi);
142        }
143    }
144
145    private boolean downloadLpisTile(QuadCache.QuadIndex qi) {
146        System.out.println ("prefetch: downloading tile: " + qi.toString());
147
148        try {
149            BBox box = QuadCache.QuadIndex.quadIndexToBBox(m_quadSize, qi);
150            m_lpisServer.prefetchRecords(box);
151        }
152        catch (Exception e) {
153            return false;
154        }
155
156        return true;
157    }
158
159    class PrefetchQueue<Element> {
160        private final ArrayList<Element> m_Queue = new ArrayList<>(9);
161
162        public boolean isEmpty () {
163            return m_Queue.isEmpty();
164        }
165
166        public Element peek () {
167            return m_Queue.get(m_Queue.size() - 1);
168        }
169
170        public boolean remove (Element e) {
171            int index = m_Queue.indexOf (e);
172            if (index < 0) {
173                return false;
174            }
175            m_Queue.remove (index);
176            return true;
177        }
178
179        public void add (Element e) {
180            int index = m_Queue.indexOf (e);
181            if (index < 0) {
182                m_Queue.add (e);
183            }
184            else if (index != m_Queue.size() - 1) {
185                m_Queue.remove (index);
186                m_Queue.add (e);
187            }
188        }
189    }
190}

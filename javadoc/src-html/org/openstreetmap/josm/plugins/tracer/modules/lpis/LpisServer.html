020package org.openstreetmap.josm.plugins.tracer.modules.lpis;
021
022import java.io.BufferedReader;
023import java.io.IOException;
024import java.io.UnsupportedEncodingException;
025import java.net.MalformedURLException;
026import java.util.List;
027import javax.xml.parsers.ParserConfigurationException;
028import javax.xml.xpath.XPathExpressionException;
029import org.openstreetmap.josm.data.coor.LatLon;
030import org.openstreetmap.josm.data.osm.BBox;
031import org.openstreetmap.josm.plugins.tracer.TracerUtils;
032import org.openstreetmap.josm.plugins.tracer.connectways.LatLonSize;
033import org.xml.sax.SAXException;
034
035public class LpisServer {
036
037    private final String m_url;
038    private final LpisCache m_lpisCache;
039
040    // LpisRecords have fixed constant coord adjustment
041    private static final double adjustLat = 0.0;
042    private static final double adjustLon = 0.0;
043
044    public LpisServer(String url, LatLonSize cache_tile_size) {
045        m_url = url;
046        m_lpisCache = new LpisCache (cache_tile_size);
047    }
048
049    /**
050     * Call Trace server.
051     * @param urlString Input parameters.
052     * @return Result text.
053     * @throws java.net.MalformedURLException Wrong url
054     * @throws java.io.UnsupportedEncodingException Unsuported encoding
055     * @throws java.io.IOException Input/Output issue
056     */
057    private String callServer(String urlString) throws MalformedURLException, UnsupportedEncodingException, IOException {
058        try (BufferedReader reader = TracerUtils.openUrlStream (urlString, "UTF-8")) {
059            StringBuilder sb = new StringBuilder();
060            String line;
061            while ((line = reader.readLine()) != null) {
062                if (sb.length() > 0)
063                    sb.append(" ");
064                sb.append(line);
065            }
066            return sb.toString();
067        }
068    }
069
070    /**
071     * Get element ID and geometry of the land on the position.
072     * @param pos Position of the land.
073     * @return Land ID and geometry.
074     * @throws java.io.UnsupportedEncodingException Unsuported encoding
075     * @throws javax.xml.parsers.ParserConfigurationException Parser Configuration Exception
076     * @throws org.xml.sax.SAXException Incorrect xml structure
077     * @throws javax.xml.xpath.XPathExpressionException Wrong XPath Expression
078     */
079    public LpisRecord getRecord (LatLon pos) throws UnsupportedEncodingException, IOException, ParserConfigurationException, SAXException, XPathExpressionException {
080
081        // cached?
082        LpisRecord rec = m_lpisCache.get (pos);
083        if (rec != null)
084            return rec;
085
086        krovak k = new krovak();
087        xyCoor xy = k.LatLon2krovak(pos);
088
089        System.out.println ("LatLon: "+pos+" <-> XY: "+xy.x()+" "+xy.y());
090        String bbox = xy.x()+","+xy.y()+","+xy.x()+","+xy.y();
091
092        String request = m_url + "?VERSION=1.1.0&SERVICE=WFS&REQUEST=GetFeature&TYPENAME=LPIS_DPB_UCINNE&bbox="+bbox+"&SRSNAME=EPSG:102067";
093
094        System.out.println("Request: " + request);
095        String content = callServer(request);
096        System.out.println("Reply: " + content);
097        LpisRecord lpis = new LpisRecord(adjustLat, adjustLon);
098        lpis.parseXML(content);
099
100        // cache record
101        if (lpis.hasData()) {
102            m_lpisCache.add(lpis);
103        }
104
105        return lpis;
106    }
107
108    void prefetchRecords (BBox bbox) throws UnsupportedEncodingException, IOException, ParserConfigurationException, SAXException, XPathExpressionException {
109        krovak k = new krovak();
110
111        LatLon a = bbox.getTopLeft();
112        LatLon b = bbox.getBottomRight();
113
114        xyCoor axy = k.LatLon2krovak(a);
115        xyCoor bxy = k.LatLon2krovak(b);
116
117        String wfsbox = axy.x()+","+axy.y()+","+bxy.x()+","+bxy.y();
118
119        String request = m_url + "?VERSION=1.1.0&SERVICE=WFS&REQUEST=GetFeature&TYPENAME=LPIS_DPB_UCINNE&bbox="+wfsbox+"&SRSNAME=EPSG:102067";
120
121        System.out.println("Request: " + request);
122        String content = callServer(request);
123        System.out.println("Reply: " + content);
124
125        List<LpisRecord> list = LpisRecord.parseBasicXML (content, adjustLat, adjustLon);
126
127        long prefetched = 0;
128        long existing = 0;
129
130        for (LpisRecord lpis: list) {
131
132            // ignore incomplete records
133            if (lpis.getLpisID() <= 0 || !lpis.hasOuter())
134                continue;
135
136            // ignore records already in cache (avoids unnecessary downloads of extra data)
137            if (m_lpisCache.containsLpisID(lpis.getLpisID())) {
138                ++existing;
139                continue;
140            }
141
142            request = m_url + "?VERSION=1.1.0&SERVICE=WFS&REQUEST=GetFeature&TYPENAME=LPIS_DPB_UCINNE&&featureID=LPIS_DPB_UCINNE."+lpis.getLpisID()+"&SRSNAME=EPSG:102067";
143            System.out.println("Request: " + request);
144            content = callServer(request);
145            System.out.println("Reply: " + content);
146            lpis.parseXML(content);
147
148            // cache record
149            if (lpis.hasData()) {
150                m_lpisCache.add(lpis);
151                ++prefetched;
152            }
153        }
154
155        System.out.println("LpisCache: prefetched: " + Long.toString(prefetched) + ", existing: " + Long.toString (existing) + " bbox: " + bbox.toString());
156    }
157}

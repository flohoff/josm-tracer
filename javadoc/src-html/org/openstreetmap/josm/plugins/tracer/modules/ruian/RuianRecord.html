037public final class RuianRecord extends TracerRecord {

058        super.init();
059        m_source = "";

063        m_usage_code = "";
064        m_usage_key = "";
065        m_usage_val = "";
066        m_finished = "";
067        m_valid_from = "";
068        m_address_places = new ArrayList<>();

071    /**
072     * Parse given JSON string and fill record with RUIAN data
073     *
074     * @param jsonStr JSON string with RUIAN data
075     */
076    public void parseJSON(String jsonStr) {
077
078        init();
079
080        long adr_id = 0;
081        String house_number = "";
082        String house_number_typ = "";
083        String street_number = "";
084        String street = "";
085        String place = "";
086        String suburb = "";
087        String city = "";
088        String district = "";
089        String region = "";
090        String postcode = "";
091
092        JsonObject obj;
093        try (JsonReader jsonReader = Json.createReader(new ByteArrayInputStream(jsonStr.getBytes()))) {
094            obj = jsonReader.readObject();
095        }
096
097        // get source
098        m_source = parseJsonString(obj, "source", m_source);
099
100        // no geometry? leave record without data
101        JsonObject geomObj = retrieveJsonObject(obj, "geometry");
102        if (geomObj == null)
103            return;
104
105        // outer geometry
106        JsonArray outerArr = geomObj.getJsonArray("outer");
107        List<LatLon> way = new ArrayList<>(outerArr.size());
108        for (int i = 0; i < outerArr.size(); i++) {
109            JsonArray node = outerArr.getJsonArray(i);
110            LatLon coor = new LatLon(
111                    LatLon.roundToOsmPrecision(node.getJsonNumber(1).doubleValue()),
112                    LatLon.roundToOsmPrecision(node.getJsonNumber(0).doubleValue())
113            );
114            System.out.println("outer[" + i + "]:coor: " + coor.toString());
115            way.add(coor);
116        }
117        super.setOuter(way);
118
119        // inner geometries
120        JsonArray innersArr = retrieveJsonArray(geomObj, "inners");
121        if (innersArr != null) {
122            for (int i = 0; i < innersArr.size(); i++) {
123                JsonArray innerArr = innersArr.getJsonArray(i);
124                ArrayList<LatLon> inner = new ArrayList<>();
125
126                System.out.println("");
127                for (int j = 0; j < innerArr.size(); j++) {
128                    JsonArray node = innerArr.getJsonArray(j);
129                    LatLon coor = new LatLon(
130                            LatLon.roundToOsmPrecision(node.getJsonNumber(1).doubleValue()),
131                            LatLon.roundToOsmPrecision(node.getJsonNumber(0).doubleValue())
132                    );
133                    System.out.println("inner[" + i + "][" + j + "]:coor: " + coor.toString());
134                    inner.add(coor);
135                }
136                super.addInner(inner);
137            }
138        }
139
140        // SO data
141        JsonObject building = retrieveJsonObject(obj, "stavebni_objekt");
142        if (building != null) {
143            m_ruian_id = parseJsonLong(building, "ruian_id", m_ruian_id);
144            house_number = parseJsonString(building, "cislo_domovni", house_number);
145            house_number_typ = parseJsonString(building, "cislo_domovni_typ", house_number_typ);
146            street_number = parseJsonString(building, "cislo_orientacni", street_number);
147            adr_id = parseJsonLong(building, "adresni_misto_kod", adr_id);
148            street = parseJsonString(building, "ulice", street);
149            place = parseJsonString(building, "cast_obce", place);
150            suburb = parseJsonString(building, "mestska_cast", suburb);
151            city = parseJsonString(building, "obec", city);
152            district = parseJsonString(building, "okres", district);
153            region = parseJsonString(building, "kraj", region);
154            postcode = parseJsonString(building, "psc", postcode);
155            m_levels = parseJsonInt(building, "pocet_podlazi", m_levels);
156            m_flats = parseJsonInt(building, "pocet_bytu", m_flats);
157            m_usage_code = parseJsonString(building, "zpusob_vyuziti_kod", m_usage_code);
158            m_usage_key = parseJsonString(building, "zpusob_vyuziti_key", m_usage_key);
159            m_usage_val = parseJsonString(building, "zpusob_vyuziti_val", m_usage_val);
160            m_valid_from = parseJsonString(building, "plati_od", m_valid_from);
161            m_finished = parseJsonString(building, "dokonceni", m_finished);
162        }
163
164        // address places
165        JsonArray addrArr = retrieveJsonArray(obj, "adresni_mista");
166        if (addrArr != null) {
167            for (int i = 0; i < addrArr.size(); i++) {
168                JsonObject addrPlace = addrArr.getJsonObject(i);
169                Address addr = new Address();
170
171                adr_id = parseJsonLong(addrPlace, "ruian_id", adr_id);
172                house_number = parseJsonString(addrPlace, "cislo_domovni", house_number);
173                street_number = parseJsonString(addrPlace, "cislo_orientacni", street_number);
174                street = parseJsonString(addrPlace, "ulice", street);
175
176                addr.setRuianID(adr_id);
177
178                switch (house_number_typ) {
179                    case "číslo popisné":
180                        addr.setConscriptionNumber(house_number);
181                        break;
182                    case "číslo evidenční":
183                        addr.setProvisionalNumber(house_number);
184                        break;
185                }
186
187                addr.setStreetNumber(street_number);
188                addr.setStreet(street);
189                addr.setPlace(place);
190                addr.setSuburb(suburb);
191                addr.setCity(city);
192                addr.setDistrict(district);
193                addr.setRegion(region);
194                addr.setCountryCode("CZ");
195                addr.setPostCode(postcode);
196
197                m_address_places.add(addr);
198            }
199        }
200
201        // create address from SO data?
202        if (m_address_places.isEmpty() && house_number.length() > 0) {
203            Address addr = new Address();
204            addr.setRuianID(adr_id);
205
206            switch (house_number_typ) {
207                case "číslo popisné":
208                    addr.setConscriptionNumber(house_number);
209                    break;
210                case "číslo evidenční":
211                    addr.setProvisionalNumber(house_number);
212                    break;
213            }
214
215            addr.setStreetNumber(street_number);
216            addr.setStreet(street);
217            addr.setPlace(place);
218            addr.setSuburb(suburb);
219            addr.setCity(city);
220            addr.setDistrict(district);
221            addr.setRegion(region);
222            addr.setCountryCode("CZ");
223            addr.setPostCode(postcode);
224
225            m_address_places.add(addr);
226        }
227    }

239        }

309    @Override
310    public Map<String, String> getKeys(boolean m_alt) {
311
312        Map<String, String> tags = new HashMap<>();
313
314        if (!m_alt) {
315            if (getBuildingTagKey().equals("building")
316                    && getBuildingTagValue().length() > 0) {
317                tags.put("building", getBuildingTagValue());
318            } else {
319                tags.put("building", "yes");
320            }
321        }
322
323        if (getBuildingID() > 0) {
324            tags.put("ref:ruian:building", Long.toString(getBuildingID()));
325        }
326
327        if (getBuildingUsageCode().length() > 0) {
328            tags.put("building:ruian:type", getBuildingUsageCode());
329        }
330
331        if (getBuildingLevels().length() > 0) {
332            tags.put("building:levels", getBuildingLevels());
333        }
334
335        if (getBuildingFlats().length() > 0) {
336            tags.put("building:flats", getBuildingFlats());
337        }
338
339        if (getBuildingFinished().length() > 0) {
340            tags.put("start_date", getBuildingFinished());
341        }
342
343        if (getSource().length() > 0) {
344            tags.put("source", getSource());
345        } else {
346            tags.put("source", "cuzk:ruian");
347        }
348
349        return tags;
350    }
351
352    @Override
353    public boolean hasData() {
354        return this.getBuildingID() > 0 && super.hasOuter();
355    }

146    class ReuseLanduseNearNodes implements IReuseNearNodePredicate {
147
148        // distance tolerancies are in meters
149        private final double m_reuseNearNodesToleranceDefault = m_connectTolerance.distanceMeters();
150        private final double m_reuseNearNodesToleranceRetracedNodes = 0.40;
151
152        private final double m_lookupDistanceMeters;
153
154        private final Set<EdNode> m_retracedNodes;
155
156        ReuseLanduseNearNodes (Set<EdNode> retraced_nodes) {
157            m_retracedNodes = retraced_nodes;
158            m_lookupDistanceMeters = Math.max(m_reuseNearNodesToleranceDefault, m_reuseNearNodesToleranceRetracedNodes);
159        }
160
161        @Override
162        public ReuseNearNodeMethod reuseNearNode(EdNode node, EdNode near_node, double distance_meters) {
163
164            boolean retraced = m_retracedNodes != null && m_retracedNodes.contains(near_node);
165
166            // be more tolerant for untagged nodes occurring in retraced ways, feel free to move them
167            if (retraced) {
168                System.out.println("RNN: retraced, dist=" + Double.toString(distance_meters));
169                if (distance_meters <= m_reuseNearNodesToleranceRetracedNodes)
170                    if (!near_node.isTagged())
171                        return ReuseNearNodeMethod.moveAndReuseNode;
172            }
173
174            // use default tolerance for others, don't move them, just reuse
175            System.out.println("RNN: default, dist=" + Double.toString(distance_meters));
176            if (distance_meters <= m_reuseNearNodesToleranceDefault)
177                return ReuseNearNodeMethod.reuseNode;
178
179            return ReuseNearNodeMethod.dontReuseNode;
180        }
181
182        @Override
183        public double lookupDistanceMeters() {
184            return m_lookupDistanceMeters;
185        }
186    }

211        @Override
212        protected EdObject createTracedPolygonImpl(WayEditor editor) {
213
214            System.out.println("  LPIS ID: " + record().getLpisID());
215            System.out.println("  LPIS usage: " + record().getUsage());
216
217            highlightHelper.clear();
218
219            // Look for object to retrace
220            EdObject retrace_object = null;
221            if (m_performRetrace) {
222                Pair<EdObject, Boolean> repl = getObjectToRetrace(editor, m_pos);
223                retrace_object = repl.a;
224                boolean ambiguous_retrace = repl.b;
225
226                if (ambiguous_retrace) {
227                    postTraceNotifications().add(tr("Multiple existing LPIS polygons found, retrace is not possible."));
228                    return null;
229                }
230            }
231
232            // Only update tags, do not change geometry of the object
233            if (m_updateTagsOnly) {
234
235                // To update tags only we need an existing object to update
236                if (retrace_object == null) {
237                    postTraceNotifications().add(tr("No existing LPIS polygon found, tags only update is not possible."));
238                    return null;
239                }
240
241//                 // TODO - allows update of Multipolygons tags
242//                 if (!retrace_object.isWay() || retrace_object.hasReferrers()) {
243//                     postTraceNotifications().add(tr("Multipolygon retrace is not supported yet."));
244//                     return null;
245//                 }
246
247                // Tag object
248                if (!tagTracedObject(retrace_object))
249                    return null;
250
251                return retrace_object;
252            }
253
254            // Update object geometry as well
255            // Create traced object
256            EdObject trobj = record().createObject(editor);
257
258            // Everything is inside DataSource bounds?
259            if (!checkInsideDataSourceBounds(trobj, retrace_object)) {
260                wayIsOutsideDownloadedAreaDialog();
261                return null;
262            }
263
264            // Connect nodes to near landuse nodes
265            // (must be done before retrace updates, we want to use as much old nodes as possible)
266            if (!m_performClipping) {
267                reuseExistingNodes(trobj);
268            }
269            else {
270                reuseNearNodes(trobj, retrace_object);
271            }
272
273            // Update geometries of retraced object
274            if (retrace_object != null) {
275                RetraceUpdater retr = new RetraceUpdater(false, postTraceNotifications());
276                trobj = retr.updateRetracedObjects(trobj, retrace_object);
277                if (trobj == null)
278                    return null;
279            }
280
281            // Tag object
282            if (!tagTracedObject(trobj))
283                return null;
284
285            // Connect to touching nodes of near landuse polygons
286            connectExistingTouchingNodes(trobj);
287
288            // Clip other areas
289            if (m_performClipping) {
290                // #### Now, it clips using only the outer way. Consider if multipolygon clip is necessary/useful.
291                AreaPredicate filter = new AreaPredicate (m_clipLanduseWayMatch);
292                ClipAreas clip = new ClipAreas(editor, m_clipSettings, postTraceNotifications());
293                clip.clipAreas(getOuterWay(trobj), filter);
294
295                // Remove needless nodes
296                AreaPredicate remove_filter = new AreaPredicate (m_clipLanduseWayMatch);
297                BBox remove_bbox = trobj.getBBox();
298                BBoxUtils.extendBBox(remove_bbox, LatLonSize.get(remove_bbox, oversizeInDataBoundsMeters));
299                RemoveNeedlessNodes remover = new RemoveNeedlessNodes(remove_filter, m_removeNeedlesNodesTolerance, (Math.PI*2)/3, remove_bbox);
300                remover.removeNeedlessNodes(editor.getModifiedWays());
301
302                // Verify that clipping did not created more ways with equal ref
303
304                // First collect all refs and linked ways
305                HashMap<String, List<EdWay>> refsMap = new HashMap<String, List<EdWay>>();
306                for (EdWay w: editor.getModifiedWays()) {
307                    if (w.isTagged() && w.hasKey("ref")) { // Add simple ways
308//                         System.out.println("W(ref): "+w.get("ref")+" [id: "+Long.toString(w.getUniqueId())+"]");
309
310                        String kr = w.get("ref");
311                        if (refsMap.containsKey(kr)) {
312                            List<EdWay> arr = refsMap.get(kr);
313                            arr.add(w);
314                            refsMap.put(kr, arr);
315                        } else {
316                            List<EdWay> arr = new ArrayList<EdWay>();
317                            arr.add(w);
318                            refsMap.put(kr, arr);
319                        }
320
321                    } else { // Add ways that are outer in multipolygons
322                        List<EdMultipolygon> mps = w.getEditorReferrers(EdMultipolygon.class);
323                        if (mps.size() == 0)
324                            continue;
325                        else {
326                            for (EdMultipolygon mp: mps) {
327                                if (mp.isTagged() && mp.hasKey("ref") && mp.containsOuterWay(w)) {
328                                    String kr = mp.get("ref");
329                                    if (refsMap.containsKey(kr)) {
330                                        List<EdWay> arr = refsMap.get(kr);
331                                        arr.add(w);
332                                        refsMap.put(kr, arr);
333                                    } else {
334                                        List<EdWay> arr = new ArrayList<EdWay>();
335                                        arr.add(w);
336                                        refsMap.put(kr, arr);
337                                    }
338                                }
339                            }
340                        }
341                    }
342                }
343
344                // Check the generated list, verify all refs with multiple ways
345                for (String i : refsMap.keySet()) {
346                    System.out.println("ref: " + i + " has " + refsMap.get(i).size() + " members.");
347                    if (refsMap.get(i).size() > 1) { // More than one way
348                        for (EdWay w : refsMap.get(i)) {
349                            if (w.getUniqueId() < 0) { // A new way
350                                postTraceNotifications().clear();
351                                postTraceNotifications().add(tr("Retrace will split a way to multiple ways with the same ref key [{0}].\nFix the issue first please.", i));
352                                // Highlight
353                                for (EdWay wo: editor.getModifiedWays()) {
354                                    if (w.isTagged()) { // Simple Way
355                                        if (wo.get("ref") == i && wo.hasOriginal()) {
356                                            editor.getDataSet().setSelected(wo.originalWay()); // Select way
357                                            if (highlightHelper.highlightOnly(wo.originalWay())) { // Highlight selection
358                                                MainApplication.getMap().mapView.repaint();
359                                                // Zoom to selection
360                                                GuiHelper.executeByMainWorkerInEDT(() -> {
361                                                    BoundingXYVisitor bbox1 = new BoundingXYVisitor();
362                                                    bbox1.visit(wo.originalWay());
363                                                    MainApplication.getMap().mapView.zoomTo(bbox1);
364                                                });
365                                            }
366                                        }
367                                    } else { // Outer way in Multipolygon
368                                        for (EdMultipolygon mp: w.getEditorReferrers(EdMultipolygon.class)) {
369                                            if (mp.isTagged() && mp.hasKey("ref") && mp.get("ref") == i && mp.containsOuterWay(w)) {
370                                                editor.getDataSet().setSelected(mp.originalMultipolygon()); // Select multipolygon
371                                                if (highlightHelper.highlightOnly(mp.originalMultipolygon())) { // Highlight selection
372                                                    MainApplication.getMap().mapView.repaint();
373                                                    // Zoom to selection
374                                                    GuiHelper.executeByMainWorkerInEDT(() -> {
375                                                        BoundingXYVisitor bbox1 = new BoundingXYVisitor();
376                                                        bbox1.visit(mp.originalMultipolygon());
377                                                        MainApplication.getMap().mapView.zoomTo(bbox1);
378                                                    });
379                                                }
380                                            }
381                                        }
382                                    }
383                                }
384                                return null;
385                            }
386                        }
387                    } else {
388                        System.out.println("Id of ref:"+i+" is "+Long.toString(refsMap.get(i).get(0).getUniqueId()));
389                        Double area = refsMap.get(i).get(0).getEastNorthArea();
390                        if (area != null)
391                            System.out.println("Area of "+i+" is "+Double.toString(area));
392                    }
393                }
394
395            }
396
397
398            // Merge duplicate ways
399            // (Note that outer way can be merged to another way too, so we must watch it.
400            // Otherwise, trobj variable would refer to an unused way.)
401            if (m_performWayMerging) {
402                AreaPredicate merge_filter = new AreaPredicate (m_mergeLanduseWayMatch);
403                MergeIdenticalWays merger = new MergeIdenticalWays(editor, merge_filter);
404                EdWay outer_way = merger.mergeWays(editor.getModifiedWays(), true, getOuterWay(trobj));
405                if (trobj.isWay()) {
406                    trobj = outer_way;
407                }
408            }
409
410            return trobj;
411        }

413        private boolean tagTracedObject (EdObject obj) {
414
415            Map <String, String> old_keys = obj.getKeys();
416            Map <String, String> new_keys = new HashMap <> (getRecord().getKeys());
417
418            // Silently replace lpis ref
419            if (old_keys.containsKey("source") &&
420                old_keys.containsKey("ref") &&
421                old_keys.get("source").equals("lpis") &&
422               !old_keys.get("ref").equals(new_keys.get("ref"))) {
423
424                old_keys.put("ref", new_keys.get("ref"));
425            }
426
427            // Silently switch between farmland and meadow
428            if (old_keys.containsKey("landuse") && new_keys.containsKey("landuse") &&
429                !old_keys.get("landuse").equals(new_keys.get("landuse"))) {
430
431                // From meadow to farmland
432                if (old_keys.get("landuse").equals("meadow") &&
433                    new_keys.get("landuse").equals("farmland")) {
434
435                    old_keys.put("landuse", new_keys.get("landuse"));
436                    if (old_keys.containsKey("meadow")) {
437                        old_keys.remove("meadow");
438                    }
439                }
440
441                // From farmland to meadow
442                if (old_keys.get("landuse").equals("farmland") &&
443                    new_keys.get("landuse").equals("meadow")) {
444
445                    old_keys.put("landuse", new_keys.get("landuse"));
446                    if (old_keys.containsKey("crop")) {
447                        old_keys.remove("crop");
448                    }
449                }
450            }
451
452            // Silently replace village_green
453            if (old_keys.containsKey("landuse") &&
454                new_keys.containsKey("landuse") &&
455                old_keys.get("landuse").equals("village_green")) {
456
457                old_keys.put("landuse", new_keys.get("landuse"));
458            }
459
460            // Allow only type=multipolygon key
461            if (old_keys.containsKey("type") &&
462                !old_keys.get("type").equals("multipolygon")) {
463                old_keys.remove("type");
464            }
465
466            // merge missing keys to avoid resolution dialog when there're no collisions
467            for (Map.Entry<String, String> tag: old_keys.entrySet()) {
468                if (!new_keys.containsKey(tag.getKey()))
469                    new_keys.put(tag.getKey(), tag.getValue());
470            }
471            for (Map.Entry<String, String> tag: new_keys.entrySet()) {
472                if (!old_keys.containsKey(tag.getKey()))
473                    old_keys.put(tag.getKey(), tag.getValue());
474            }
475
476            // combine and resolve conflicting keys
477            Map<String, String> result = CombineTagsResolver.launchIfNecessary(old_keys, new_keys);
478            if (result == null)
479                return false;
480
481            obj.setKeys(result);
482            return true;
483        }

485        private Pair<EdObject, Boolean> getObjectToRetrace(WayEditor editor, LatLon pos) {
486            AreaPredicate filter = new AreaPredicate(m_retraceAreaMatch);
487            Set<EdObject> areas = editor.useNonEditedAreasContainingPoint(pos, filter);
488
489            String lpisref = Long.toString(record().getLpisID());
490
491            // restrict to LPIS areas only, yet ... #### improve in the future
492            boolean multiple_areas = false;
493            EdObject lpis_area = null;
494            for (EdObject area: areas) {
495//                 String source = area.get("source");
496//                 if (source == null || !source.equals("lpis"))
497//                     continue;
498
499                if (area.isWay())
500                    System.out.println("Retrace candidate EdWay: " + Long.toString(area.getUniqueId()));
501                else if (area.isMultipolygon())
502                    System.out.println("Retrace candidate EdMultipolygon: " + Long.toString(area.getUniqueId()));
503
504                String ref = area.get("ref");
505                if (ref != null && ref.equals(lpisref)) // exact match ;)
506                    return new Pair<>(area, false);
507
508                if (lpis_area == null)
509                    lpis_area = area;
510                else
511                    multiple_areas = true;
512            }
513
514            if (multiple_areas) {
515                return new Pair<>(null, true);
516            }
517
518            if (lpis_area != null) {
519                return new Pair<>(lpis_area, false);
520            }
521
522            return new Pair<>(null, false);
523        }

529        private void reuseExistingNodes(EdObject obj) {
530            obj.reuseExistingNodes (reuseExistingNodesFilter(obj));
531        }

533        private void reuseNearNodes(EdObject obj, EdObject retrace_object) {
534            Set<EdNode> retrace_nodes = null;
535            if (retrace_object != null)
536                retrace_nodes = retrace_object.getAllNodes();
537            obj.reuseNearNodes (new ReuseLanduseNearNodes(retrace_nodes), reuseExistingNodesFilter(obj));
538        }
539
540        private IEdNodePredicate reuseExistingNodesFilter(EdObject obj) {
541            // Setup filters - include landuse nodes only, exclude all nodes of the object itself
542            IEdNodePredicate nodes_filter = new AreaBoundaryWayNodePredicate(m_reuseExistingLanduseNodeMatch);
543            IEdNodePredicate exclude_my_nodes = new ExcludeEdNodesPredicate(obj);
544            return new EdNodeLogicalAndPredicate (exclude_my_nodes, nodes_filter);
545        }

547        private boolean checkInsideDataSourceBounds(EdObject new_object, EdObject retrace_object) {
548            LatLonSize bounds_oversize = LatLonSize.get(new_object.getBBox(), oversizeInDataBoundsMeters);
549            if (retrace_object != null && !retrace_object.isInsideDataSourceBounds(bounds_oversize))
550                return false;
551            return new_object.isInsideDataSourceBounds(bounds_oversize);
552        }

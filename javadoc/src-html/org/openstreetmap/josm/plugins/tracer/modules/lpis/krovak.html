039public class krovak {
040
041  private xyCoor xy;
042  private LatLon ll;
043
044  private CoordinateReferenceSystem LatLonCRS;
045  private CoordinateReferenceSystem krovakCRS;
046
047
048  /* Define Krovak projection */
049  private String wkt = "PROJCS[\"S-JTSK / Krovak East North\","+
050      "GEOGCS[\"S-JTSK\","+
051      "  DATUM[\"System_Jednotne_Trigonometricke_Site_Katastralni\","+
052      "    SPHEROID[\"Bessel 1841\",6377397.155,299.1528128,AUTHORITY[\"EPSG\",\"7004\"]],"+
053      "    TOWGS84[570.8,85.7,462.8,4.998,1.587,5.261,3.56],AUTHORITY[\"EPSG\",\"6156\"]],"+
054      "  PRIMEM[\"Greenwich\",0,AUTHORITY[\"EPSG\",\"8901\"]],"+
055      "  UNIT[\"degree\",0.0174532925199433,AUTHORITY[\"EPSG\",\"9122\"]],"+

070  public krovak () {
071    init();
072  }
073
074  private void init () {
075    xy = new xyCoor();
076    ll = new LatLon(0,0);
077
078    try {
079      LatLonCRS = DefaultGeographicCRS.WGS84;
080      krovakCRS = CRS.parseWKT(wkt);
081
082    } catch (Exception e) {
083      System.out.println("krovak init() exception: " + e.getMessage());
084    }
085  }
086
087  public void setXY (Double x, Double y) {
088    xy = new xyCoor(x, y);
089  }
090
091  public void setLatLon (LatLon pll) {
092    ll = pll;
093  }
094
095  public LatLon getLatLon () {
096
097    LatLon ll = new LatLon(0,0);
098
099    try {
100      org.locationtech.jts.geom.GeometryFactory gf = new org.locationtech.jts.geom.GeometryFactory();
101      org.locationtech.jts.geom.Coordinate c = new org.locationtech.jts.geom.Coordinate(xy.x(), xy.y());
102
103      org.locationtech.jts.geom.Point p = gf.createPoint(c);
104
105      MathTransform mathTransform = CRS.findMathTransform(krovakCRS, LatLonCRS, false);
106      org.locationtech.jts.geom.Point p1 = (org.locationtech.jts.geom.Point) JTS.transform(p, mathTransform);
107
108//       System.out.println(p1.getCoordinate());
109      ll = new LatLon(LatLon.roundToOsmPrecision(p1.getY()),
110                      LatLon.roundToOsmPrecision(p1.getX()));
111    } catch (Exception e) {
112      System.out.println("CRS conversion exception: " + e.getMessage());
113    }
114
115    return ll;
116  }
117
118  public xyCoor getXY () {
119
120    xyCoor xy = new xyCoor();
121
122    try {
123      org.locationtech.jts.geom.GeometryFactory gf = new org.locationtech.jts.geom.GeometryFactory();
124      org.locationtech.jts.geom.Coordinate c = new org.locationtech.jts.geom.Coordinate(ll.lon(), ll.lat());
125
126      org.locationtech.jts.geom.Point p = gf.createPoint(c);
127
128      MathTransform mathTransform = CRS.findMathTransform(LatLonCRS, krovakCRS, false);
129      org.locationtech.jts.geom.Point p1 = (org.locationtech.jts.geom.Point) JTS.transform(p, mathTransform);
130
131//       System.out.println(p1.getCoordinate());
132      xy = new xyCoor(p1.getX(), p1.getY());
133    } catch (Exception e) {
134      System.out.println("CRS conversion exception: " + e.getMessage());
135    }
136    return xy;
137  }
138
139  /**
140    * Convert coordinates from krovak to LatLon
141    * @param x - the X coordinate
142    * @param y - the Y coordinate
143    * @return LatLon coordinates
144    */
145  public LatLon krovak2LatLon (String x, String y) {
146    setXY(Double.parseDouble(x),Double.parseDouble(y));
147    return getLatLon();
148  }
149
150  /**
151    * Convert coordinates from krovak to LatLon
152    * @param x - the X coordinate
153    * @param y - the Y coordinate
154    * @return LatLon coordinates
155    */
156  public LatLon krovak2LatLon (Double x, Double y) {
157    setXY(x,y);
158    return getLatLon();
159  }
160
161  /**
162    * Convert coordinates from LatLon to krovak
163    * @param ll - LatLon coordinates
164    * @return Krovak coordinates
165    */
166  public xyCoor LatLon2krovak (LatLon ll) {
167    setLatLon(ll);
168    return getXY();
169  }
170}

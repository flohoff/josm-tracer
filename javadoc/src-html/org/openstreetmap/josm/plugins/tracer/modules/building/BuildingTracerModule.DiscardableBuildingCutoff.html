095    class DiscardableBuildingCutoff implements IDiscardableCutoffPredicate {
096
097        @Override
098        public boolean canSilentlyDiscard(EdWay way, double cutoffs_percent) {
099
100            if (cutoffs_percent > m_discardCutoffsPercentMaxSum)
101                return false;
102
103            // can silently discard given building as a result of clipping/cutoff removal?
104            Map<String, String> keys = way.getInterestingKeys();
105            for (Map.Entry<String, String> entry: keys.entrySet()) {
106                String k = entry.getKey();
107                String v = entry.getValue();
108                if (k != null && v != null && !(k.equals("building") && v.equals("yes")))
109                    return false;
110            }
111            return true;
112        }
113    }

115    class ReuseBuildingNearNodes implements IReuseNearNodePredicate {
116
117        // distance tolerancies are in meters
118        private final double m_reuseNearNodesToleranceDefault = m_connectTolerance.distanceMeters();
119        private final double m_reuseNearNodesToleranceNonRuian = 0.30;
120        private final double m_reuseNearNodesToleranceRetracedNodes = 0.40;
121
122        private final double m_lookupDistanceMeters;
123        private final ReuseNearNodeMethod m_reuseMethod = ReuseNearNodeMethod.moveAndReuseNode;
124        private final AreaBoundaryWayNodePredicate m_ruianArea = new AreaBoundaryWayNodePredicate(m_ruianSourceMatch);
125
126        private final Set<EdNode> m_retracedNodes;
127
128        ReuseBuildingNearNodes (Set<EdNode> retraced_nodes) {
129            m_retracedNodes = retraced_nodes;
130            m_lookupDistanceMeters = Math.max (Math.max(m_reuseNearNodesToleranceDefault, m_reuseNearNodesToleranceRetracedNodes), m_reuseNearNodesToleranceNonRuian);
131        }
132
133        @Override
134        public ReuseNearNodeMethod reuseNearNode(EdNode node, EdNode near_node, double distance_meters) {
135
136            boolean retraced = m_retracedNodes != null && m_retracedNodes.contains(near_node);
137            boolean ruian = m_ruianArea.evaluate(near_node);
138
139            // be more tolerant for nodes occurring in retraced building
140            if (retraced && !ruian) {
141                System.out.println("RNN: retraced, dist=" + Double.toString(distance_meters));
142                if (distance_meters <= m_reuseNearNodesToleranceRetracedNodes)
143                    return m_reuseMethod;
144            }
145
146            // be more tolerant for non-ruian buildings
147            if (!ruian) {
148                System.out.println("RNN: non-ruian, dist=" + Double.toString(distance_meters));
149                if (distance_meters <= m_reuseNearNodesToleranceNonRuian)
150                    return m_reuseMethod;
151            }
152
153            // use default tolerance for others
154            System.out.println("RNN: default, dist=" + Double.toString(distance_meters));
155            if (distance_meters <= m_reuseNearNodesToleranceDefault)
156                return m_reuseMethod;
157
158            return ReuseNearNodeMethod.dontReuseNode;
159        }
160
161        @Override
162        public double lookupDistanceMeters() {
163            return m_lookupDistanceMeters;
164        }
165    }

171
172        public BuildingTracerTask (LatLon pos, boolean ctrl, boolean alt, boolean shift) {
173            super (pos, ctrl, alt ,shift);
174
175            this.m_performNearBuildingsEdit = !m_ctrl;
176        }

178        @Override
179        protected EdObject createTracedPolygonImpl(WayEditor editor) {
180
181            System.out.println("  " + getName() + " keys: " + getRecord().getKeys(m_alt));
182
183            // Look for object to retrace
184            EdObject retrace_object = null;
185            if (m_performRetrace) {
186                Pair<EdObject, Boolean> repl = getObjectToRetrace(editor, m_pos, m_retraceAreaMatch);
187                retrace_object = repl.a;
188                boolean ambiguous_retrace = repl.b;
189
190                if (ambiguous_retrace) {
191                    postTraceNotifications().add(tr("Multiple existing Ruian building polygons found, retrace is not possible."));
192                    return null;
193                }
194            }
195
196            // Only update tags, do not change geometry of the object
197            if (m_updateTagsOnly) {
198
199                // To update tags only we need an existing object to update
200                if (retrace_object == null) {
201                    postTraceNotifications().add(tr("No existing Ruian building polygon found, tags only update is not possible."));
202                    return null;
203                }
204
205                // Tag object
206                if (!tagTracedObject(retrace_object))
207                    return null;
208
209                return retrace_object;
210            }
211
212            // Update object geometry as well
213            // Create traced object
214            EdObject trobj = getRecord().createObject(editor);
215
216            // Everything is inside DataSource bounds?
217            if (!checkInsideDataSourceBounds(trobj, retrace_object)) {
218                wayIsOutsideDownloadedAreaDialog();
219                return null;
220            }
221
222            // Connect to near building polygons
223            // (must be done before retrace updates, we want to use as much old nodes as possible)
224            if (!m_performNearBuildingsEdit) {
225                reuseExistingNodes(trobj);
226            }
227            else {
228                reuseNearNodes(trobj, retrace_object);
229            }
230
231            // Update geometries of retraced object
232            if (retrace_object != null) {
233                RetraceUpdater retr = new RetraceUpdater(true, postTraceNotifications());
234                trobj = retr.updateRetracedObjects(trobj, retrace_object);
235                if (trobj == null)
236                    return null;
237            }
238
239            // Connect to touching nodes of near buildings
240            // (must be done after retrace updates, we don't want to connect to the old polygon)
241            if (m_performNearBuildingsEdit) {
242                connectExistingTouchingNodes(trobj);
243            }
244
245            // Tag object
246            if (!tagTracedObject(trobj))
247                return null;
248
249            // Clip other areas
250            if (m_performClipping) {
251                // #### Now, it clips using only the outer way. Consider if multipolygon clip is necessary/useful.
252                AreaPredicate filter = new AreaPredicate (m_clipBuildingWayMatch);
253                ClipAreas clip = new ClipAreas(editor, m_clipSettings, postTraceNotifications());
254                clip.clipAreas(getOuterWay(trobj), filter);
255
256                // Remove needless nodes
257                AreaPredicate remove_filter = new AreaPredicate (m_clipBuildingWayMatch);
258                RemoveNeedlessNodes remover = new RemoveNeedlessNodes(remove_filter, m_removeNeedlesNodesTolerance, (Math.PI*2)/3);
259                remover.removeNeedlessNodes(editor.getModifiedWays());
260            }
261
262            // Merge duplicate ways
263            // (Note that outer way can be merged to another way too, so we must watch it.
264            // Otherwise, trobj variable would refer to an unused way.)
265            if (m_performWayMerging) {
266                AreaPredicate merge_filter = new AreaPredicate (m_mergeBuildingWayMatch);
267                MergeIdenticalWays merger = new MergeIdenticalWays(editor, merge_filter);
268                EdWay outer_way = merger.mergeWays(editor.getModifiedWays(), true, getOuterWay(trobj));
269                if (trobj.isWay()) {
270                    trobj = outer_way;
271                }
272            }
273
274            return trobj;
275        }

277        private boolean tagTracedObject (EdObject obj) {
278
279            Map <String, String> old_keys = obj.getKeys();
280            Map <String, String> new_keys = new HashMap <> (getRecord().getKeys(m_alt));
281
282            String old_building_tag = old_keys.get("building");
283            String new_building_tag = new_keys.get("building");
284
285            // always silently replace building=yes with the new building tag
286            if (new_building_tag != null && "yes".equals(old_building_tag))
287                old_keys.put("building", new_building_tag);
288            // always silently replace building=construction with the new building tag
289            if (new_building_tag != null && "construction".equals(old_building_tag))
290                old_keys.put("building", new_building_tag);
291            // always keep old building tag if the new tag is building=yes
292            else if (old_building_tag != null && "yes".equals(new_building_tag))
293                new_keys.put("building", old_building_tag);
294            // keep church/chapel tag if the new one is civic
295            else if ("civic".equals(new_building_tag) &&
296                      ("church".equals(old_building_tag) ||
297                       "chapel".equals(old_building_tag) ||
298                       "cathedral".equals(old_building_tag) ||
299                       "synagogue".equals(old_building_tag) ||
300                       "university".equals(old_building_tag) ||
301                       "college".equals(old_building_tag) ||
302                       "school".equals(old_building_tag) ||
303                       "kindergarten".equals(old_building_tag) ||
304                       "train_station".equals(old_building_tag) ||
305                       "hotel".equals(old_building_tag) ||
306                       "apartments".equals(old_building_tag) ||
307                       "transportation".equals(old_building_tag) ||
308                       "hospital".equals(old_building_tag) ||
309                       "hall".equals(old_building_tag) ||
310                       "public".equals(old_building_tag)
311                      )
312                    )
313                new_keys.put("building", old_building_tag);
314            // keep train_station
315            else if ("transportation".equals(new_building_tag) && "train_station".equals(old_building_tag))
316                new_keys.put("building", old_building_tag);
317            // keep apartments, detached, house, terrace tag if the new one is residential
318            else if ("residential".equals(new_building_tag) &&
319                      ("apartments".equals(old_building_tag) ||
320                       "detached".equals(old_building_tag) ||
321                       "house".equals(old_building_tag) ||
322                       "terrace".equals(old_building_tag)
323                      )
324                    )
325                new_keys.put("building", old_building_tag);
326            // replace residential if new tag is house
327            else if ("house".equals(new_building_tag) && "residential".equals(old_building_tag)) {
328                new_keys.put("building", old_building_tag);
329                // also replace building:ruian:type
330                if (old_keys.containsKey("building:ruian:type") &&
331                    new_keys.containsKey("building:ruian:type") &&
332                    (old_keys.get("building:ruian:type")).equals("3") &&
333                     new_keys.get("building:ruian:type").equals("7"))
334                    old_keys.put("building:ruian:type", new_keys.get("building:ruian:type"));
335                }
336            // ... add other known conflicting building tags that should be resolved automatically
337
338            // silently discard old source=cuzk:km or source=ruian or source=bing tags
339            if (new_keys.containsKey("source") && old_keys.containsKey("source") &&
340                    (old_keys.get("source").startsWith("cuzk") ||
341                    old_keys.get("source").equals("ruian") ||
342                    old_keys.get("source").equals("bing")||
343                    old_keys.get("source").startsWith("uhul")||
344                    old_keys.get("source").startsWith("orto")
345                    ))
346                old_keys.put("source", new_keys.get("source"));
347
348            // silently replace wrong ruian date format
349            if (old_keys.containsKey("start_date") &&
350                new_keys.containsKey("start_date") &&
351                (TracerUtils.convertDate(old_keys.get("start_date")).equals(new_keys.get("start_date")) ||
352                 old_keys.get("start_date").startsWith(new_keys.get("start_date"))))
353                old_keys.put("start_date", new_keys.get("start_date"));
354
355            // silently replace building:levels=0
356            if (old_keys.containsKey("building:levels") &&
357                new_keys.containsKey("building:levels") &&
358                (old_keys.get("building:levels")).equals("0") &&
359                 !new_keys.get("building:levels").equals("0"))
360                old_keys.put("building:levels", new_keys.get("building:levels"));
361
362            // silently replace building:flats=0
363            if (old_keys.containsKey("building:flats") &&
364                new_keys.containsKey("building:flats") &&
365                (old_keys.get("building:flats")).equals("0") &&
366                 !new_keys.get("building:flats").equals("0"))
367                old_keys.put("building:flats", new_keys.get("building:flats"));
368
369
370            // merge missing keys to avoid resolution dialog when there're no collisions
371            for (Map.Entry<String, String> tag: old_keys.entrySet()) {
372                if (!new_keys.containsKey(tag.getKey()))
373                    new_keys.put(tag.getKey(), tag.getValue());
374            }
375            for (Map.Entry<String, String> tag: new_keys.entrySet()) {
376                if (!old_keys.containsKey(tag.getKey()))
377                    old_keys.put(tag.getKey(), tag.getValue());
378            }
379
380            // combine and resolve conflicting keys
381            Map<String, String> result = CombineTagsResolver.launchIfNecessary(old_keys, new_keys);
382            if (result == null)
383                return false;
384
385            obj.setKeys(result);
386            return true;
387        }

389        private Pair<EdObject, Boolean> getObjectToRetrace(WayEditor editor, LatLon pos, SearchCompiler.Match retraceAreaMatch) {
390            AreaPredicate filter = new AreaPredicate(retraceAreaMatch);
391            Set<EdObject> areas = editor.useNonEditedAreasContainingPoint(pos, filter);
392
393            // look for exact RUIAN ID in case of RUIAN buildings
394            String ruianref = null;
395            if (getRecord() instanceof RuianRecord) {
396                ruianref = Long.toString(((RuianRecord)getRecord()).getBuildingID());
397            }
398
399            boolean multiple_areas = false;
400            EdObject building_area = null;
401            for (EdObject area: areas) {
402                if (area.isWay())
403                    System.out.println("Retrace candidate EdWay: " + Long.toString(area.getUniqueId()));
404                else if (area.isMultipolygon())
405                    System.out.println("Retrace candidate EdMultipolygon: " + Long.toString(area.getUniqueId()));
406
407                String ref = area.get("ref:ruian:building");
408                if (ref != null && ref.equals(ruianref)) // exact match ;)
409                    return new Pair<>(area, false);
410
411                if (building_area == null)
412                    building_area = area;
413                else
414                    multiple_areas = true;
415            }
416
417            if (multiple_areas) {
418                return new Pair<>(null, true);
419            }
420
421            if (building_area != null) {
422                return new Pair<>(building_area, false);
423            }
424
425            return new Pair<>(null, false);
426        }

428        private void connectExistingTouchingNodes(EdObject obj) {
429            IEdNodePredicate filter = reuseExistingNodesFilter(obj);
430            obj.connectExistingTouchingNodes(m_connectTolerance, filter);
431        }
432
433        private void reuseExistingNodes(EdObject obj) {
434            obj.reuseExistingNodes (reuseExistingNodesFilter(obj));
435        }
436
437        private void reuseNearNodes(EdObject obj, EdObject retrace_object) {
438            Set<EdNode> retrace_nodes = null;
439            if (retrace_object != null)
440                retrace_nodes = retrace_object.getAllNodes();
441            obj.reuseNearNodes (new ReuseBuildingNearNodes(retrace_nodes), reuseExistingNodesFilter(obj));
442        }
443
444        private IEdNodePredicate reuseExistingNodesFilter(EdObject obj) {
445            // Setup filtes - include building nodes only, exclude all nodes of the object itself
446            IEdNodePredicate nodes_filter = new AreaBoundaryWayNodePredicate(m_reuseExistingBuildingNodeMatch);
447            IEdNodePredicate exclude_my_nodes = new ExcludeEdNodesPredicate(obj);
448            return new EdNodeLogicalAndPredicate (exclude_my_nodes, nodes_filter);
449        }
450
451        private boolean checkInsideDataSourceBounds(EdObject new_object, EdObject retrace_object) {
452            LatLonSize bounds_oversize = LatLonSize.get(new_object.getBBox(), oversizeInDataBoundsMeters);
453            if (retrace_object != null && !retrace_object.isInsideDataSourceBounds(bounds_oversize))
454                return false;
455            return new_object.isInsideDataSourceBounds(bounds_oversize);
456        }

460            return LatLonSize.get(pos, 3 * oversizeInDataBoundsMeters);
461        }

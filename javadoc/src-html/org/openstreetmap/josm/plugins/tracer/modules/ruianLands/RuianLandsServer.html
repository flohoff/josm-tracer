020package org.openstreetmap.josm.plugins.tracer.modules.ruianLands;
021
022import java.io.BufferedReader;
023import java.io.IOException;
024import java.net.MalformedURLException;
025import org.openstreetmap.josm.data.coor.LatLon;
026import org.openstreetmap.josm.plugins.tracer.TracerUtils;
027
028
029public class RuianLandsServer {
030
031    public RuianLandsServer() {
032
033    }
034
035    /**
036     * Call Trace server.
037     * @param urlString Input parameters.
038     * @return Result text.
039     * @throws java.net.MalformedURLException Wrong url
040     * @throws java.io.IOException Input/Output issue
041     */
042    private String callServer(String urlString) throws MalformedURLException, IOException {
043        try (BufferedReader reader = TracerUtils.openUrlStream (urlString)) {
044            StringBuilder sb = new StringBuilder();
045            String line;
046            while ((line = reader.readLine()) != null) {
047                if (sb.length() > 0)
048                    sb.append(" ");
049                sb.append(line);
050            }
051            return sb.toString();
052        }
053    }
054
055    public RuianLandsRecord trace(LatLon pos, String url, double adjlat, double adjlon) throws IOException {
056        System.out.println("Request: "+ url + "/ruian-lands/beta/?lat=" + pos.lat() + "&lon=" + pos.lon());
057        String content = callServer(url + "/ruian-lands/beta/?lat=" + pos.lat() + "&lon=" + pos.lon());
058        System.out.println("Reply: " + content);
059        RuianLandsRecord ruian = new RuianLandsRecord(adjlat, adjlon);
060        ruian.parseJSON(content);
061        return ruian;
062    }
063}

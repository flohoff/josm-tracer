054enum TracerTaskStep {

116    public abstract class AbstractTracerTask {
117
118        TracerTaskStep m_taskStep;
119        private boolean m_cancelled;
120
121        protected final LatLon m_pos;
122        protected final boolean m_ctrl;
123        protected final boolean m_alt;
124        protected final boolean m_shift;
125
126        protected boolean m_performRetrace;
127        protected boolean m_performClipping;
128        protected boolean m_performWayMerging;
129        protected boolean m_updateTagsOnly;
130
131        private TracerRecord m_record;
132
133        private static final double resurrectNodesDistanceMeters = 10.0;
134        private static final double defaultAutomaticOsmDownloadMeters = 500.0;
135
136        private final PostTraceNotifications m_postTraceNotifications = new PostTraceNotifications();
137
138        protected AbstractTracerTask (LatLon pos, boolean ctrl, boolean alt, boolean shift) {
139            this.m_taskStep = TracerTaskStep.ttsInit;
140            this.m_cancelled = false;
141
142            this.m_pos = pos;
143            this.m_ctrl = ctrl;
144            this.m_alt = alt;
145            this.m_shift = shift;
146            this.m_record = null;
147
148            this.m_performClipping = !m_ctrl;
149            this.m_performRetrace = !m_ctrl;
150            this.m_performWayMerging = !m_ctrl;
151            this.m_updateTagsOnly=m_shift;
152        }
153
154        public void run () {
155            if (m_taskStep != TracerTaskStep.ttsInit)
156                throw new AssertionError("Tracer task already in progress");
157            nextStep ();
158        }
159
160        private void nextStep () {
161
162            if (m_cancelled)
163                return;
164
165            switch (m_taskStep) {
166                case ttsInit:
167                    stepDownloadRecord ();
168                    break;
169                case ttsDownloadRecord:
170                    stepDownloadMissingArea ();
171                    break;
172                case ttsDownloadMissingArea:
173                    stepDownloadIncompleteMultipolygons ();
174                    break;
175                case ttsDownloadIncompleteMultipolygons:
176                    stepCreateTracedPolygon ();
177                    break;
178                case ttsCreateTracedPolygon:
179                    throw new AssertionError ("Internal error, no next step available");
180                default:
181                    throw new AssertionError ("Unknown TracerTaskStep");
182            }
183        }
184
185        /**
186         * Returns ExecutorService to be used for background asynchronous download of a traced record.
187         * @return An ExecutorService instance if the download should run on background. Null if
188         * foreground PleaseWaitRunnable task should be used.
189         */
190        protected ExecutorService getDownloadRecordExecutor() {
191            return null;
192        }
193
194        private void downloadRecordTaskBody (boolean async) {
195            System.out.println("");
196            System.out.println("----- Trace -----");
197            System.out.println("");
198
199            try {
200                m_record = downloadRecord(m_pos);
201            }
202            catch (final Exception e) {
203                e.printStackTrace();
204                TracerUtils.showNotification(tr("{0} download failed ({1}).\nException: {2}", getName(), m_pos.toDisplayString(), e.getLocalizedMessage()), "error");
205                return;
206            }
207
208            if (!async) {
209                nextStep ();
210                return;
211            }
212
213            // Explicitly serialize subsequent tasks after asynchronous download.
214            // This reduces the chance that the same area/relation downloads are
215            // triggered multiple times.
216            MainApplication.worker.submit (new Runnable () {
217                @Override
218                public void run() {
219                    nextStep ();
220                }
221            });
222        }
223
224        private void stepDownloadRecord () {
225
226            // Initial step, always schedule download record task
227            m_taskStep = TracerTaskStep.ttsDownloadRecord;
228
229            ExecutorService exec = getDownloadRecordExecutor();
230
231            // run as background asynchronous task
232            if (exec != null) {
233                exec.submit(new Runnable() {
234                    @Override
235                    public void run() {
236                        downloadRecordTaskBody (true);
237                    }
238                });
239                return;
240            }
241
242            // run as blocking PleaseWaitRunnable window
243            MainApplication.worker.submit (new PleaseWaitRunnable (tr("Tracing")) {
244                    @Override
245                    protected void cancel() {
246                        m_cancelled = true;
247                    }
248
249                    @Override
250                    protected void realRun() throws SAXException, IOException, OsmTransferException {
251                        this.getProgressMonitor().indeterminateSubTask(tr("Downloading {0} data..." , getName()));
252                        downloadRecordTaskBody (false);
253                        this.getProgressMonitor().subTask(tr("Creating {0} polygon...", getName()));
254                    }
255
256                    @Override
257                    protected void finish() {}
258                });
259        }
260
261        private void stepDownloadMissingArea () {
262
263            // No data available?
264            if (m_record == null || !m_record.hasData()) {
265                TracerUtils.showNotification(tr("Data not available.")+ "\n(" + m_pos.toDisplayString() + ")", "warning");
266                return;
267            }
268
269            // Download missing area
270            m_taskStep = TracerTaskStep.ttsDownloadMissingArea;
271
272            LatLonSize downloadsize = LatLonSize.get (m_pos, this.getAutomaticOsmDownloadMeters ());
273            LatLonSize extrasize = this.getMissingAreaCheckExtraSize(m_pos);
274            Bounds area = m_record.getMissingAreaToDownload(MainApplication.getLayerManager().getEditDataSet(), extrasize, downloadsize);
275
276            // nothing to download? go ahead
277            if (area == null) {
278                nextStep ();
279                return;
280            }
281
282            // Schedule missing area download
283            final DownloadOsmTask task = new DownloadOsmMissingAreaTask();
284            final Future<?> future = task.download(new DownloadParams(), area, null);
285            // Note: we don't start PostDownloadHandler after download because we're
286            // not interested in download errors.
287
288            // Note: be careful, this download doesn't guarantee that the required area will be available.
289            // First, the download can fail at any time. Second, getMissingAreaToDownload() doesn't
290            // consider object retracing. That is, if the retraced object is significantly different
291            // from the traced one, it still can be outside of downloaded bounds. Because retrace object
292            // is selected inside WayEditor transaction, there's no easy way to get it now. Also, retraced
293            // object might be an incomplete multipolygon that is not downloaded yet. So, we ignore these
294            // corner cases here and all modules still must must carefully check if data bounds requirements
295            // are satisfied.
296            MainApplication.worker.submit(new Runnable() {
297                @Override
298                    public void run() {
299                        try {
300                            future.get();
301                            if (task.isCanceled())
302                                m_cancelled = true;
303                        } catch (Exception e) {}
304                        nextStep ();
305                    }
306                });
307        }
308
309        private void stepDownloadIncompleteMultipolygons () {
310
311            // Look for incomplete multipolygons that might participate in clipping
312            m_taskStep = TracerTaskStep.ttsDownloadIncompleteMultipolygons;
313
314            List<Relation> incomplete_multipolygons = null;
315            if (m_performClipping)
316                incomplete_multipolygons = getIncompleteMultipolygonsForDownload ();
317
318            // nothing to download? go ahead
319            if (incomplete_multipolygons == null || incomplete_multipolygons.isEmpty()) {
320                nextStep();
321                return;
322            }
323
324            // Schedule task to download incomplete multipolygons
325            final DownloadRelationTask task = new DownloadRelationTask(incomplete_multipolygons, MainApplication.getLayerManager().getEditLayer());
326            final Future<?> future = MainApplication.worker.submit(task);
327            MainApplication.worker.submit (new Runnable() {
328                @Override
329                    public void run() {
330                        try {
331                            future.get();
332                            // mmhm, DownloadRelationTask doesn't expose "cancelled" flag :-(
333                        } catch (Exception e) {}
334                        nextStep ();
335                    }
336                });
337        }
338
339        private void stepCreateTracedPolygon() {
340
341            m_taskStep = TracerTaskStep.ttsCreateTracedPolygon;
342
343            GuiHelper.runInEDT(new Runnable() {
344                @Override
345                @SuppressWarnings("CallToPrintStackTrace")
346                public void run() {
347                    long start_time = System.nanoTime();
348                    DataSet data_set = MainApplication.getLayerManager().getEditDataSet();
349                    data_set.beginUpdate();
350                    try {
351                        WayEditor editor = new WayEditor (data_set);
352                        EdObject object = createTracedPolygonImpl (editor);
353                        if (object != null) {
354                            finalizeEdit(editor, object);
355                        }
356                        postTraceNotifications().show();
357                    }
358                    catch (Exception e) {
359                        e.printStackTrace();
360                        throw e;
361                    }
362                    finally {
363                        data_set.endUpdate();
364                        long end_time = System.nanoTime();
365                        long time_msecs = (end_time - start_time) / (1000*1000);
366                        System.out.println("Polygon time (ms): " + Long.toString(time_msecs));
367                    }
368                }
369
370                private void finalizeEdit(WayEditor editor, EdObject object) {
371
372                    List<Command> commands = editor.finalizeEdit(object, getResurrectNodesDistanceMeters());
373
374                    if (commands.isEmpty()) {
375                        postTraceNotifications().add(tr("Nothing changed."));
376                        return;
377                    }
378
379                    long start_time = System.nanoTime();
380
381                    UndoRedoHandler.getInstance().add(new SequenceCommand(tr("Trace object"), commands));
382
383                    OsmPrimitive sel = null;
384
385                    if (object.isMultipolygon()) {
386                        sel = ((EdMultipolygon)object).finalMultipolygon();
387                    }
388                    else if (object.isWay()) {
389                        sel = ((EdWay)object).finalWay();
390                    }
391                    else {
392                        sel = ((EdNode)object).finalNode();
393                    }
394
395                    if (m_shift) {
396                        editor.getDataSet().addSelected(sel);
397                    } else {
398                        editor.getDataSet().setSelected(sel);
399                    }
400                    long end_time = System.nanoTime();
401                    long time_msecs = (end_time - start_time) / (1000 * 1000);
402                    System.out.println("undoRedo time (ms): " + Long.toString(time_msecs));
403                }
404            });
405        }
406
407        /**
408         * Returns list of all existing incomplete multipolygons that might participate in
409         * traced polygon clipping. These relations must be downloaded first, clipping
410         * doesn't support incomplete multipolygons.
411         * @return List of incomplete multipolygon relations
412         */
413        private List<Relation> getIncompleteMultipolygonsForDownload() {
414            DataSet ds = MainApplication.getLayerManager().getEditDataSet();
415            ds.getReadLock().lock();
416            BBox bbox = m_record.getBBox();
417            try {
418                List<Relation> list = new ArrayList<>();
419                for (Relation rel : ds.searchRelations(bbox)) {
420                    if (!rel.isMultipolygon() || (rel.isMultipolygon() && rel.hasKey("boundary"))) {
421                        System.out.println("[-] Skipping "+(rel.isMultipolygon() ? "boundary" : "non Multipolygon")+" relation "+rel.getId());
422                        continue;
423                    }
424                    if (rel.isIncomplete() || rel.hasIncompleteMembers()) {
425                        System.out.println("[+] Using relation "+rel.getId());
426                        list.add(rel);
427                    }
428                }
429                return list;
430            } finally {
431                ds.getReadLock().unlock();
432            }
433        }

461        protected EdWay getOuterWay(EdObject obj) {
462            if (obj.isWay()) {
463                return (EdWay)obj;
464            }
465            if (obj.isMultipolygon()) {
466                EdMultipolygon mp = (EdMultipolygon)obj;
467                List<EdWay> ways = mp.outerWays();
468                if (ways.size() == 1) {
469                    return ways.get(0);
470                }
471            }
472            throw new AssertionError("Cannot determine outer way of EdObject");
473        }

020package org.openstreetmap.josm.plugins.tracer;
021
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.HashSet;
025import java.util.List;
026import java.util.Map;
027import java.util.Set;
028import org.openstreetmap.josm.data.coor.LatLon;
029import org.openstreetmap.josm.data.osm.BBox;
030import org.openstreetmap.josm.plugins.tracer.connectways.LatLonSize;
031import org.openstreetmap.josm.tools.Pair;
032
033public class QuadCache<T extends IQuadCacheObject> {
034
035    private final LatLonSize m_quadSize;
036    private final Map<QuadIndex, Bucket<T>> m_buckets;
037
038    public QuadCache (LatLonSize quad_size) {
039        m_quadSize = quad_size;
040        m_buckets = new HashMap<> ();
041    }
042
043    public void add (T object) {
044        BBox bbox = object.getBBox();
045        Pair<QuadIndex, QuadIndex> qibox = bboxToQuadIndexBox (bbox);
046        QuadIndex qi1 = qibox.a;
047        QuadIndex qi2 = qibox.b;
048
049        // #### fixme: doesn't handle longitude corner cases for objects crossing Pacific Ocean meridian!!
050        // maybe restrict usage of this class to Czech republic latlons only
051
052        for (long ilat = qi1.iLat(); ilat <= qi2.iLat(); ilat++) {
053            for (long ilon = qi1.iLon(); ilon <= qi2.iLon(); ilon++) {
054                QuadIndex qi = new QuadIndex (ilat, ilon);
055                Bucket<T> bucket = m_buckets.get(qi);
056                if (bucket == null) {
057                    bucket = new Bucket<>();
058                    m_buckets.put(qi, bucket);
059                }
060                System.out.println ("QuadCache: adding to bucket: " + qi.toString());
061                bucket.add (object);
062            }
063        }
064    }
065
066    public List<T> search (LatLon latlon) {
067        QuadIndex qi = latLonToQuadIndex(latlon);
068
069        Bucket<T> bucket = m_buckets.get (qi);
070        if (bucket == null)
071            return null;
072        return bucket.search (latlon, qi);
073    }
074
075    public static final class QuadIndex {
076        private final long m_ilat;
077        private final long m_ilon;
078        private final int m_hashCode;
079
080        public QuadIndex (long ilat, long ilon) {
081            m_ilat = ilat;
082            m_ilon = ilon;
083            m_hashCode = Long.valueOf(m_ilat).hashCode() ^ Long.valueOf(m_ilon).hashCode();
084        }
085
086        public long iLat () {
087            return m_ilat;
088        }
089
090        public long iLon () {
091            return m_ilon;
092        }
093
094        @Override
095        public boolean equals(Object obj) {
096            if (obj == null) {
097                return false;
098            }
099            if (getClass() != obj.getClass()) {
100                return false;
101            }
102            final QuadIndex other = (QuadIndex) obj;
103            return other.m_ilat == m_ilat && other.m_ilon == m_ilon;
104        }
105
106        @Override
107        public int hashCode() {
108            return m_hashCode;
109        }
110
111        @Override
112        public String toString () {
113            return Long.toString(m_ilat) + "+" + Long.toString (m_ilon);
114        }
115
116        public static QuadIndex latLonToQuadIndex (LatLonSize quad_size, double alat, double alon) {
117            double lat = alat + 90;
118            double lon = alon + 180;
119            long ilat = (long) Math.floor (lat / quad_size.latSize());
120            long ilon = (long) Math.floor (lon / quad_size.lonSize());
121            return new QuadIndex (ilat, ilon);
122        }
123
124        public static BBox quadIndexToBBox (LatLonSize quad_size, QuadIndex qi) {
125            double alat = (double)qi.iLat() * quad_size.latSize();
126            double alon = (double)qi.iLon() * quad_size.lonSize();
127            alat -= 90;
128            alon -= 180;
129            LatLon a = new LatLon (alat, alon);
130            LatLon b = new LatLon (alat + quad_size.latSize(), alon + quad_size.lonSize());
131            return new BBox (a, b);
132        }
133    }
134
135    private class Bucket<T extends IQuadCacheObject> {
136        private final Set<T> m_contents;
137
138        Bucket () {
139            m_contents = new HashSet<>();
140        }
141
142        private void add(T object) {
143            m_contents.add(object);
144        }
145
146        private List<T> search(LatLon ll, QuadIndex qi) {
147            System.out.println ("QuadCache: searching in bucket: " + qi.toString() + ", total: " + Integer.toString (m_contents.size()));
148            List<T> list = null;
149            for (T object: m_contents) {
150                if (object.containsPoint (ll)) {
151                    if (list == null) {
152                        list = new ArrayList<> (5);
153                    }
154                    list.add(object);
155                }
156            }
157            return list;
158        }
159    }
160
161    public QuadIndex latLonToQuadIndex (LatLon latlon) {
162        return latLonToQuadIndex (latlon.lat(), latlon.lon());
163    }
164
165    public QuadIndex latLonToQuadIndex (double alat, double alon) {
166        return QuadIndex.latLonToQuadIndex(m_quadSize, alat, alon);
167    }
168
169    public Pair<QuadIndex, QuadIndex> bboxToQuadIndexBox(BBox bbox) {
170        double lat1 = bbox.getTopLeftLat();
171        double lon1 = bbox.getTopLeftLon();
172        double lat2 = bbox.getBottomRightLat();
173        double lon2 = bbox.getBottomRightLon();
174        QuadIndex qi1 = latLonToQuadIndex (lat1, lon1);
175        QuadIndex qi2 = latLonToQuadIndex (lat1, lon2);
176        QuadIndex qi3 = latLonToQuadIndex (lat2, lon1);
177        QuadIndex qi4 = latLonToQuadIndex (lat2, lon2);
178        long ilat1 = Math.min(Math.min (qi1.iLat(), qi2.iLat()), Math.min (qi3.iLat(), qi4.iLat()));
179        long ilon1 = Math.min(Math.min (qi1.iLon(), qi2.iLon()), Math.min (qi3.iLon(), qi4.iLon()));
180        long ilat2 = Math.max(Math.max (qi1.iLat(), qi2.iLat()), Math.max (qi3.iLat(), qi4.iLat()));
181        long ilon2 = Math.max(Math.max (qi1.iLon(), qi2.iLon()), Math.max (qi3.iLon(), qi4.iLon()));
182        return new Pair<> (new QuadIndex (ilat1, ilon1), new QuadIndex (ilat2, ilon2));
183    }
184}

037package org.openstreetmap.josm.plugins.tracer.clipper;
038
039import java.util.ArrayList;
040import java.util.List;
041
042  public class PolyTree extends PolyNode
043  {
044      List<PolyNode> m_AllPolys = new ArrayList<>();
045
046      public void clear()
047      {
048          m_AllPolys.clear();
049          m_Childs.clear();
050      }
051
052      public PolyNode getFirst()
053      {
054          if (m_Childs.size() > 0)
055              return m_Childs.get(0);
056          else
057              return null;
058      }
059
060      public int getTotal()
061      {
062            int result = m_AllPolys.size();
063            //with negative offsets, ignore the hidden outer polygon ...
064            if (result > 0 && m_Childs.get(0) != m_AllPolys.get(0)) result--;
065            return result;
066      }
067
068  }

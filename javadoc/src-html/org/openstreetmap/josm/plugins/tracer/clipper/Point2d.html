037package org.openstreetmap.josm.plugins.tracer.clipper;
038
039public class Point2d {
040    public long X;
041    public long Y;
042
043    public Point2d() {
044        X = 0;
045        Y = 0;
046    }
047
048    public Point2d(long x, long y) {
049        X = x;
050        Y = y;
051    }
052
053    public Point2d(Point2d p) {
054        X = p.X;
055        Y = p.Y;
056    }
057
058    public boolean equals(Point2d p) {
059        return p != null &&
060                p.X == this.X &&
061                p.Y == this.Y;
062    }
063
064    public boolean not_equals(Point2d p) {
065        return !this.equals(p);
066    }
067
068    public void assign(Point2d p) {
069        X = p.X;
070        Y = p.Y;
071    }
072
073    @Override
074    public Point2d clone() {
075        return new Point2d(this);
076    }
077
078    @Override
079    public boolean equals(Object obj) {
080        if (this == obj)
081            return true;
082        if (obj == null || getClass() != obj.getClass())
083            return false;
084        Point2d other = (Point2d) obj;
085        if (X != other.X)
086            return false;
087        if (Y != other.Y)
088            return false;
089        return true;
090    }
091
092    @Override
093    public int hashCode() {
094        int hash = 7;
095        hash = 83 * hash + (int) (this.X ^ (this.X >>> 32));
096        hash = 83 * hash + (int) (this.Y ^ (this.Y >>> 32));
097        return hash;
098    }
099
100}

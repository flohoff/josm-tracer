037package org.openstreetmap.josm.plugins.tracer.clipper;
038
039import java.util.ArrayList;
040import java.util.List;
041
042public class PolyNode {
043      PolyNode m_Parent;
044      Path m_polygon = new Path();
045      int m_Index;
046      JoinType m_jointype;
047      EndType m_endtype;
048      List<PolyNode> m_Childs = new ArrayList<>();
049      boolean m_isOpen;
050
051      private boolean isHoleNode()
052      {
053          boolean result = true;
054          PolyNode node = m_Parent;
055          while (node != null)
056          {
057              result = !result;
058              node = node.m_Parent;
059          }
060          return result;
061      }
062
063      public int getChildCount()
064      {
065          return m_Childs.size();
066      }
067
068      public Path getContour()
069      {
070          return m_polygon;
071      }
072
073      void addChild(PolyNode Child)
074      {
075          int cnt = m_Childs.size();
076          m_Childs.add(Child);
077          Child.m_Parent = this;
078          Child.m_Index = cnt;
079      }
080
081      public PolyNode getNext()
082      {
083          if (m_Childs.size() > 0)
084              return m_Childs.get(0);
085          else
086              return getNextSiblingUp();
087      }
088
089      PolyNode getNextSiblingUp()
090      {
091          if (m_Parent == null)
092              return null;
093          else if (m_Index == m_Parent.m_Childs.size() - 1)
094              return m_Parent.getNextSiblingUp();
095          else
096              return m_Parent.m_Childs.get(m_Index + 1);
097      }
098
099      public List<PolyNode> getChilds()
100      {
101          return m_Childs;
102      }
103
104      public PolyNode getParent()
105      {
106          return m_Parent;
107      }
108
109      public boolean isHole()
110      {
111          return isHoleNode();
112      }
113
114      public boolean isOpen()
115      {
116          return m_isOpen;
117      }
118}

191    boolean pointOnLineSegment(Point2d pt, Point2d linePt1, Point2d linePt2, boolean UseFullRange)
192    {
193      if (UseFullRange)
194        return ((pt.X == linePt1.X) && (pt.Y == linePt1.Y)) ||
195          ((pt.X == linePt2.X) && (pt.Y == linePt2.Y)) ||
196          (((pt.X > linePt1.X) == (pt.X < linePt2.X)) &&
197          ((pt.Y > linePt1.Y) == (pt.Y < linePt2.Y)) &&
198          ((Int128Mul((pt.X - linePt1.X), (linePt2.Y - linePt1.Y)).equals(
199          Int128Mul((linePt2.X - linePt1.X), (pt.Y - linePt1.Y))))));
200      else
201        return ((pt.X == linePt1.X) && (pt.Y == linePt1.Y)) ||
202            ((pt.X == linePt2.X) && (pt.Y == linePt2.Y)) ||
203            (((pt.X > linePt1.X) == (pt.X < linePt2.X)) &&
204            ((pt.Y > linePt1.Y) == (pt.Y < linePt2.Y)) &&
205            ((pt.X - linePt1.X) * (linePt2.Y - linePt1.Y) ==
206            (linePt2.X - linePt1.X) * (pt.Y - linePt1.Y)));
207    }

220        }
221        return false;

224    static boolean slopesEqual(TEdge e1, TEdge e2, boolean UseFullRange)
225    {
226        if (UseFullRange)
227          return Int128Mul(e1.iDelta.Y, e2.iDelta.X).equals(
228              Int128Mul(e1.iDelta.X, e2.iDelta.Y));
229        else return (long)(e1.iDelta.Y) * (e2.iDelta.X) ==
230          (long)(e1.iDelta.X) * (e2.iDelta.Y);
231    }
232    //------------------------------------------------------------------------------
233
234    static boolean slopesEqual(Point2d pt1, Point2d pt2, Point2d pt3, boolean UseFullRange)
235    {
236        if (UseFullRange)
237            return Int128Mul(pt1.Y - pt2.Y, pt2.X - pt3.X).equals(
238              Int128Mul(pt1.X - pt2.X, pt2.Y - pt3.Y));
239        else return
240          (long)(pt1.Y - pt2.Y) * (pt2.X - pt3.X) - (long)(pt1.X - pt2.X) * (pt2.Y - pt3.Y) == 0;
241    }

251          (long)(pt1.Y - pt2.Y) * (pt3.X - pt4.X) - (long)(pt1.X - pt2.X) * (pt3.Y - pt4.Y) == 0;

281    boolean rangeTest(Point2d Pt, boolean useFullRange) throws ClipperException
282    {
283      if (useFullRange)
284      {
285        if (Pt.X > hiRange || Pt.Y > hiRange || -Pt.X > hiRange || -Pt.Y > hiRange)
286          throw new ClipperException("Coordinate outside allowed range");
287      }
288      else if (Pt.X > loRange || Pt.Y > loRange || -Pt.X > loRange || -Pt.Y > loRange)
289      {
290        useFullRange = true;
291        rangeTest(Pt, useFullRange);
292      }
293      return useFullRange;
294    }

323    private TEdge findNextLocMin(TEdge E)
324    {
325      TEdge E2;
326      for (;;)
327      {
328        while (E.iBot.not_equals(E.Prev.iBot) || E.iCurr.equals(E.iTop)) E = E.Next;
329        if (E.Dx != horizontal && E.Prev.Dx != horizontal) break;
330        while (E.Prev.Dx == horizontal) E = E.Prev;
331        E2 = E;
332        while (E.Dx == horizontal) E = E.Next;
333        if (E.iTop.Y == E.Prev.iBot.Y) continue; //ie just an intermediate horz.
334        if (E2.Prev.iBot.X < E.iBot.X) E = E2;
335        break;
336      }
337      return E;
338    }

342    private TEdge processBound(TEdge E, boolean LeftBoundIsForward)
343    {
344      TEdge EStart, Result = E;
345      TEdge Horz;
346
347      if (Result.OutIdx == Skip)
348      {
349        //check if there are edges beyond the skip edge in the bound and if so
350        //create another LocMin and calling ProcessBound once more ...
351        E = Result;
352        if (LeftBoundIsForward)
353        {
354          while (E.iTop.Y == E.Next.iBot.Y) E = E.Next;
355          while (E != Result && E.Dx == horizontal) E = E.Prev;
356        }
357        else
358        {
359          while (E.iTop.Y == E.Prev.iBot.Y) E = E.Prev;
360          while (E != Result && E.Dx == horizontal) E = E.Next;
361        }
362        if (E == Result)
363        {
364          if (LeftBoundIsForward) Result = E.Next;
365          else Result = E.Prev;
366        }
367        else
368        {
369          //there are more edges in the bound beyond result starting with E
370          if (LeftBoundIsForward)
371            E = Result.Next;
372          else
373            E = Result.Prev;
374          LocalMinima locMin = new LocalMinima();
375          locMin.Next = null;
376          locMin.Y = E.iBot.Y;
377          locMin.LeftBound = null;
378          locMin.RightBound = E;
379          E.WindDelta = 0;
380          Result = processBound(E, LeftBoundIsForward);
381          insertLocalMinima(locMin);
382        }
383        return Result;
384      }
385
386      if (E.Dx == horizontal)
387      {
388        //We need to be careful with open paths because this may not be a
389        //true local minima (ie E may be following a skip edge).
390        //Also, consecutive horz. edges may start heading left before going right.
391        if (LeftBoundIsForward) EStart = E.Prev;
392        else EStart = E.Next;
393        if (EStart.OutIdx != Skip)
394        {
395          if (EStart.Dx == horizontal) //ie an adjoining horizontal skip edge
396          {
397            if (EStart.iBot.X != E.iBot.X && EStart.iTop.X != E.iBot.X)
398              reverseHorizontal(E);
399          }
400          else if (EStart.iBot.X != E.iBot.X)
401            reverseHorizontal(E);
402        }
403      }
404
405      EStart = E;
406      if (LeftBoundIsForward)
407      {
408        while (Result.iTop.Y == Result.Next.iBot.Y && Result.Next.OutIdx != Skip)
409          Result = Result.Next;
410        if (Result.Dx == horizontal && Result.Next.OutIdx != Skip)
411        {
412          //nb: at the top of a bound, horizontals are added to the bound
413          //only when the preceding edge attaches to the horizontal's left vertex
414          //unless a Skip edge is encountered when that becomes the top divide
415          Horz = Result;
416          while (Horz.Prev.Dx == horizontal) Horz = Horz.Prev;
417          if (Horz.Prev.iTop.X == Result.Next.iTop.X)
418          {
419            if (!LeftBoundIsForward) Result = Horz.Prev;
420          }
421          else if (Horz.Prev.iTop.X > Result.Next.iTop.X) Result = Horz.Prev;
422        }
423        while (E != Result)
424        {
425          E.NextInLML = E.Next;
426          if (E.Dx == horizontal && E != EStart && E.iBot.X != E.Prev.iTop.X)
427            reverseHorizontal(E);
428          E = E.Next;
429        }
430        if (E.Dx == horizontal && E != EStart && E.iBot.X != E.Prev.iTop.X)
431          reverseHorizontal(E);
432        Result = Result.Next; //move to the edge just beyond current bound
433      }
434      else
435      {
436        while (Result.iTop.Y == Result.Prev.iBot.Y && Result.Prev.OutIdx != Skip)
437          Result = Result.Prev;
438        if (Result.Dx == horizontal && Result.Prev.OutIdx != Skip)
439        {
440          Horz = Result;
441          while (Horz.Next.Dx == horizontal) Horz = Horz.Next;
442          if (Horz.Next.iTop.X == Result.Prev.iTop.X)
443          {
444            if (!LeftBoundIsForward) Result = Horz.Next;
445          }
446          else if (Horz.Next.iTop.X > Result.Prev.iTop.X) Result = Horz.Next;
447        }
448
449        while (E != Result)
450        {
451          E.NextInLML = E.Prev;
452          if (E.Dx == horizontal && E != EStart && E.iBot.X != E.Next.iTop.X)
453            reverseHorizontal(E);
454          E = E.Prev;
455        }
456        if (E.Dx == horizontal && E != EStart && E.iBot.X != E.Next.iTop.X)
457          reverseHorizontal(E);
458        Result = Result.Prev; //move to the edge just beyond current bound
459      }
460      return Result;
461    }

464    public boolean addPath(Path pg, PolyType polyType, boolean Closed) throws ClipperException
465    {
466      if (!Closed && polyType == PolyType.ptClip)
467        throw new ClipperException("addPath: Open paths must be subject.");
468
469      int highI = (int)pg.size() - 1;
470      if (Closed) while (highI > 0 && (pg.get(highI) == pg.get(0))) --highI;
471      while (highI > 0 && (pg.get(highI) == pg.get(highI - 1))) --highI;
472      if ((Closed && highI < 2) || (!Closed && highI < 1)) return false;
473
474      //create a new edge array ...
475      List<TEdge> edges = new ArrayList<>(highI+1);
476      for (int i = 0; i <= highI; i++) edges.add(new TEdge());
477
478      boolean IsFlat = true;
479
480      //1. Basic (first) edge initialization ...
481      edges.get(1).iCurr.assign(pg.get(1));
482      m_UseFullRange = rangeTest(pg.get(0), m_UseFullRange);
483      m_UseFullRange = rangeTest(pg.get(highI), m_UseFullRange);
484      initEdge(edges.get(0), edges.get(1), edges.get(highI), pg.get(0));
485      initEdge(edges.get(highI), edges.get(0), edges.get(highI - 1), pg.get(highI));
486      for (int i = highI - 1; i >= 1; --i)
487      {
488        m_UseFullRange = rangeTest(pg.get(i), m_UseFullRange);
489        initEdge(edges.get(i), edges.get(i + 1), edges.get(i - 1), pg.get(i));
490      }
491      TEdge eStart = edges.get(0);
492
493      //2. Remove duplicate vertices, and (when closed) collinear edges ...
494      TEdge E = eStart, eLoopStop = eStart;
495      for (;;)
496      {
497        //nb: allows matching start and end points when not Closed ...
498        if (E.iCurr.equals(E.Next.iCurr) && (Closed || E.Next != eStart))
499        {
500          if (E == E.Next) break;
501          if (E == eStart) eStart = E.Next;
502          E = removeEdge(E);
503          eLoopStop = E;
504          continue;
505        }
506        if (E.Prev == E.Next)
507          break; //only two vertices
508        else if (Closed &&
509          slopesEqual(E.Prev.iCurr, E.iCurr, E.Next.iCurr, m_UseFullRange) &&
510          (!getPreserveCollinear() ||
511          !Pt2IsBetweenPt1AndPt3(E.Prev.iCurr, E.iCurr, E.Next.iCurr)))
512        {
513          //Collinear edges are allowed for open paths but in closed paths
514          //the default is to merge adjacent collinear edges into a single edge.
515          //However, if the PreserveCollinear property is enabled, only overlapping
516          //collinear edges (ie spikes) will be removed from closed paths.
517          if (E == eStart) eStart = E.Next;
518          E = removeEdge(E);
519          E = E.Prev;
520          eLoopStop = E;
521          continue;
522        }
523        E = E.Next;
524        if ((E == eLoopStop) || (!Closed && E.Next == eStart)) break;
525      }
526
527      if ((!Closed && (E == E.Next)) || (Closed && (E.Prev == E.Next)))
528        return false;
529
530      if (!Closed)
531      {
532        m_HasOpenPaths = true;
533        eStart.Prev.OutIdx = Skip;
534      }
535
536      //3. Do second stage of edge initialization ...
537      E = eStart;
538      do
539      {
540        initEdge2(E, polyType);
541        E = E.Next;
542        if (IsFlat && E.iCurr.Y != eStart.iCurr.Y) IsFlat = false;
543      }
544      while (E != eStart);
545
546      //4. Finally, add edge bounds to LocalMinima list ...
547
548      //Totally flat paths must be handled differently when adding them
549      //to LocalMinima list to avoid endless loops etc ...
550      if (IsFlat)
551      {
552        if (Closed) return false;
553        E.Prev.OutIdx = Skip;
554        if (E.Prev.iBot.X < E.Prev.iTop.X) reverseHorizontal(E.Prev);
555        LocalMinima locMin = new LocalMinima();
556        locMin.Next = null;
557        locMin.Y = E.iBot.Y;
558        locMin.LeftBound = null;
559        locMin.RightBound = E;
560        locMin.RightBound.Side = EdgeSide.esRight;
561        locMin.RightBound.WindDelta = 0;
562        while (E.Next.OutIdx != Skip)
563        {
564          E.NextInLML = E.Next;
565          if (E.iBot.X != E.Prev.iTop.X) reverseHorizontal(E);
566          E = E.Next;
567        }
568        insertLocalMinima(locMin);
569        m_edges.add(edges);
570        return true;
571      }
572
573      m_edges.add(edges);
574      boolean leftBoundIsForward;
575      TEdge EMin = null;
576
577      //workaround to avoid an endless loop in the while loop below when
578      //open paths have matching start and end points ...
579      if (E.Prev.iBot.equals(E.Prev.iTop)) E = E.Next;
580
581      for (;;)
582      {
583        E = findNextLocMin(E);
584        if (E == EMin) break;
585        else if (EMin == null) EMin = E;
586
587        //E and E.Prev now share a local minima (left aligned if horizontal).
588        //Compare their slopes to find which starts which bound ...
589        LocalMinima locMin = new LocalMinima();
590        locMin.Next = null;
591        locMin.Y = E.iBot.Y;
592        if (E.Dx < E.Prev.Dx)
593        {
594          locMin.LeftBound = E.Prev;
595          locMin.RightBound = E;
596          leftBoundIsForward = false; //Q.nextInLML = Q.prev
597        } else
598        {
599          locMin.LeftBound = E;
600          locMin.RightBound = E.Prev;
601          leftBoundIsForward = true; //Q.nextInLML = Q.next
602        }
603        locMin.LeftBound.Side = EdgeSide.esLeft;
604        locMin.RightBound.Side = EdgeSide.esRight;
605
606        if (!Closed) locMin.LeftBound.WindDelta = 0;
607        else if (locMin.LeftBound.Next == locMin.RightBound)
608          locMin.LeftBound.WindDelta = -1;
609        else locMin.LeftBound.WindDelta = 1;
610        locMin.RightBound.WindDelta = -locMin.LeftBound.WindDelta;
611
612        E = processBound(locMin.LeftBound, leftBoundIsForward);
613        if (E.OutIdx == Skip) E = processBound(E, leftBoundIsForward);
614
615        TEdge E2 = processBound(locMin.RightBound, !leftBoundIsForward);
616        if (E2.OutIdx == Skip) E2 = processBound(E2, !leftBoundIsForward);
617
618        if (locMin.LeftBound.OutIdx == Skip)
619          locMin.LeftBound = null;
620        else if (locMin.RightBound.OutIdx == Skip)
621          locMin.RightBound = null;
622        insertLocalMinima(locMin);
623        if (!leftBoundIsForward) E = E2;
624      }
625      return true;
626
627    }

639    boolean Pt2IsBetweenPt1AndPt3(Point2d pt1, Point2d pt2, Point2d pt3)
640    {
641      if ((pt1.equals(pt3)) || (pt1.equals(pt2)) || (pt3.equals(pt2))) return false;
642      else if (pt1.X != pt3.X) return (pt2.X > pt1.X) == (pt2.X < pt3.X);
643      else return (pt2.Y > pt1.Y) == (pt2.Y < pt3.Y);
644    }

647    TEdge removeEdge(TEdge e)
648    {
649      //removes e from double_linked_list (but without removing from memory)
650      e.Prev.Next = e.Next;
651      e.Next.Prev = e.Prev;
652      TEdge result = e.Next;
653      e.Prev = null; //flag as removed (see ClipperBase.Clear)
654      return result;

659    {
660      e.iDelta.X = (e.iTop.X - e.iBot.X);
661      e.iDelta.Y = (e.iTop.Y - e.iBot.Y);
662      if (e.iDelta.Y == 0) e.Dx = horizontal;
663      else e.Dx = (double)(e.iDelta.X) / (e.iDelta.Y);

667    private void insertLocalMinima(LocalMinima newLm)
668    {
669      if( m_MinimaList == null )
670      {
671        m_MinimaList = newLm;
672      }
673      else if( newLm.Y >= m_MinimaList.Y )
674      {
675        newLm.Next = m_MinimaList;
676        m_MinimaList = newLm;
677      } else
678      {
679        LocalMinima tmpLm = m_MinimaList;
680        while( tmpLm.Next != null  && ( newLm.Y < tmpLm.Next.Y ) )
681          tmpLm = tmpLm.Next;
682        newLm.Next = tmpLm.Next;
683        tmpLm.Next = newLm;
684      }
685    }

689    {
690        if (m_CurrentLM == null) return;
691        m_CurrentLM = m_CurrentLM.Next;
692    }

695    private void reverseHorizontal(TEdge e)
696    {
697      //swap horizontal edges' top and bottom x's so they follow the natural
698      //progression of the bounds - ie so their xbots will align with the
699      //adjoining lower edge. [Helpful in the ProcessHorizontal() method.]
700      long aux;
701
702      aux = e.iTop.X;
703      e.iTop.X = e.iBot.X;
704      e.iBot.X = aux;
705    }

708    protected void reset()
709    {
710      m_CurrentLM = m_MinimaList;
711      if (m_CurrentLM == null) return; //ie nothing to process
712
713      //reset all edges ...
714      LocalMinima lm = m_MinimaList;
715      while (lm != null)
716      {
717        TEdge e = lm.LeftBound;
718        if (e != null)
719        {
720          e.iCurr.assign(e.iBot);
721          e.Side = EdgeSide.esLeft;
722          e.OutIdx = Unassigned;
723        }
724        e = lm.RightBound;
725        if (e != null)
726        {
727          e.iCurr.assign(e.iBot);
728          e.Side = EdgeSide.esRight;
729          e.OutIdx = Unassigned;
730        }
731        lm = lm.Next;
732      }
733    }

934      {
935        try
936        {
937          reset();
938          if (m_CurrentLM == null) return false;
939
940          long botY = popScanbeam();
941          do
942          {
943            insertLocalMinimaIntoAEL(botY);
944            m_GhostJoins.clear();
945            processHorizontals(false);
946            if (m_Scanbeam == null) break;
947            long topY = popScanbeam();
948            if (!processIntersections(topY)) return false;
949            processEdgesAtTopOfScanbeam(topY);
950            botY = topY;
951          } while (m_Scanbeam != null || m_CurrentLM != null);

955                if ((outRec.IsHole ^ getReverseSolution()) == (area(outRec) > 0))

959          joinCommonEdges();
960
961            for (OutRec outRec : m_PolyOuts) {
962                if (outRec.Pts != null && !outRec.IsOpen)
963                    fixupOutPolygon(outRec);
964            }
965
966          if (getStrictlySimple()) doSimplePolygons();
967          return true;
968        }
969        //catch { return false; }
970        finally
971        {
972          m_Joins.clear();
973          m_GhostJoins.clear();
974        }

1021      private void insertLocalMinimaIntoAEL(long botY)
1022      {
1023        while(  m_CurrentLM != null  && ( m_CurrentLM.Y == botY ) )
1024        {
1025          TEdge lb = m_CurrentLM.LeftBound;
1026          TEdge rb = m_CurrentLM.RightBound;
1027          popLocalMinima();
1028
1029          OutPt Op1 = null;
1030          if (lb == null)
1031          {
1032            insertEdgeIntoAEL(rb, null);
1033            setWindingCount(rb);
1034            if (isContributing(rb))
1035              Op1 = addOutPt(rb, rb.iBot);
1036          }
1037          else if (rb == null)
1038          {
1039            insertEdgeIntoAEL(lb, null);
1040            setWindingCount(lb);
1041            if (isContributing(lb))
1042              Op1 = addOutPt(lb, lb.iBot);
1043            insertScanbeam(lb.iTop.Y);
1044          }
1045          else
1046          {
1047            insertEdgeIntoAEL(lb, null);
1048            insertEdgeIntoAEL(rb, lb);
1049            setWindingCount(lb);
1050            rb.WindCnt = lb.WindCnt;
1051            rb.WindCnt2 = lb.WindCnt2;
1052            if (isContributing(lb))
1053              Op1 = addLocalMinPoly(lb, rb, lb.iBot);
1054            insertScanbeam(lb.iTop.Y);
1055          }
1056
1057          if (rb != null)
1058          {
1059            if (isHorizontal(rb))
1060              addEdgeToSEL(rb);
1061            else
1062              insertScanbeam(rb.iTop.Y);
1063          }
1064
1065          if (lb == null || rb == null) continue;
1066
1067          //if output polygons share an Edge with a horizontal rb, they'll need joining later ...
1068          if (Op1 != null && isHorizontal(rb) &&
1069            m_GhostJoins.size() > 0 && rb.WindDelta != 0)
1070          {
1071              for (Join j : m_GhostJoins) {
1072                  if (horzSegmentsOverlap(j.OutPt1.iPt.X, j.iOffPt.X, rb.iBot.X, rb.iTop.X))
1073                      addJoin(j.OutPt1, Op1, j.iOffPt);
1074              }
1075          }
1076
1077          if (lb.OutIdx >= 0 && lb.PrevInAEL != null &&
1078            lb.PrevInAEL.iCurr.X == lb.iBot.X &&
1079            lb.PrevInAEL.OutIdx >= 0 &&
1080            slopesEqual(lb.PrevInAEL, lb, m_UseFullRange) &&
1081            lb.WindDelta != 0 && lb.PrevInAEL.WindDelta != 0)
1082          {
1083            OutPt Op2 = addOutPt(lb.PrevInAEL, lb.iBot);
1084            addJoin(Op1, Op2, lb.iTop);
1085          }
1086
1087          if( lb.NextInAEL != rb )
1088          {
1089
1090            if (rb.OutIdx >= 0 && rb.PrevInAEL.OutIdx >= 0 &&
1091              slopesEqual(rb.PrevInAEL, rb, m_UseFullRange) &&
1092              rb.WindDelta != 0 && rb.PrevInAEL.WindDelta != 0)
1093            {
1094              OutPt Op2 = addOutPt(rb.PrevInAEL, rb.iBot);
1095              addJoin(Op1, Op2, rb.iTop);
1096            }
1097
1098            TEdge e = lb.NextInAEL;
1099            if (e != null)
1100              while (e != rb)
1101              {
1102                //nb: For calculating winding counts etc, IntersectEdges() assumes
1103                //that param1 will be to the right of param2 ABOVE the intersection ...
1104                intersectEdges(rb, e, lb.iCurr); //order important here
1105                e = e.NextInAEL;
1106              }
1107          }
1108        }
1109      }

1172      private boolean isContributing(TEdge edge)
1173      {
1174          PolyFillType pft, pft2;
1175          if (edge.PolyTyp == PolyType.ptSubject)
1176          {
1177              pft = m_SubjFillType;
1178              pft2 = m_ClipFillType;
1179          }
1180          else
1181          {
1182              pft = m_ClipFillType;
1183              pft2 = m_SubjFillType;
1184          }
1185
1186          switch (pft)
1187          {
1188              case pftEvenOdd:
1189                  //return false if a subj line has been flagged as inside a subj polygon
1190                  if (edge.WindDelta == 0 && edge.WindCnt != 1) return false;
1191                  break;
1192              case pftNonZero:
1193                  if (Math.abs(edge.WindCnt) != 1) return false;
1194                  break;
1195              case pftPositive:
1196                  if (edge.WindCnt != 1) return false;
1197                  break;
1198              default: //PolyFillType.pftNegative
1199                  if (edge.WindCnt != -1) return false;
1200                  break;
1201          }
1202
1203          switch (m_ClipType)
1204          {
1205            case ctIntersection:
1206                switch (pft2)
1207                {
1208                    case pftEvenOdd:
1209                    case pftNonZero:
1210                        return (edge.WindCnt2 != 0);
1211                    case pftPositive:
1212                        return (edge.WindCnt2 > 0);
1213                    default:
1214                        return (edge.WindCnt2 < 0);
1215                }
1216            case ctUnion:
1217                switch (pft2)
1218                {
1219                    case pftEvenOdd:
1220                    case pftNonZero:
1221                        return (edge.WindCnt2 == 0);
1222                    case pftPositive:
1223                        return (edge.WindCnt2 <= 0);
1224                    default:
1225                        return (edge.WindCnt2 >= 0);
1226                }
1227            case ctDifference:
1228                if (edge.PolyTyp == PolyType.ptSubject)
1229                    switch (pft2)
1230                    {
1231                        case pftEvenOdd:
1232                        case pftNonZero:
1233                            return (edge.WindCnt2 == 0);
1234                        case pftPositive:
1235                            return (edge.WindCnt2 <= 0);
1236                        default:
1237                            return (edge.WindCnt2 >= 0);
1238                    }
1239                else
1240                    switch (pft2)
1241                    {
1242                        case pftEvenOdd:
1243                        case pftNonZero:
1244                            return (edge.WindCnt2 != 0);
1245                        case pftPositive:
1246                            return (edge.WindCnt2 > 0);
1247                        default:
1248                            return (edge.WindCnt2 < 0);
1249                    }
1250            case ctXor:
1251                if (edge.WindDelta == 0) //XOr always contributing unless open
1252                  switch (pft2)
1253                  {
1254                    case pftEvenOdd:
1255                    case pftNonZero:
1256                      return (edge.WindCnt2 == 0);
1257                    case pftPositive:
1258                      return (edge.WindCnt2 <= 0);
1259                    default:
1260                      return (edge.WindCnt2 >= 0);
1261                  }
1262                else
1263                  return true;
1264          }
1265          return true;
1266      }

1270      private void setWindingCount(TEdge edge)
1271      {
1272        TEdge e = edge.PrevInAEL;
1273        //find the edge of the same polytype that immediately preceeds 'edge' in AEL
1274        while (e != null && ((e.PolyTyp != edge.PolyTyp) || (e.WindDelta == 0))) e = e.PrevInAEL;
1275        if (e == null)
1276        {
1277          edge.WindCnt = (edge.WindDelta == 0 ? 1 : edge.WindDelta);
1278          edge.WindCnt2 = 0;
1279          e = m_ActiveEdges; //ie get ready to calc WindCnt2
1280        }
1281        else if (edge.WindDelta == 0 && m_ClipType != ClipType.ctUnion)
1282        {
1283          edge.WindCnt = 1;
1284          edge.WindCnt2 = e.WindCnt2;
1285          e = e.NextInAEL; //ie get ready to calc WindCnt2
1286        }
1287        else if (isEvenOddFillType(edge))
1288        {
1289          //EvenOdd filling ...
1290          if (edge.WindDelta == 0)
1291          {
1292            //are we inside a subj polygon ...
1293            boolean Inside = true;
1294            TEdge e2 = e.PrevInAEL;
1295            while (e2 != null)
1296            {
1297              if (e2.PolyTyp == e.PolyTyp && e2.WindDelta != 0)
1298                Inside = !Inside;
1299              e2 = e2.PrevInAEL;
1300            }
1301            edge.WindCnt = (Inside ? 0 : 1);
1302          }
1303          else
1304          {
1305            edge.WindCnt = edge.WindDelta;
1306          }
1307          edge.WindCnt2 = e.WindCnt2;
1308          e = e.NextInAEL; //ie get ready to calc WindCnt2
1309        }
1310        else
1311        {
1312          //nonZero, Positive or Negative filling ...
1313          if (e.WindCnt * e.WindDelta < 0)
1314          {
1315            //prev edge is 'decreasing' WindCount (WC) toward zero
1316            //so we're outside the previous polygon ...
1317            if (Math.abs(e.WindCnt) > 1)
1318            {
1319              //outside prev poly but still inside another.
1320              //when reversing direction of prev poly use the same WC
1321              if (e.WindDelta * edge.WindDelta < 0) edge.WindCnt = e.WindCnt;
1322              //otherwise continue to 'decrease' WC ...
1323              else edge.WindCnt = e.WindCnt + edge.WindDelta;
1324            }
1325            else
1326              //now outside all polys of same polytype so set own WC ...
1327              edge.WindCnt = (edge.WindDelta == 0 ? 1 : edge.WindDelta);
1328          }
1329          else
1330          {
1331            //prev edge is 'increasing' WindCount (WC) away from zero
1332            //so we're inside the previous polygon ...
1333            if (edge.WindDelta == 0)
1334              edge.WindCnt = (e.WindCnt < 0 ? e.WindCnt - 1 : e.WindCnt + 1);
1335            //if wind direction is reversing prev then use same WC
1336            else if (e.WindDelta * edge.WindDelta < 0)
1337              edge.WindCnt = e.WindCnt;
1338            //otherwise add to WC ...
1339            else edge.WindCnt = e.WindCnt + edge.WindDelta;
1340          }
1341          edge.WindCnt2 = e.WindCnt2;
1342          e = e.NextInAEL; //ie get ready to calc WindCnt2
1343        }
1344
1345        //update WindCnt2 ...

1718          }

1725          {

1727            dups = dups.Next;
1728            while (dups.iPt.not_equals(pp.iPt)) dups = dups.Next;

1903      private void intersectEdges(TEdge e1, TEdge e2, final Point2d _pt)
1904      {
1905        final Point2d pt = _pt.clone();
1906
1907          //e1 will be to the left of e2 BELOW the intersection. Therefore e1 is before
1908          //e2 in AEL except when e1 is being inserted at the intersection point ...
1909
1910        boolean e1Contributing = (e1.OutIdx >= 0);
1911        boolean e2Contributing = (e2.OutIdx >= 0);
1912
1913          //if either edge is on an OPEN path ...
1914          if (e1.WindDelta == 0 || e2.WindDelta == 0)
1915          {
1916            //ignore subject-subject open path intersections UNLESS they
1917            //are both open paths, AND they are both 'contributing maximas' ...
1918            if (e1.WindDelta == 0 && e2.WindDelta == 0) return;
1919            //if intersecting a subj line with a subj poly ...
1920            else if (e1.PolyTyp == e2.PolyTyp &&
1921              e1.WindDelta != e2.WindDelta && m_ClipType == ClipType.ctUnion)
1922            {
1923              if (e1.WindDelta == 0)
1924              {
1925                if (e2Contributing)
1926                {
1927                  addOutPt(e1, pt);
1928                  if (e1Contributing) e1.OutIdx = Unassigned;
1929                }
1930              }
1931              else
1932              {
1933                if (e1Contributing)
1934                {
1935                  addOutPt(e2, pt);
1936                  if (e2Contributing) e2.OutIdx = Unassigned;
1937                }
1938              }
1939            }
1940            else if (e1.PolyTyp != e2.PolyTyp)
1941            {
1942              if ((e1.WindDelta == 0) && Math.abs(e2.WindCnt) == 1 &&
1943                (m_ClipType != ClipType.ctUnion || e2.WindCnt2 == 0))
1944              {
1945                addOutPt(e1, pt);
1946                if (e1Contributing) e1.OutIdx = Unassigned;
1947              }
1948              else if ((e2.WindDelta == 0) && (Math.abs(e1.WindCnt) == 1) &&
1949                (m_ClipType != ClipType.ctUnion || e1.WindCnt2 == 0))
1950              {
1951                addOutPt(e2, pt);
1952                if (e2Contributing) e2.OutIdx = Unassigned;
1953              }
1954            }
1955            return;
1956          }
1957
1958          //update winding counts...
1959  //assumes that e1 will be to the Right of e2 ABOVE the intersection
1960          if (e1.PolyTyp == e2.PolyTyp)
1961          {
1962              if (isEvenOddFillType(e1))
1963              {
1964                  int oldE1WindCnt = e1.WindCnt;

3530      private static double distanceFromLineSqrd(Point2d pt, Point2d ln1, Point2d ln2)
3531      {
3532        //The equation of a line in general form (Ax + By + C = 0)
3533        //given 2 points (x¹,y¹) & (x²,y²) is ...
3534        //(y¹ - y²)x + (x² - x¹)y + (y² - y¹)x¹ - (x² - x¹)y¹ = 0
3535        //A = (y¹ - y²); B = (x² - x¹); C = (y² - y¹)x¹ - (x² - x¹)y¹
3536        //perpendicular distance of point (x³,y³) = (Ax³ + By³ + C)/Sqrt(A² + B²)
3537        //see http://en.wikipedia.org/wiki/Perpendicular_distance
3538        double A = ln1.Y - ln2.Y;
3539        double B = ln2.X - ln1.X;
3540        double C = A * ln1.X  + B * ln1.Y;
3541        C = A * pt.X + B * pt.Y - C;
3542        return (C * C) / (A * A + B * B);
3543      }

3547          Point2d pt2, Point2d pt3, double distSqrd)
3548      {
3549        //this function is more accurate when the point that's GEOMETRICALLY

3553              {

3556          else if ((pt2.X > pt1.X) == (pt2.X < pt3.X))

3558                      else

3597      {

3599        //Default ~= sqrt(2) so when adjacent vertices or semi-adjacent vertices have
3600        //both x & y coords within 1 unit, then the second vertex will be stripped.
3601
3602        int cnt = path.size();
3603
3604        if (cnt == 0) return new Path();
3605
3606        OutPt [] outPts = new OutPt[cnt];
3607        for (int i = 0; i < cnt; ++i) outPts[i] = new OutPt();
3608

3610        {
3611          outPts[i].iPt.assign(path.get(i));

3613          outPts[i].Next.Prev = outPts[i];

3615        }

3617        double distSqrd = distance * distance;
3618        OutPt op = outPts[0];
3619        while (op.Idx == 0 && op.Next != op.Prev)
3620        {
3621          if (pointsAreClose(op.iPt, op.Prev.iPt, distSqrd))
3622          {
3623            op = excludeOp(op);

3625          }
3626          else if (pointsAreClose(op.Prev.iPt, op.Next.iPt, distSqrd))
3627          {
3628            excludeOp(op.Next);
3629            op = excludeOp(op);
3630            cnt -= 2;
3631          }
3632          else if (slopesNearCollinear(op.Prev.iPt, op.iPt, op.Next.iPt, distSqrd))
3633          {
3634            op = excludeOp(op);
3635            cnt--;
3636          }
3637          else
3638          {
3639            op.Idx = 1;
3640            op = op.Next;
3641          }
3642        }
3643
3644        if (cnt < 3) cnt = 0;
3645        Path result = new Path();
3646        for (int i = 0; i < cnt; ++i)
3647        {
3648          result.add(op.iPt);
3649          op = op.Next;
3650        }
3651        return result;

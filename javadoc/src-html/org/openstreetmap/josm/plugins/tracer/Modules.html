019package org.openstreetmap.josm.plugins.tracer;
020
021import java.util.*;
022
023import org.openstreetmap.josm.plugins.tracer.modules.classic.*;
024import org.openstreetmap.josm.plugins.tracer.modules.lpis.*;
025import org.openstreetmap.josm.plugins.tracer.modules.ruian.*;
026import org.openstreetmap.josm.plugins.tracer.modules.ruianLands.*;
027
028/**
029 * Private class to store modules
030 *
031 */
032
033public class Modules {
034    private Map<String, TracerModule> m_modules; // map of all modules
035    private Map.Entry<String, TracerModule> m_active_module; // active module
036
037    private Iterator<Map.Entry<String, TracerModule>>  m_it; // holds current iterator
038
039    private int m_activeModulesCount = 0;
040
041    TracerPreferences pref = TracerPreferences.getInstance();
042
043    private void init () {
044      m_modules = new LinkedHashMap<String, TracerModule>();
045      m_modules.put("classic", new ClassicModule(false));
046      m_modules.put("ruian", new RuianModule(true));
047      m_modules.put("ruian-lands", new RuianLandsModule(false));
048      m_modules.put("lpis", new LpisModule(false));
049
050      List<String> m_modules_array = new LinkedList<String>();
051      Map.Entry<String, TracerModule> m_current_module;
052
053      // Refresh modules list in configuration
054      m_it = m_modules.entrySet().iterator();
055      while (m_it.hasNext()) {
056        m_current_module = m_it.next();
057        m_modules_array.add(m_current_module.getKey());
058      }
059      pref.setAvailableModules(m_modules_array);
060
061      // Set enabled modules
062      for (String s: pref.getActiveModules()) {
063        m_modules.get(s).setModuleIsEnabled(true);
064      }
065
066      // Reorder modules
067      // First add enabled modules
068      Map<String, TracerModule> m_bckModules = new LinkedHashMap<String, TracerModule>(m_modules);
069      m_modules.clear();
070      for (String s: pref.getActiveModules()) {
071        m_modules.put(s, m_bckModules.get(s));
072      }
073
074      // Second: add remaining modules
075      Iterator<Map.Entry<String, TracerModule>> it = m_bckModules.entrySet().iterator();
076      while (it.hasNext()) {
077        Map.Entry<String, TracerModule> module = it.next();
078        if (!m_modules.containsValue(module.getValue())) {
079          m_modules.put(module.getKey(), module.getValue());
080        }
081      }
082
083      countActiveModules();
084
085      if (m_activeModulesCount == 0) {
086        return;
087      }
088
089      m_it = m_modules.entrySet().iterator();
090      m_active_module = m_it.next();
091      while (!m_active_module.getValue().moduleIsEnabled()) {
092        m_active_module = m_it.next();
093      }
094
095    }
096
097    private void countActiveModules() {
098
099      m_activeModulesCount = 0;
100
101      Iterator<Map.Entry<String, TracerModule>> it = m_modules.entrySet().iterator();
102      while (it.hasNext()) {
103        if (it.next().getValue().moduleIsEnabled()) {
104          m_activeModulesCount++;
105        }
106      }
107    }
108
109// -----------------------------------------------------------------
110
111    public Modules () {
112      init();
113    }
114
115    public void refreshModulesStatus() {
116
117      pref.reloadSettings();
118
119      // Reorder modules
120      // First add enabled modules
121      Map<String, TracerModule> m_bckModules = new LinkedHashMap<String, TracerModule>(m_modules);
122      m_modules.clear();
123      for (String s: pref.getActiveModules()) {
124        m_modules.put(s, m_bckModules.get(s));
125        m_modules.get(s).setModuleIsEnabled(true);
126      }
127
128      // Second: add remaining modules
129      Iterator<Map.Entry<String, TracerModule>> it = m_bckModules.entrySet().iterator();
130      while (it.hasNext()) {
131        Map.Entry<String, TracerModule> module = it.next();
132        if (!m_modules.containsValue(module.getValue())) {
133          m_modules.put(module.getKey(), module.getValue());
134          m_modules.get(module.getKey()).setModuleIsEnabled(false);
135        }
136      }
137
138      countActiveModules();
139
140      if (m_activeModulesCount == 0) {
141        return;
142      }
143
144      String m_bckKey= new String(m_active_module.getKey());
145      m_it = m_modules.entrySet().iterator();
146      m_active_module = m_it.next();
147      if (m_modules.containsKey(m_bckKey) && pref.isModuleEnabled(m_bckKey)) {
148        while (!m_active_module.getKey().equals(m_bckKey)) {
149          m_active_module = m_it.next();
150        }
151      }
152
153    }
154
155    public String getActiveModuleName() {
156      if (m_active_module == null) {
157        return null;
158      }
159      return m_active_module.getValue().getName();
160    }
161
162    public TracerModule getActiveModule() {
163      if (m_active_module == null) {
164        return null;
165      }
166      return m_active_module.getValue();
167    }
168
169    public TracerModule nextModule() {
170      while (true) {
171        if (m_it.hasNext()) {
172          m_active_module = m_it.next();
173          if (m_active_module.getValue().moduleIsEnabled())
174            break;
175        } else {
176          m_it = m_modules.entrySet().iterator();
177          m_active_module = m_it.next();
178          if (m_active_module.getValue().moduleIsEnabled())
179            break;
180        }
181      }
182      return m_active_module.getValue();
183    }
184
185    public void setActiveModule(String m) {
186      m_it = m_modules.entrySet().iterator();
187      m_active_module = null;
188
189      Map.Entry<String, TracerModule> ent;
190
191      while (m_it.hasNext()) {
192        ent = m_it.next();
193        if (ent.getKey() == m) {
194          m_active_module = ent;
195          break;
196        }
197      }
198    }
199
200    public int getActiveModulesCount() {
201      return m_activeModulesCount;
202    }
203}

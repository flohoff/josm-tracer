020package org.openstreetmap.josm.plugins.tracer;
021
022
023import java.io.BufferedReader;
024import java.io.IOException;
025import java.io.InputStreamReader;
026import java.net.MalformedURLException;
027import java.net.URL;
028import java.net.URLConnection;
029import javax.swing.JOptionPane;
030import org.openstreetmap.josm.gui.Notification;
031import org.openstreetmap.josm.gui.util.GuiHelper;
032
033
034public abstract class TracerUtils {
035
036    /**
037     * Show notification for 3 seconds. Runs in EDT thread.
038     * @param message Message to shown.
039     * @param type Type of message (info, warning, error, plain).
040     */
041    public static void showNotification (String message, String type) {
042      showNotification(message, type, Notification.TIME_SHORT);
043    }
044
045    /**
046     * Show notification. Runs in EDT thread.
047     * @param message Message to shown.
048     * @param type Type of message (info, warning, error, plain).
049     * @param time How long will be the message displayed.
050     */
051    public static void showNotification (final String message, final String type, final int time) {
052        GuiHelper.runInEDT(new Runnable() {
053            @Override
054            public void run() {
055
056                Notification note = new Notification(message);
057
058                if (type.equals("info"))
059                    note.setIcon(JOptionPane.INFORMATION_MESSAGE);
060                else if (type.equals("warning"))
061                    note.setIcon(JOptionPane.WARNING_MESSAGE);
062                else if (type.equals("error"))
063                    note.setIcon(JOptionPane.ERROR_MESSAGE);
064                else
065                    note.setIcon(JOptionPane.PLAIN_MESSAGE);
066
067                note.setDuration(time);
068                note.show();
069            }
070        });
071    }
072
073    /**
074     * Convert date from Czech to OSM format
075     * @param ruianDate Date in RUIAN (Czech) format DD.MM.YYYY
076     * @return String with date converted to OSM data format YYYY-MM-DD
077     */
078    public static String convertDate (String ruianDate) {
079        try {
080            String[] parts = ruianDate.split("\\.");
081            int day =   Integer.parseInt(parts[0]);
082            int month = Integer.parseInt(parts[1]);
083            int year =  Integer.parseInt(parts[2]);
084            return Integer.toString(year) + "-" + String.format("%02d", month) + "-" + String.format("%02d", day);
085        } catch (NumberFormatException e) {
086            return "";
087        }
088    }
089
090    private final static int defaultStreamTimeout = 10000;
091
092    public static BufferedReader openUrlStream (String url, int timeout) throws IOException {
093        return openUrlStream (url, timeout, null);
094    }
095
096    public static BufferedReader openUrlStream (String url) throws IOException {
097        return openUrlStream (url, defaultStreamTimeout, null);
098    }
099
100    public static BufferedReader openUrlStream (String url, String charset) throws IOException {
101        return openUrlStream (url, defaultStreamTimeout, charset);
102    }
103
104    public static BufferedReader openUrlStream (String url, int timeout, String charset) throws MalformedURLException, IOException {
105        URLConnection conn = null;
106        boolean succeeded = false;
107        try {
108             conn = new URL(url).openConnection();
109
110             // set timeouts
111             conn.setConnectTimeout(timeout);
112             conn.setReadTimeout(timeout);
113
114             InputStreamReader isr = charset != null ?
115                     new InputStreamReader(conn.getInputStream(), charset) : new InputStreamReader(conn.getInputStream());
116             BufferedReader reader = new BufferedReader(isr);
117             succeeded = true;
118             return reader;
119        }
120        finally {
121            if (!succeeded && conn != null) {
122                try {
123                    conn.getInputStream().close();
124                }
125                catch (IOException e) {
126                }
127            }
128        }
129    }
130 }

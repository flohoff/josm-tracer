001package org.openstreetmap.josm.plugins.tracer;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.HashSet;
006import java.util.List;
007import java.util.Map;
008import java.util.Map.Entry;
009import java.util.Set;
010import org.openstreetmap.josm.data.osm.Node;
011import org.openstreetmap.josm.data.osm.OsmPrimitive;
012import org.openstreetmap.josm.data.osm.Relation;
013import org.openstreetmap.josm.data.osm.Tag;
014import org.openstreetmap.josm.data.osm.TagCollection;
015import org.openstreetmap.josm.gui.conflict.tags.CombinePrimitiveResolverDialog;
016import org.openstreetmap.josm.gui.conflict.tags.TagConflictResolutionUtil;
017import org.openstreetmap.josm.gui.MainApplication;
018
019import static org.openstreetmap.josm.tools.I18n.*;
020
021public abstract class CombineTagsResolver {
022
023    public static Map<String, String> launchIfNecessary(
024            final Map<String, String> old_keys,
025            final Map<String, String> new_keys) {
026
027        // Note: this method was copied from CombinePrimitiveResolverDialog.launchIfNecessary() and
028        // adapted to resolution of tag sets only. It's not possible to use the original launchIfNecessary()
029        // because (a) it requires primitives occurring in DataSet, (b) it tries to resolve relation
030        // memberships, (c) it builds a list of DataSet Commands instead of the resolved set of tags.
031
032        // Prepare tags - for collision tags put theirs source
033        Map<String, String> m_old_keys = new HashMap<>(old_keys);
034        Map<String, String> m_new_keys = new HashMap<>(new_keys);
035
036        String m_spacer = "    .. ";
037        String m_osm =    tr("/osm/");
038        String m_traced = tr("/traced/");
039
040        for (Entry<String, String> entry : old_keys.entrySet())
041        {
042            if (new_keys.containsKey(entry.getKey())) {
043                // Both lists contains the same key
044                if (new_keys.get(entry.getKey()).equals(entry.getValue())) {
045                    // ... and values of this key are equal
046                    m_old_keys.put(entry.getKey(), entry.getValue());
047                    m_new_keys.put(entry.getKey(), entry.getValue());
048                } else {
049                    m_old_keys.put(entry.getKey(), entry.getValue() + m_spacer + m_osm);
050                    m_new_keys.put(entry.getKey(), new_keys.get(entry.getKey()) + m_spacer + m_traced);
051                }
052            } else {
053                // Unique key in old_keys
054                m_old_keys.put(entry.getKey(), entry.getValue());
055            }
056        }
057
058        // Finish update from new_keys
059        for (Entry<String, String> entry : new_keys.entrySet())
060        {
061            if (!m_new_keys.containsKey(entry.getKey())) {
062                m_new_keys.put(entry.getKey(), entry.getValue());
063            }
064        }
065        // Setup collection of all tags
066        TagCollection tagsOfPrimitives = TagCollection.from(m_old_keys);
067        tagsOfPrimitives.add(TagCollection.from(m_new_keys));
068
069        // Create faked primitives for resolution functions
070        OsmPrimitive prim1 = new Node();
071        OsmPrimitive prim2 = new Node();
072        prim1.setKeys(m_old_keys);
073        prim2.setKeys(m_new_keys);
074        List<OsmPrimitive> primitives = new ArrayList<>(2);
075        primitives.add(prim1);
076        primitives.add(prim2);
077
078        // Setup tag collections for resolution dialog
079        final TagCollection completeWayTags = new TagCollection(tagsOfPrimitives);
080        TagConflictResolutionUtil.applyAutomaticTagConflictResolution(completeWayTags);
081        TagConflictResolutionUtil.normalizeTagCollectionBeforeEditing(completeWayTags, primitives);
082        final TagCollection tagsToEdit = new TagCollection(completeWayTags);
083        TagConflictResolutionUtil.completeTagCollectionForEditing(tagsToEdit);
084
085        // Fake relations, we don't want to edit any relation memberships
086        final Set<Relation> parentRelations = new HashSet<>();
087
088        // Build conflict resolution dialog
089        final CombinePrimitiveResolverDialog dialog = new CombinePrimitiveResolverDialog (MainApplication.getMainFrame());
090
091        dialog.getTagConflictResolverModel().populate(tagsToEdit, completeWayTags.getKeysWithMultipleValues());
092        dialog.getRelationMemberConflictResolverModel().populate(parentRelations, primitives);
093        dialog.prepareDefaultDecisions();
094        dialog.setTargetPrimitive(null); // Use universal dialog title. Setting targetPrimitive to null seems to be safe if dialog.buildResolutionCommands() function isn't called.
095
096        // Resolve tag conflicts if necessary
097        if (!dialog.isResolvedCompletely()) {
098            dialog.setVisible(true);
099            if (!dialog.isApplied()) {
100                return null;
101            }
102        }
103
104        // Create final set
105        Map<String, String> result = new HashMap<>(new_keys);
106        TagCollection resolution = dialog.getTagConflictResolverModel().getAllResolutions();
107        for (Tag tag: resolution) {
108            result.put(tag.getKey(), tag.getValue().replaceAll("    \\.\\. "+ m_osm,"").replaceAll("    \\.\\. " + m_traced,""));
109        }
110
111        return result;
112    }
113}

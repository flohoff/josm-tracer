022import java.util.List;
023import org.openstreetmap.josm.data.coor.EastNorth;

031public abstract class GeomUtils {
032
033    /**
034     * Calculates approximate planar area determined by the given polygon,
035     * in east/north coordinates. Relies on current projection and works
036     * only for small areas. Use it only for relative comparisons of areas!
037     *
038     * Both closed and non-closed polygons are supported.
039     *
040     * @param nodes List of points defining the polygon
041     * @return Calculated area
042     */
043    public static double getEastNorthArea(List<EdNode> nodes) {
044
045        int count = nodes.size();
046
047        // for closed ways, ignore last node
048        if (count > 1 && nodes.get(0).equals(nodes.get(count - 1))) {
049            --count;
050        }
051
052        if (count <= 2) {
053            return 0;
054        }
055
056        double area = 0;
057        EastNorth ej = nodes.get(count - 1).getEastNorth();
058        for (int i = 0; i < count; ++i) {
059            EastNorth ei = nodes.get(i).getEastNorth();
060            area += (ej.getX() + ei.getX()) * (ej.getY() - ei.getY());
061            ej = ei;
062        }
063        return Math.abs(area / 2);
064    }
065
066    /**
067     * Calculates unoriented angle between two line segments, in radians.
068     * The angle is in range 0 .. PI.
069     * @param p0 First point of the first segment
070     * @param p1 Vertex point, shared by both segments
071     * @param p2 Second point of the second segment
072     * @return Angle in radians
073     */
074    public static double unorientedAngleBetween (EdNode p0, EdNode p1, EdNode p2) {
075        return unorientedAngleBetween(p0.currentNodeUnsafe(), p1.currentNodeUnsafe(), p2.currentNodeUnsafe());
076    }
077
078    /**
079     * Calculates unoriented angle between two line segments, in radians.
080     * The angle is in range 0 .. PI.
081     * @param p0 First point of the first segment
082     * @param p1 Vertex point, shared by both segments
083     * @param p2 Second point of the second segment
084     * @return Angle in radians
085     */
086    public static double unorientedAngleBetween(Node p0, Node p1, Node p2) {
087        double a1 = p1.getCoor().bearing(p0.getCoor());
088        double a2 = p1.getCoor().bearing(p2.getCoor());
089        double angle = Math.abs(a2 - a1) % (2 * Math.PI);
090        if (angle < 0)
091            angle += 2 * Math.PI;
092        if (angle > Math.PI)
093            angle = (2 * Math.PI) - angle;
094        return angle;
095    }
096
097    /**
098     * Calculates distance of a point to a line segment.
099     * Distance is calculated as a great circle distance between <code>point</code>
100     * and the closest point within the line segment.
101     *
102     * @param point Point for which the minimal distance is calculated
103     * @param segp1 First point determining the line segment
104     * @param segp2 Second point determining the line segment
105     * @return Distance in meters
106     */
107    public static double distanceToSegmentMeters(EdNode point, EdNode segp1, EdNode segp2) {
108        return distanceToSegmentMeters(point.currentNodeUnsafe(), segp1.currentNodeUnsafe(), segp2.currentNodeUnsafe());
109    }
110
111    /**
112     * Calculates distance of a point to a line segment.
113     * Distance is calculated as a great circle distance between <code>point</code>
114     * and the closest point within the line segment.
115     *
116     * @param point Point for which the minimal distance is calculated
117     * @param segp1 First point determining the line segment
118     * @param segp2 Second point determining the line segment
119     * @return Distance in meters
120     */
121    public static double distanceToSegmentMeters(Node point, Node segp1, Node segp2) {
122        EastNorth cp = Geometry.closestPointToSegment(
123                segp1.getEastNorth(), segp2.getEastNorth(), point.getEastNorth());
124        return point.getCoor().greatCircleDistance(ProjectionRegistry.getProjection().eastNorth2latlon(cp));
125    }
126
127    /**
128     * Returns distance and angle deviations of a point from a line segment.
129     * Distance deviation is the minimal distance of the point to the line. Angle
130     * deviation is the greater of unoriented vertex angles (point, segp1, segp2) and
131     * (point, segp2, segp1).
132     * @param point Point for which the deviation is calculated
133     * @param segp1 First point determining the line segment
134     * @param segp2 Second point determining the line segment
135     * @return Calculated GeomDeviation
136     */
137    public static GeomDeviation pointDeviationFromSegment(EdNode point, EdNode segp1, EdNode segp2) {
138        return pointDeviationFromSegment(point.currentNodeUnsafe(), segp1.currentNodeUnsafe(), segp2.currentNodeUnsafe());
139    }
140
141    /**
142     * Returns distance and angle deviations of a point from a line segment.
143     * Distance deviation is the minimal distance of the point to the line. Angle
144     * deviation is the greater of unoriented vertex angles (point, segp1, segp2) and
145     * (point, segp2, segp1).
146     * @param point Point for which the deviation is calculated
147     * @param segp1 First point determining the line segment
148     * @param segp2 Second point determining the line segment
149     * @return Calculated GeomDeviation
150     */
151    public static GeomDeviation pointDeviationFromSegment(Node point, Node segp1, Node segp2) {
152        EastNorth ep = point.getEastNorth();
153        EastNorth ex = segp1.getEastNorth();
154        EastNorth ey = segp2.getEastNorth();
155
156        EastNorth cp = Geometry.closestPointToSegment(ex, ey, ep);
157        double dev_distance_meters = point.getCoor().greatCircleDistance(ProjectionRegistry.getProjection().eastNorth2latlon(cp));
158
159        double a1 = GeomUtils.unorientedAngleBetween(segp1, segp2, point);
160        double a2 = GeomUtils.unorientedAngleBetween(segp2, segp1, point);
161        double dev_angle = Math.max(a1, a2);
162
163        return new GeomDeviation (dev_distance_meters, dev_angle);
164    }
165
166    public static double distanceOfNodesMeters(EdNode x, EdNode y) {
167        return x.getCoor().greatCircleDistance(y.getCoor());
168    }
169
170    private static LatLon roundCoor(LatLon coor, double precision) {
171        if (precision <= 0)
172            return coor.getRoundedToOsmPrecision();
173        return new LatLon(
174            Math.round(coor.lat() / precision) * precision,
175            Math.round(coor.lon() / precision) * precision);
176    }
177
178    public static boolean duplicateNodes(LatLon l1, LatLon l2, double precision) {
179        return (l1 == l2) ||
180            (l1 != null && l2 != null && roundCoor(l1, precision).equals(roundCoor(l2, precision)));
181    }
182
183    public static double duplicateNodesPrecision() {
184        return Config.getPref().getDouble("validator.duplicatenodes.precision", 0.0);
185    }
186
187    public static double getMetersPerDegreeOfLatitude (LatLon ll) {
188        double phi = Math.toRadians(ll.lat());
189        return 111132.954 - 559.822 * Math.cos(2 * phi) + 1.175 * Math.cos(4 * phi);
190    }
191
192    public static double getMetersPerDegreeOfLongitude(LatLon ll) {
193        double phi = Math.toRadians(ll.lat());
194        double a = Ellipsoid.WGS84.a;
195        double e2 = Ellipsoid.WGS84.e2;
196        double v1 = Math.PI * a * Math.cos(phi);
197        double sinphi = Math.sin(phi);
198        double v2 = 180.0 * Math.sqrt(1 - e2 * sinphi * sinphi);
199        return v1/v2;
200    }

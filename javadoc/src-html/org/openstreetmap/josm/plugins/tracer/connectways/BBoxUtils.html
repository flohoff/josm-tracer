020package org.openstreetmap.josm.plugins.tracer.connectways;
021
022import java.util.List;
023import org.openstreetmap.josm.data.Bounds;
024import org.openstreetmap.josm.data.coor.LatLon;
025import org.openstreetmap.josm.data.osm.BBox;
026
027public abstract class BBoxUtils {
028
029    public static void extendBBox(BBox box, LatLonSize oversize) {
030        box.add(box.getTopLeftLon() - oversize.lonSize(), box.getBottomRightLat() - oversize.latSize());
031        box.add(box.getBottomRightLon() + oversize.lonSize(), box.getTopLeftLat() + oversize.latSize());
032    }
033
034    public static boolean isInsideBounds(LatLon p, List<Bounds> bounds, LatLonSize extrasize) {
035        if (extrasize.isZero()) {
036            return isInsideBounds(p, bounds);
037        }
038
039        // check given point and four corners around it specified by extrasize
040        LatLon p1 = new LatLon(p.lat() - extrasize.latSize(), p.lon() - extrasize.lonSize());
041        LatLon p2 = new LatLon(p.lat() - extrasize.latSize(), p.lon() + extrasize.lonSize());
042        LatLon p3 = new LatLon(p.lat() + extrasize.latSize(), p.lon() - extrasize.lonSize());
043        LatLon p4 = new LatLon(p.lat() + extrasize.latSize(), p.lon() + extrasize.lonSize());
044        return isInsideBounds(p, bounds) && isInsideBounds(p1, bounds) &&
045                isInsideBounds(p2, bounds) && isInsideBounds(p3, bounds) &&
046                isInsideBounds(p4, bounds);
047    }
048
049    private static boolean isInsideBounds(LatLon p, List<Bounds> bounds) {
050        for (Bounds b: bounds) {
051            if (b.contains(p))
052                return true;
053        }
054        return false;
055    }
056}

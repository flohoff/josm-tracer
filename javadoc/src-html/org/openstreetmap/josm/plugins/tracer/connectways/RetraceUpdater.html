022import java.util.List;
023import java.util.Map;

029public class RetraceUpdater {
030
031    private final boolean m_convertOldStyleMultipolygons;
032    private final PostTraceNotifications m_postTraceNotifications;
033
034    public RetraceUpdater(boolean convert_oldstyle_multipolygons, PostTraceNotifications posttrace_notifications) {
035        m_convertOldStyleMultipolygons = convert_oldstyle_multipolygons;
036        m_postTraceNotifications = posttrace_notifications;
037    }
038
039    public EdObject updateRetracedObjects(EdObject new_object, EdObject retrace_object) {
040
041        boolean retrace_is_simple_way = retrace_object.isWay() && !retrace_object.hasReferrers();
042        boolean retrace_is_inner_way = retrace_object.isWay() && retrace_object.hasReferrers();
043        boolean new_is_way = new_object.isWay();
044        boolean new_is_multipolygon = new_object.isMultipolygon();
045
046        // Disallow retrace if it disconnects an important single-referrer node and makes it an orphan
047        EdNode dn = findAnyImportantNodeToBeOrphaned (new_object, retrace_object);
048        if (dn != null) {
049            m_postTraceNotifications.add(tr("Retrace would disconnect an important node {0}, this is not supported.", dn.getUniqueId()));
050            return null;
051        }
052
053        // Simple way -> Simple way
054        if ((retrace_is_simple_way || retrace_is_inner_way ) && new_is_way) {
055            EdWay outer_way = (EdWay) new_object;
056            EdWay retrace_way = (EdWay) retrace_object;
057            retrace_way.setNodes(outer_way.getNodes());
058            outer_way = retrace_way;
059            return outer_way;
060        }
061
062        // Simple way -> Multipolygon
063        // Move all non-linear tags from way to multipolygon, use retraced way as outer way of the new multipolygon
064        if ((retrace_is_simple_way || retrace_is_inner_way ) && new_is_multipolygon) {
065            EdWay retrace_way = (EdWay) retrace_object;
066            EdMultipolygon multipolygon = (EdMultipolygon)new_object;
067            multipolygon.moveAllNonLinearTagsFrom(retrace_way);
068            EdWay outer_way = multipolygon.outerWays().get(0);
069            multipolygon.removeOuterWay(outer_way);
070            retrace_way.setNodes(outer_way.getNodes());
071            outer_way = retrace_way;
072            multipolygon.addOuterWay(outer_way);
073            return multipolygon;
074        }
075
076        // Multipolygon -> Multipolygon
077        if (retrace_object.isMultipolygon() && new_is_multipolygon) {
078            EdMultipolygon retrace_multipolygon = (EdMultipolygon)retrace_object;
079            EdMultipolygon new_multipolygon = (EdMultipolygon)new_object;
080            EdObject res = updateRetracedMultipolygons(new_multipolygon, retrace_multipolygon);
081            if (res != null)
082                return res;
083        }
084
085        // Multipolygon -> Simple way
086        // - Move all non-linear tags from multipolygon to way
087        // - Use retraced way instead of outer way
088        // - Remove inner ways from existing multipolygon and delete untaged ways
089        // - Remove existing multipolygon
090        if (retrace_object.isMultipolygon() && new_is_way) {
091            EdMultipolygon retrace_multipolygon = (EdMultipolygon)retrace_object;
092            EdWay way = (EdWay)new_object;
093            EdWay outer_way = retrace_multipolygon.outerWays().get(0);
094            outer_way.setNodes(way.getNodes());
095            way=outer_way;
096            Map<String,String> keys = retrace_multipolygon.getKeys();
097            keys.remove("type");
098            way.setKeys(keys);
099            retrace_multipolygon.removeAllWays();
100            retrace_multipolygon.deleteShallow();
101
102            return way;
103        }
104
105        m_postTraceNotifications.add(tr("This kind of multipolygon retrace is not supported."));
106        return null;
107    }
108
109    private EdObject updateRetracedMultipolygons(EdMultipolygon new_multipolygon, EdMultipolygon retrace_multipolygon) {
110
111//         // don't retrace non-standalone multipolygons
112//         boolean retrace_is_standalone = !retrace_multipolygon.hasReferrers() && retrace_multipolygon.allWaysHaveSingleReferrer();
113//         if (!retrace_is_standalone)
114//             return null;
115
116        // don't retrace multipolygons with nonclosed ways
117        boolean retrace_is_closed = !retrace_multipolygon.containsNonClosedWays();
118        if (!retrace_is_closed) {
119            System.out.println("Multipolygon " + retrace_multipolygon.getUniqueId() + " contains non-closed ways.");
120            m_postTraceNotifications.add(tr("Multipolygon {0} contains non-closed ways.", retrace_multipolygon.getUniqueId()));
121            return null;
122        }
123
124        // try to convert old-style multipolygon to new-style
125        if (m_convertOldStyleMultipolygons && retrace_multipolygon.containsTaggedWays() && Config.getPref().getBoolean("multipoly.movetags", true))
126            retrace_multipolygon.removeTagsFromWaysIfNeeded();
127
128        // agressive retrace of multipolygon with untagged ways
129        if (!retrace_multipolygon.containsTaggedWays()) {
130            updateRetracedMultipolygonWaysAgressive(retrace_multipolygon, new_multipolygon, true);
131            updateRetracedMultipolygonWaysAgressive(retrace_multipolygon, new_multipolygon, false);
132            new_multipolygon.deleteShallow();
133            return retrace_multipolygon;
134        }
135
136        // other possibilites not supported yet
137        return null;
138    }
139
140    private void updateRetracedMultipolygonWaysAgressive(EdMultipolygon retrace_multipolygon, EdMultipolygon new_multipolygon, boolean out) {
141        // #### use pairing based on way area intersection?
142
143        List<EdWay> retraces = out ? retrace_multipolygon.outerWays() : retrace_multipolygon.innerWays();
144        List<EdWay> news = out ? new_multipolygon.outerWays() : new_multipolygon.innerWays();
145        int ridx = 0;
146
147        for (EdWay new_way : news) {
148            // update geometry of existing way
149            if (ridx < retraces.size()) {
150                EdWay retrace_way = retraces.get(ridx++);
151                retrace_way.setNodes(new_way.getNodes());
152            }
153            // no more existing ways available, add new way
154            else if (out)
155                retrace_multipolygon.addOuterWay(new_way);
156            else
157                retrace_multipolygon.addInnerWay(new_way);
158        }
159
160        // remove unused old ways if the new multipolygon has less ways than the old one
161        for (int i = ridx; i < retraces.size(); i++) {
162            if (out)
163                retrace_multipolygon.removeOuterWay(retraces.get(i));
164            else
165                retrace_multipolygon.removeInnerWay(retraces.get(i));
166        }
167    }
168
169    private EdNode findAnyImportantNodeToBeOrphaned(EdObject new_object, EdObject retrace_object) {
170        Set<EdNode> old_nodes = retrace_object.getAllNodes();
171        Set<EdNode> new_nodes = null;
172        for (EdNode node: old_nodes) {
173            // untagged nodes aren't important
174            if (!node.isTagged())
175                continue;
176            if (new_nodes == null)
177                new_nodes = new_object.getAllNodes();
178            // node is included in new geometry too, fine
179            if (new_nodes.contains(node))
180                continue;
181            // node has more than one referrer, it won't be orphaned
182            if (!node.hasSingleReferrer())
183                continue;
184            return node;
185        }
186        return null;
187    }

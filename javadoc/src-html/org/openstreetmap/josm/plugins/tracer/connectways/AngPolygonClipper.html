089    @SuppressWarnings("CallToPrintStackTrace")
090    public void polygonDifference (EdObject clip, EdObject subj) {
091
092        // initialize collections
093        m_outers = new ArrayList<>();
094        m_inners = new ArrayList<>();
095        m_nodesMap = new HashMap<>();
096        m_subjectNodes = new HashSet<>();
097
098        m_DiscardedPercent = 0.0;
099
100        double subj_area = Double.NaN;
101        if (m_DiscardCutoffsPercent > 0.0)
102            subj_area = subj.getEastNorthArea();
103
104        try {
105            // Note: always preserve collinear nodes! Otherwise, clipper disconnects
106            // non-intersecting touching nodes of the subject from other polygons!
107
108            Clipper clipper = new Clipper(Clipper.ioStrictlySimple + Clipper.ioPreserveCollinear);
109            clipper.addPaths(edObjectToPaths(clip, false), PolyType.ptClip, true);
110            clipper.addPaths(edObjectToPaths(subj, true), PolyType.ptSubject, true);
111
112            PolyTree ptree = new PolyTree();
113            clipper.execute(ClipType.ctDifference, ptree);
114
115            List<PolyNode> pnodes = ptree.getChilds();
116            for (PolyNode pn: pnodes) {
117                processPolyNode(pn, m_outers, m_inners, subj_area);
118            }
119        }
120        catch (ClipperException e) {
121            e.printStackTrace();
122            m_nodesMap = null;
123            m_outers = null;
124            m_inners = null;
125            throw new AssertionError("AngPolygonClipper.polygonDifference failed, ClipperException", e);
126        }
127
128        m_nodesMap = null;
129        m_outers = Collections.unmodifiableList(m_outers);
130        m_inners = Collections.unmodifiableList(m_inners);
131    }

133    public boolean changesOutsideDataBounds() {
134        if (m_subjectNodes == null || m_outers == null || m_inners == null)
135            throw new IllegalStateException();
136
137        List<Bounds> bounds = m_editor.getDataSet().getDataSourceBounds();
138        Set<EdNode> cur_nodes = new HashSet<>();
139
140        // Test new nodes not occurring in subject nodes
141        for (List<EdNode> list: m_outers) {
142            for (EdNode n: list) {
143                if (!m_subjectNodes.contains(n))
144                    if (!n.isInsideBounds(bounds, LatLonSize.Zero)) {
145                        System.out.println("Clip adds new outer node outside downloaded area: " + Long.toString(n.getUniqueId()) + ", " + n.getCoor().toDisplayString());
146                        return true;
147                    }
148                cur_nodes.add(n);
149            }
150        }
151        for (List<EdNode> list: m_inners) {
152            for (EdNode n: list) {
153                if (!m_subjectNodes.contains(n))
154                    if (!n.isInsideBounds(bounds, LatLonSize.Zero)) {
155                        System.out.println("Clip adds new inner node outside downloaded area: " + Long.toString(n.getUniqueId()) + ", " + n.getCoor().toDisplayString());
156                        return true;
157                    }
158                cur_nodes.add(n);
159            }
160        }
161
162        // Test removed subject nodes not occurring in current nodes
163        for (EdNode node: m_subjectNodes) {
164            if (!cur_nodes.contains(node))
165                if (!node.isInsideBounds(bounds, LatLonSize.Zero)) {
166                    System.out.println("Clip removes node outside downloaded area: " + Long.toString(node.getUniqueId()) + ", " + node.getCoor().toDisplayString());
167                    return true;
168                }
169        }
170
171        return false;
172    }

174    private void processPolyNode(PolyNode pn, List<List<EdNode>> aouters, List<List<EdNode>> ainners, double subj_area) {
175
176        List<List<EdNode>> outers = aouters;
177        List<List<EdNode>> inners = ainners;
178
179        Path path = pn.getContour();
180        boolean hole = pn.isHole();
181        boolean discard_cutoffs = !hole && m_DiscardCutoffsPercent > 0.0 && subj_area > 0.0;
182
183        if (discard_cutoffs) {
184            inners = new ArrayList<>();
185            outers = new ArrayList<>();
186        }
187
188        List<EdNode> nodes = pathToEdNodes(path);
189        if (nodes.size() > 3) {
190            if (hole)
191                inners.add(Collections.unmodifiableList(nodes));
192            else
193                outers.add(Collections.unmodifiableList(nodes));
194        }
195
196        if (pn.getChildCount() > 0) {
197            List<PolyNode> children = pn.getChilds();
198            for (PolyNode child: children) {
199                processPolyNode(child, outers, inners, subj_area);
200            }
201        }
202
203        if (!discard_cutoffs)
204            return;
205
206        double area = getEastNorthArea(outers, inners);
207        double percent = (area/subj_area) * 100.0;
208        if (percent >= 0.0 && percent < m_DiscardCutoffsPercent) {
209            System.out.println("Discarding cutoff area, percent=" + Double.toString(percent));
210            m_DiscardedPercent += percent;
211            return;
212        }
213        else {
214            System.out.println("Cutoff out of limit, percent=" + Double.toString(percent));
215        }
216
217        aouters.addAll(outers);
218        ainners.addAll(inners);
219    }

221    private double getEastNorthArea(List<List<EdNode>> outers, List<List<EdNode>> inners) {
222        double area = 0.0;
223
224        for (List<EdNode> outer: outers) {
225            area += GeomUtils.getEastNorthArea(outer);
226        }
227
228        for (List<EdNode> inner: inners) {
229            area -= GeomUtils.getEastNorthArea(inner);
230        }
231
232        return area;
233    }

235    private List<EdNode> pathToEdNodes (Path p)
236    {
237        //System.out.println("d: poly:");
238        List<EdNode> list = new ArrayList<> ();
239
240        LatLon prev_coor = null;
241        for (Point2d point : p) {
242            EdNode node = point2dToNode(point);
243            // avoid two consecutive duplicate nodes ..x,x..
244            if (!GeomUtils.duplicateNodes(node.getCoor(), prev_coor, m_duplicateNodesPrecision)) {
245                list.add(node);
246                prev_coor = node.getCoor();
247                //System.out.println(" - d: node " + Long.toString(node.getUniqueId()));
248            }
249        }
250
251        // Avoid consecutive duplicate nodes around the end of polygon
252        while ((list.size() >= 2) &&
253               (GeomUtils.duplicateNodes(list.get(0).getCoor(), list.get(list.size() - 1).getCoor(), m_duplicateNodesPrecision)))
254            list.remove(list.size() - 1);
255
256        boolean changed;
257        int i;
258        do {
259            changed = false;
260            // Remove false degnerated tails of the form ..x,y,z.., where either "z" is
261            // on line "xy" or "x" is on line "yz".
262            i = 0;
263            while ((list.size() >= 3) && i < list.size ()) {
264                int i1 = (i + 1) % list.size();
265                int i2 = (i + 2) % list.size();
266                EdNode p0 = list.get(i);
267                EdNode p1 = list.get(i1);
268                EdNode p2 = list.get(i2);
269                if (GeomUtils.pointDeviationFromSegment(p2, p0, p1).inTolerance(m_tolerance) ||
270                    GeomUtils.pointDeviationFromSegment(p0, p1, p2).inTolerance(m_tolerance)) {
271                    list.remove(i1);
272                    i = i >= 1 ? i - 1 : 0;
273                    changed = true;
274                }
275                else {
276                    i++;
277                }
278            }
279
280            // Remove degenerated tails of the form ..x,y,x..
281            i = 0;
282            while ((list.size() >= 3) && i < list.size ()) {
283                int i1 = (i + 1) % list.size();
284                int i2 = (i + 2) % list.size();
285                if (GeomUtils.duplicateNodes(list.get(i).getCoor(), list.get(i2).getCoor(), m_duplicateNodesPrecision)) {
286                    System.out.println(" x d: tail " + Long.toString(list.get(i).getUniqueId()));
287                    list.remove(i1);
288                    list.remove(i2 > i1 ? i1 : 0);
289                    i = i >= 3 ? i - 3 : 0;
290                    changed = true;
291                }
292                else {
293                    i++;
294                }
295            }
296        } while (changed);
297
298        // Polygon degenerated to two or one nodes
299        if (list.size() < 3) {
300            list.clear();
301            return list;
302        }
303
304        // Close the polygon
305        list.add(list.get(0));
306
307        return list;
308    }

311    private Path wayToPath(EdWay w, boolean issubj) {
312        if (!w.isClosed())
313            throw new IllegalArgumentException ("Way must be closed");
314
315        Path p = new Path();
316
317        for (int i = 0; i < w.getNodesCount(); i++)
318        {
319            EdNode node = w.getNode(i);
320            p.add (nodeToPoint2d(node));
321            if (issubj)
322                m_subjectNodes.add(node);
323        }
324        return p;
325    }

327    // #### improve! the scale must be derived from EastNorth precision, which depends on
328    // present JOSM projection! Or try to rewrite clipper to floating point...
329    private final static double fixedPointScale = 10000000000.0;
330
331    private Point2d nodeToPoint2d(EdNode node) {
332        EastNorth en = node.getEastNorth();
333        long x = (long)(en.getX() * fixedPointScale);
334        long y = (long)(en.getY() * fixedPointScale);
335        Point2d pt = new Point2d(x, y);
336        m_nodesMap.put(node.getCoor().getRoundedToOsmPrecision(), node);
337        return pt;
338    }
339
340    private EdNode point2dToNode(Point2d pt) {
341        // perform inverse projection to LatLon
342        double x = ((double)pt.X) / fixedPointScale;
343        double y = ((double)pt.Y) / fixedPointScale;
344        EastNorth en = new EastNorth (x,y);
345        LatLon ll = ProjectionRegistry.getProjection().eastNorth2latlon(en);
346
347        // lookup in LatLon map
348        EdNode node = m_nodesMap.get(ll.getRoundedToOsmPrecision());
349        if (node != null)
350            return node;
351
352        // create new node
353        node = m_editor.newNode(ll);
354        m_nodesMap.put(ll, node);
355        return node;
356    }

358    private Paths edObjectToPaths(EdObject obj, boolean issubj) {
359        if (obj.isWay()) {
360            Path p = wayToPath((EdWay)obj, issubj);
361            Paths pp = new Paths();
362            pp.add(p);
363            return pp;
364        }
365        if (obj.isMultipolygon())
366            return multipolygonToPaths((EdMultipolygon)obj, issubj);
367        throw new IllegalArgumentException("EdObject must be either EdWay or EdMultipolygon");
368    }
369
370    private Paths multipolygonToPaths(EdMultipolygon mp, boolean issubj) {
371        Paths pp = new Paths();
372        for (EdWay w: mp.outerWays())
373            pp.add(wayToPath(w, issubj));
374        for (EdWay w: mp.innerWays())
375            pp.add(wayToPath(w, issubj));
376        return pp;
377    }

020package org.openstreetmap.josm.plugins.tracer.connectways;
021
022import java.util.ArrayList;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Set;
026import org.openstreetmap.josm.data.Bounds;
027import org.openstreetmap.josm.data.coor.EastNorth;
028import org.openstreetmap.josm.data.coor.LatLon;
029import org.openstreetmap.josm.data.osm.BBox;
030import org.openstreetmap.josm.data.osm.Node;
031import org.openstreetmap.josm.data.osm.OsmPrimitive;
032import org.openstreetmap.josm.data.osm.Way;
033import static org.openstreetmap.josm.tools.I18n.tr;
034
035
036public class EdNode extends EdObject {
037
038    private Node m_node;
039
040    EdNode (WayEditor editor, LatLon latlon) {
041        super(editor, null);
042        m_node = new Node(latlon);
043    }
044
045    EdNode (WayEditor editor, Node original_node) {
046        super(editor, original_node);
047        m_node = new Node(original_node);
048    }
049
050    @Override
051    protected OsmPrimitive currentPrimitive() {
052        return m_node;
053    }
054
055    void forgeOriginalNodeDangerous (Node original_node) {
056        checkEditable();
057        super.forgeOriginalDangerous(original_node);
058        Node n = new Node(original_node);
059        n.setKeys(m_node.getKeys());
060        n.setCoor(m_node.getCoor());
061        m_node = n;
062        setModified();
063    }
064
065    public Node originalNode() {
066        if (!hasOriginal())
067            throw new IllegalStateException(tr("EdNode has no original Node"));
068        return (Node)originalPrimitive();
069    }
070
071    public Node finalNode() {
072        checkNotDeleted();
073        if (!isFinalized()) {
074            if (hasOriginal() && !isModified())
075                m_node = originalNode();
076            setFinalized();
077        }
078        return m_node;
079    }
080
081    /**
082     * Returns a final Node that can be referenced in other EdObject's finalization.
083     * The main difference from finalNode() is that for a modified node, it doesn't
084     * return new final Node but -surprisingly- the original Node. Indeed, because
085     * JOSM's ChangeCommand does not replace the original object wit the new one but
086     * just copies it's contents, EdObjects must be finalized with references to
087     * the original Node. Command ordering in WayEditor.finalizeEdit() guarantees that
088     * the original Node is changed prior the referencing EdObject uses it.
089     * @return final Way
090     */
091    Node finalReferenceableNode() {
092        checkNotDeleted();
093        Node fin = finalNode();
094        if (hasOriginal() && isModified())
095            return originalNode();
096        else
097            return fin;
098    }
099
100
101    public LatLon getCoor() {
102        return m_node.getCoor();
103    }
104
105    public void setCoor(LatLon ll) {
106        checkEditable();
107        if (ll.equals(m_node.getCoor()))
108                return;
109        m_node.setCoor(ll);
110        setModified();
111    }
112
113    @Override
114    public BBox getBBox() {
115        return m_node.getBBox();
116    }
117
118    public EastNorth getEastNorth() {
119        return m_node.getEastNorth();
120    }
121
122    Node currentNodeUnsafe() {
123        checkNotDeleted();
124        // Be careful, never modify returned Node!!
125        return m_node;
126    }
127
128    @Override
129    void updateModifiedFlag() {
130        checkEditable();
131        if (!hasOriginal() || isDeleted() || !isModified())
132            return;
133        Node orig = originalNode();
134
135        if (orig.getUniqueId() != m_node.getUniqueId())
136            return;
137        if (!orig.getCoor().equals(m_node.getCoor()))
138            return;
139        if (!hasIdenticalKeys(orig))
140            return;
141        resetModified();
142    }
143
144    /**
145     * Returns all way referrers of this EdNode that match given
146     * area filter predicate. Non-edited matching ways are automatically
147     * included into node's WayEditor.
148     * @param filter area ways filter
149     * @return list of all EdWay referrers of this EdNode that match given predicate
150     */
151    List<EdWay> getAllAreaWayReferrers(IEdAreaPredicate filter) {
152        List<EdWay> result = new ArrayList<>();
153
154        List<EdWay> edways = this.getEditorReferrers(EdWay.class);
155        for (EdWay way: edways) {
156            if (filter.evaluate(way))
157                result.add(way);
158        }
159
160        List<Way> ways = this.getExternalReferrers(Way.class);
161        for (Way way: ways) {
162            if (filter.evaluate(way))
163                result.add(getEditor().useWay(way));
164        }
165
166        return result;
167    }
168
169    @Override
170    public boolean reuseExistingNodes(IEdNodePredicate filter) {
171        throw new UnsupportedOperationException("Not supported yet.");
172    }
173
174    @Override
175    public boolean connectExistingTouchingNodes(GeomDeviation tolerance, IEdNodePredicate filter) {
176        throw new UnsupportedOperationException("This operation is not supported for nodes.");
177    }
178
179    @Override
180    public boolean connectNonIncludedTouchingNodes(GeomDeviation tolerance, EdObject obj) {
181        throw new UnsupportedOperationException("This operation is not supported for nodes.");
182    }
183
184    @Override
185    public double getEastNorthArea() {
186        return 0.0;
187    }
188
189    @Override
190    protected void deleteContentsShallow() {
191        this.m_node.removeAll();
192    }
193
194    @Override
195    public boolean reuseNearNodes(IReuseNearNodePredicate reuse, IEdNodePredicate filter) {
196        throw new UnsupportedOperationException("Not supported yet.");
197    }
198
199    @Override
200    public Set<EdNode> getAllNodes() {
201        Set<EdNode> s = new HashSet<>();
202        s.add(this);
203        return s;
204    }
205
206    @Override
207    public Set<EdWay> getAllWays () {
208        return new HashSet<> ();
209    }
210
211    @Override
212    public boolean isInsideBounds(List<Bounds> bounds, LatLonSize extrasize) {
213        checkNotDeleted();
214        return BBoxUtils.isInsideBounds(m_node.getCoor(), bounds, extrasize);
215    }
216
217    @Override
218    public EdWay getFirstOuterWay () {
219        throw new UnsupportedOperationException("This operation is not supported for nodes.");
220    }
221}

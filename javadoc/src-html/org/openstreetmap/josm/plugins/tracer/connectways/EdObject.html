113    public void addRef(EdObject ref) {
114        if (m_refs == null) {
115            m_refs = ref;
116            return;
117        }
118        if (m_refs instanceof EdObject) {
119            if (m_refs != ref)
120                m_refs = new EdObject[] { (EdObject)m_refs, ref };
121            return;
122        }
123        for (EdObject r: (EdObject[]) m_refs) {
124            if (r == ref)
125                return;
126        }
127
128        EdObject[] refs = (EdObject[]) m_refs;
129        EdObject[] nrefs = new EdObject[refs.length + 1];
130        System.arraycopy(refs, 0, nrefs, 0, refs.length);
131        nrefs[refs.length] = ref;
132        m_refs = nrefs;
133    }
134
135    public void removeRef(EdObject ref) {
136        if (m_refs == null) {
137            return;
138        }
139        if (m_refs instanceof EdObject) {
140            if (m_refs == ref)
141                m_refs = null;
142            return;
143        }
144        EdObject[] refs = (EdObject[])m_refs;
145
146        int idx = -1;
147        for (int i = 0; i < refs.length; i++) {
148            if (refs[i] == ref) {
149                idx = i;
150                break;
151            }
152        }
153
154        if (idx == -1)
155            return;
156
157        if (refs.length == 2) {
158            m_refs = refs[1-idx];
159            return;
160        }
161
162        EdObject[] nrefs = new EdObject[refs.length - 1];
163        System.arraycopy(refs, 0, nrefs, 0, idx);
164        System.arraycopy(refs, idx + 1, nrefs, idx, nrefs.length - idx);
165        m_refs = nrefs;
166    }

260    /**
261     * Checks if the given EdObject is the only one referrer of this EdObject.
262     * That is, if the object has only this given editor referrer and no
263     * other external or editor referrers.
264     * @param referrer referrer to test
265     * @return true if the referrer is the only referrer of this EdObject
266     */
267    public boolean hasSingleReferrer(EdObject referrer) {
268        if (m_refs == null)
269            return false;
270        if (m_refs instanceof EdObject[])
271            return false;
272        if ((EdObject)m_refs != referrer)
273            return false;
274        if (this.hasExternalReferrers())
275            return false;
276        return true;
277    }
278
279    public boolean hasSingleReferrer() {
280        if (m_refs instanceof EdObject[])
281            return false;
282        int count = (m_refs == null) ? 0 : 1;
283        if (hasOriginal()) {
284            List<OsmPrimitive> parents = m_original.getReferrers();
285            for (OsmPrimitive parent: parents) {
286                if (getEditor().isEdited(parent))
287                    continue;
288                count++;
289                if (count > 1)
290                    return false;
291            }
292        }
293        return count == 1;
294    }
295
296    public <T extends EdObject> List<T> getEditorReferrers(Class<T> type) {
297        if (m_refs == null)
298            return Collections.emptyList();
299        if (m_refs instanceof EdObject) {
300            if (!type.isInstance(m_refs))
301                return Collections.emptyList();
302
303            List<T> result = new ArrayList<>(1);
304            result.add(type.cast(m_refs));
305            return result;
306        }
307
308        EdObject[] refs = (EdObject[])m_refs;
309        List<T> result = new ArrayList<> ();
310        for (EdObject obj: refs) {
311            if (type.isInstance(obj)) {
312                result.add(type.cast(obj));
313            }
314        }
315        return result;
316    }
317
318    public <T extends OsmPrimitive> List<T> getExternalReferrers(Class<T> type) {
319
320        if (!hasOriginal())
321            return Collections.emptyList();
322
323        List<OsmPrimitive> parents = m_original.getReferrers();
324        List<T> result = new ArrayList<> ();
325        for (OsmPrimitive parent: parents) {
326            if (!type.isInstance(parent))
327                continue;
328            if (getEditor().isEdited(parent))
329                continue;
330            result.add(type.cast(parent));
331        }
332        return result;
333    }
334
335    public boolean hasExternalReferrers () {
336        if (!hasOriginal())
337            return false;
338
339        List<OsmPrimitive> parents = m_original.getReferrers();
340        for (OsmPrimitive parent: parents) {
341            if (!getEditor().isEdited(parent))
342                return true;
343        }
344        return false;
345    }
346
347    public boolean hasReferrers() {
348        return hasEditorReferrers() || hasExternalReferrers();
349    }

364    public void moveAllNonLinearTagsFrom(EdObject src) {
365        checkNotDeleted();
366
367        Map<String, String> values = src.getKeys();
368
369        for (String linear_tag: Config.getPref().getList("multipoly.lineartagstokeep", Arrays.asList(new String[] {"barrier", "source"})))
370            values.remove(linear_tag);
371
372        if ("coastline".equals(values.get("natural")))
373            values.remove("natural");
374
375        for (Map.Entry<String, String> entry : values.entrySet()) {
376            this.put (entry.getKey(), entry.getValue());
377            src.remove (entry.getKey());
378        }
379    }

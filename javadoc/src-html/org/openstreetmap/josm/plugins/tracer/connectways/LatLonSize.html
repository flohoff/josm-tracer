020package org.openstreetmap.josm.plugins.tracer.connectways;
021
022import org.openstreetmap.josm.data.coor.LatLon;
023import org.openstreetmap.josm.data.osm.BBox;
024
025public class LatLonSize {
026    private final double m_lat;
027    private final double m_lon;
028
029    public LatLonSize(double lat, double lon) {
030        m_lat = lat;
031        m_lon = lon;
032    }
033
034    public double latSize() {
035        return m_lat;
036    }
037
038    public double lonSize() {
039        return m_lon;
040    }
041
042    public boolean isZero() {
043        return m_lat == 0 && m_lon == 0;
044    }
045
046    public static final LatLonSize Zero = new LatLonSize(0, 0);
047
048    public static LatLonSize get(LatLon ll, double radius_in_meters) {
049        double meters_per_degree_lat = GeomUtils.getMetersPerDegreeOfLatitude(ll);
050        double meters_per_degree_lon = GeomUtils.getMetersPerDegreeOfLongitude(ll);
051        double radius_lat = radius_in_meters / meters_per_degree_lat;
052        double radius_lon = radius_in_meters / meters_per_degree_lon;
053        return new LatLonSize(radius_lat, radius_lon);
054    }
055
056    public static LatLonSize get(BBox box, double radius_in_meters) {
057        double meters_per_degree_lat1 = GeomUtils.getMetersPerDegreeOfLatitude(box.getTopLeft());
058        double meters_per_degree_lon1 = GeomUtils.getMetersPerDegreeOfLongitude(box.getTopLeft());
059        double meters_per_degree_lat2 = GeomUtils.getMetersPerDegreeOfLatitude(box.getBottomRight());
060        double meters_per_degree_lon2 = GeomUtils.getMetersPerDegreeOfLongitude(box.getBottomRight());
061        double radius_lat = radius_in_meters / Math.min(meters_per_degree_lat1, meters_per_degree_lat2);
062        double radius_lon = radius_in_meters / Math.min(meters_per_degree_lon1, meters_per_degree_lon2);
063        return new LatLonSize(radius_lat, radius_lon);
064    }
065}

105    public EdNode useNode(Node node) {
106        if (node == null)
107            throw new IllegalArgumentException();
108        if (node.getUniqueId() <= m_node_idGuard)
109            throw new IllegalArgumentException(tr("Non-original Node created outside WayEditor scope"));
110        if (node.getDataSet() != this.getDataSet())
111            throw new IllegalArgumentException(tr("Cannot use Node from with a different/null DataSet"));
112        if (!node.isUsable())
113            throw new IllegalArgumentException(tr("Cannot edit unusable Node"));
114
115        EdNode en = m_originalNodes.get(node.getUniqueId());
116        if (en != null) {
117            if (en.originalNode() != node)
118                throw new IllegalArgumentException(tr("Original Node ID mapped to a different Node object"));
119            return en;
120        }
121
122        en = new EdNode(this, node);
123        m_originalNodes.put(node.getUniqueId(), en);
124        m_nodes.add(en);
125        return en;
126    }

128    public EdWay useWay(Way way) {
129        if (way == null)
130            throw new IllegalArgumentException();
131        if (way.getUniqueId() <= m_way_idGuard)
132            throw new IllegalArgumentException(tr("Non-original Way created outside WayEditor scope"));
133        if (way.getDataSet() != this.getDataSet())
134            throw new IllegalArgumentException(tr("Cannot use Way from with a different/null DataSet"));
135        if (!way.isUsable() || way.hasIncompleteNodes())
136            throw new IllegalArgumentException(tr("Cannot edit unusable Way"));
137
138        EdWay ew = m_originalWays.get(way.getUniqueId());
139        if (ew != null) {
140            if (ew.originalWay() != way)
141                throw new IllegalArgumentException(tr("Original Way ID mapped to a different Way object"));
142            return ew;
143        }
144
145        ew = new EdWay(this, way);
146        m_originalWays.put(way.getUniqueId(), ew);
147        m_ways.add(ew);
148        return ew;
149    }

151    public EdMultipolygon useMultipolygon(Relation rel) {
152        if (rel == null)
153            throw new IllegalArgumentException();
154        if (rel.getUniqueId() <= m_rel_idGuard)
155            throw new IllegalArgumentException(tr("Non-original Relation created outside WayEditor scope"));
156        if (rel.getDataSet() != this.getDataSet())
157            throw new IllegalArgumentException(tr("Cannot use Relation from with a different/null DataSet"));
158        if (!rel.isUsable() || rel.hasIncompleteMembers())
159            throw new IllegalArgumentException(tr("Cannot edit unusable/incomplete Relation"));
160        if (!rel.isMultipolygon())
161            throw new IllegalArgumentException(tr("Relation is not a multipolygon"));
162
163        EdMultipolygon emp = m_originalMultipolygons.get(rel.getUniqueId());
164        if (emp != null) {
165            if (emp.originalMultipolygon() != rel)
166                throw new IllegalArgumentException(tr("Original Relation ID mapped to a different Relation object"));
167            return emp;
168        }
169
170        emp = new EdMultipolygon(this, rel);
171        m_originalMultipolygons.put(rel.getUniqueId(), emp);
172        m_multipolygons.add(emp);
173        return emp;
174    }

176    boolean isEdited(OsmPrimitive prim) {
177        if (prim instanceof Node) {
178            EdNode en = m_originalNodes.get(prim.getUniqueId());
179            return en != null && en.originalNode() == prim;
180        }
181        if (prim instanceof Way) {
182            EdWay ew = m_originalWays.get(prim.getUniqueId());
183            return ew != null && ew.originalWay() == prim;
184        }
185        if (prim instanceof Relation) {
186            EdMultipolygon emp = m_originalMultipolygons.get(prim.getUniqueId());
187            return emp != null && emp.originalMultipolygon() == prim;
188        }
189        return false;
190    }

207    Set<EdNode> findExistingNodesTouchingWaySegment(GeomDeviation tolerance, LatLonSize oversize, EdNode x, EdNode y, IEdNodePredicate filter) {
208        Node nx = new Node(x.currentNodeUnsafe());
209        Node ny = new Node(y.currentNodeUnsafe());
210        BBox bbox = new BBox ((INode) nx);
211        bbox.addPrimitive (ny, 0);
212        BBoxUtils.extendBBox(bbox, oversize);
213
214        Set<EdNode> result = new HashSet<>();
215
216        // (1) edited nodes
217        for (EdNode ednd: searchEdNodes(bbox)) {
218            if (ednd.isDeleted())
219                continue;
220            if (!filter.evaluate(ednd))
221                continue;
222            if (!GeomUtils.pointDeviationFromSegment(ednd.currentNodeUnsafe(), nx, ny).inTolerance(tolerance))
223                continue;
224            result.add(ednd);
225        }
226
227        // (2) original nodes that are not tracked yet
228        for (Node nd : getDataSet().searchNodes(bbox)) {
229            if (!nd.isUsable() || nd.isOutsideDownloadArea())
230                continue;
231            if (isEdited(nd))
232                continue;
233            if (!filter.evaluate(nd))
234                continue;
235            if (!GeomUtils.pointDeviationFromSegment(nd, nx, ny).inTolerance(tolerance))
236                continue;
237            result.add(useNode(nd));
238        }
239
240        return result;
241    }

270    EdNode findExistingNodeForDuplicateMerge(EdNode src, IEdNodePredicate filter) {
271        if (src == null || src.isDeleted())
272            throw new IllegalArgumentException();
273
274        BBox bbox = src.getBBox(minimalOversize);
275
276        // (1) original nodes that are either not tracked or didn't move from original position;
277        //     prefer edited EdNodes, use a matching one with the highest ID
278        Node node1 = null;
279        EdNode ednode1 = null;
280        for (Node nd : getDataSet().searchNodes(bbox)) {
281            if (!nd.isUsable() || nd.isOutsideDownloadArea())
282                continue;
283            EdNode ednd = m_originalNodes.get(nd.getUniqueId());
284            if (ednd != null) {
285                if (ednd.isDeleted())
286                    continue;
287                if (src == ednd || ((GeomUtils.duplicateNodes(ednd.getCoor(), src.getCoor(), m_duplicateNodesPrecision) && filter.evaluate(ednd)))) {
288                    if (ednode1 == null)
289                        ednode1 = ednd;
290                    else if (ednd.originalNode().getUniqueId() > ednode1.originalNode().getUniqueId())
291                        ednode1 = ednd;
292                }
293            }
294            else {
295                if ((GeomUtils.duplicateNodes(nd.getCoor(), src.getCoor(), m_duplicateNodesPrecision) && filter.evaluate(nd))) {
296                    if (node1 == null)
297                        node1 = nd;
298                    else if (nd.getUniqueId() > node1.getUniqueId())
299                        node1 = nd;
300                }
301            }
302        }
303        if (ednode1 != null) {
304            if (ednode1 == src)
305                return null; // best match is the source node itself, nothing do merge
306            else
307                return ednode1;
308        }
309        if (node1 != null)
310            return useNode(node1);
311
312        // (2) look in edited nodes; prefer nodes having an original Node over new nodes,
313        //     use a matching one with the highest ID
314        EdNode ornode2 = null;
315        EdNode nwnode2 = null;
316        for (EdNode ednd: searchEdNodes(bbox)) {
317            if (ednd.isDeleted())
318                continue;
319            if (!ednd.hasEditorReferrers())
320                continue;
321            if (src == ednd || ((GeomUtils.duplicateNodes(ednd.getCoor(), src.getCoor(), m_duplicateNodesPrecision) && filter.evaluate(ednd)))) {
322                if (ednd.hasOriginal()) {
323                    if (ornode2 == null)
324                        ornode2 = ednd;
325                    else if (ednd.originalNode().getUniqueId() > ornode2.originalNode().getUniqueId())
326                        ornode2 = ednd;
327                }
328                else {
329                    if (nwnode2 == null)
330                        nwnode2 = ednd;
331                    else if (ednd.getUniqueId() > nwnode2.getUniqueId())
332                        nwnode2 = ednd;
333                }
334            }
335        }
336        EdNode node3 = (ornode2 != null ? ornode2 : nwnode2);
337        if (node3 == src)
338            return null; // best match is the source node itself, nothing to merge
339
340        return node3;
341    }

343    public void updateModifiedFlags() {
344        for (EdNode n: m_nodes)
345            if (!n.isDeleted())
346                n.updateModifiedFlag();
347        for (EdWay w: m_ways)
348            if (!w.isDeleted())
349                w.updateModifiedFlag();
350        for (EdMultipolygon mp: m_multipolygons)
351            if (!mp.isDeleted())
352                mp.updateModifiedFlag();
353    }

355    public List<Command> finalizeEdit (EdObject required_object, double resurrect_dist) {
356
357        System.out.println("WayEditor.finalizeEdit(): ");
358
359        // reset modified flags, if possible
360        updateModifiedFlags();
361
362        // remember required object components
363        Set<EdWay> required_object_ways = null;
364        EdNode required_object_node = null;
365        if (required_object != null && !required_object.isDeleted()) {
366            if (required_object.isNode())
367                required_object_node = (EdNode)required_object;
368            else {
369                required_object_ways = new HashSet<>();
370                if (required_object.isWay())
371                    required_object_ways.add((EdWay)required_object);
372                else if (required_object.isMultipolygon()) {
373                    required_object_ways.addAll(((EdMultipolygon)required_object).allWays());
374                }
375            }
376        }
377
378        // get ways and nodes required in the resulting DataSet
379        Set<EdWay> required_ways = new HashSet<> ();
380        Set<EdNode> required_nodes = new HashSet<> ();
381        for (EdWay w: m_ways) {
382            if (w.isDeleted())
383                continue;
384            if (w.hasEditorReferrers() || w.isTagged() || w.hasExternalReferrers() || (required_object_ways != null && required_object_ways.contains(w))) {
385                required_ways.add(w);
386                for (int i = 0; i < w.getNodesCount(); i++)
387                    required_nodes.add(w.getNode(i));
388            }
389        }
390        for (EdNode n: m_nodes) {
391            if (n.isDeleted())
392                continue;
393            if (required_nodes.contains(n))
394                continue;
395            if (n.isTagged() || n.hasExternalReferrers() || required_object_node == n)
396                required_nodes.add(n);
397        }
398
399        // nodes to be added/changed
400        Set<EdNode> add_nodes = new HashSet<> ();
401        Set<EdNode> change_nodes = new HashSet<> ();
402        for (EdNode n: required_nodes) {
403            if (!n.hasOriginal())
404                add_nodes.add(n);
405            else if (n.isModified())
406                change_nodes.add(n);
407        }
408
409        // original nodes to be deleted
410        Set<EdNode> delete_nodes = new HashSet<> ();
411        for (EdNode n: m_nodes) {
412            if (!n.hasOriginal())
413                continue;
414            if (required_nodes.contains(n))
415                continue;
416            delete_nodes.add(n);
417        }
418
419        // resurrect anonymous deleted nodes
420        if (resurrect_dist > 0) {
421            Set<EdWay> affected_ways = resurrectNodes(add_nodes, change_nodes, delete_nodes, resurrect_dist);
422            if (affected_ways != null) {
423                for (EdWay way: affected_ways)
424                    way.updateModifiedFlag();
425            }
426        }
427
428        // ways to be added/changed
429        Set<EdWay> add_ways = new HashSet<>();
430        Set<EdWay> change_ways = new HashSet<>();
431        for (EdWay w: required_ways) {
432            if (!w.hasOriginal() && !w.isDeleted())
433                add_ways.add(w);
434            else if (w.hasOriginal() && !w.isDeleted() && w.isModified()) {
435                change_ways.add(w);
436            }
437        }
438
439        // ways to be deleted
440        Set<EdWay> delete_ways = new HashSet<>();
441        for (EdWay w: m_ways) {
442            if (!w.hasOriginal())
443                continue;
444            if (required_ways.contains(w))
445                continue;
446            delete_ways.add(w);
447        }
448
449        List<Command> cmds = new LinkedList<>();
450
451        // commands to add new nodes
452        for (EdNode n: add_nodes)
453            cmds.add(new AddCommand(m_dataSet, n.finalNode()));
454
455        // commands to change original nodes
456        for (EdNode n: change_nodes)
457            cmds.add(new ChangeCommand(n.originalNode(), n.finalNode()));
458
459        // commands to add new ways
460        for (EdWay w: add_ways) {
461            cmds.add(new AddCommand(m_dataSet, w.finalWay()));
462            System.out.println(" - add way: " + Long.toString(w.getUniqueId()));
463        }
464
465        // commands to change original ways
466        for (EdWay w: change_ways) {
467            cmds.add(new ChangeCommand(w.originalWay(), w.finalWay()));
468            System.out.println(" - change way: " + Long.toString(w.getUniqueId()));
469        }
470
471        // multipolygon commands
472        for (EdMultipolygon emp: m_multipolygons) {
473            if (!emp.hasOriginal() && !emp.isDeleted()) {
474                cmds.add(new AddCommand(m_dataSet, emp.finalMultipolygon()));
475                System.out.println(" - add multipolygon: " + Long.toString(emp.getUniqueId()));
476            }
477            else if (emp.hasOriginal() && !emp.isDeleted() && emp.isModified()) {
478                cmds.add(new ChangeCommand(emp.originalMultipolygon(), emp.finalMultipolygon()));
479                System.out.println(" - change multipolygon: " + Long.toString(emp.getUniqueId()));
480            }
481            else if (emp.hasOriginal() && emp.isDeleted()) {
482                cmds.add(new DeleteCommand(emp.finalMultipolygon()));
483                System.out.println(" - delete multipolygon: " + Long.toString(emp.getUniqueId()));
484            }
485        }
486
487        // commands to delete original ways
488        for (EdWay w: delete_ways) {
489            cmds.add(new DeleteCommand(w.originalWay()));
490            System.out.println(" - delete way: " + Long.toString(w.getUniqueId()));
491        }
492
493        // commands to delete original nodes
494        for (EdNode n: delete_nodes)
495            cmds.add(new DeleteCommand(n.originalNode()));
496
497        return cmds;
498    }

500    private List<EdNode> searchEdNodes(BBox bbox) {
501        List<EdNode> result = new ArrayList<>();
502        for (EdNode ednd: m_nodes) {
503            if (ednd.isDeleted())
504                continue;
505            if (bbox.bounds(ednd.getCoor()))
506                result.add(ednd);
507        }
508        return result;
509    }

551    public Set<EdObject> useNonEditedAreasContainingPoint(LatLon pos, IEdAreaPredicate filter) {
552        Set<EdObject> areas = new HashSet<>();
553        BBox bbox = new BBox(pos, pos);
554        Node posnode = new Node (pos);
555
556        // look for non-edited multipolygons
557        for (Relation rel: getDataSet().searchRelations(bbox)) {
558            if (isEdited(rel))
559                continue;
560            if (!EdMultipolygon.isUsableRelation(rel))
561                continue;
562            if (!filter.evaluate(rel))
563                continue;
564            if (!Geometry.isNodeInsideMultiPolygon(posnode, rel, null))
565                continue;
566            areas.add(this.useMultipolygon(rel));
567        }
568
569        // look for non-edited ways
570        for (Way w : getDataSet().searchWays(bbox)) {
571            if (!w.isUsable() || isEdited(w) || w.hasIncompleteNodes())
572                continue;
573            if (!filter.evaluate(w))
574                continue;
575            if (!Geometry.nodeInsidePolygon(posnode, w.getNodes()))
576                continue;
577
578            areas.add(useWay(w));
579        }
580
581        return areas;
582    }

584    public Set<EdObject> useAllAreasInBBox(BBox bbox, IEdAreaPredicate filter) {
585        Set<EdObject> areas = new HashSet<>();
586
587        // look for non-edited multipolygons
588        for (Relation rel: getDataSet().searchRelations(bbox)) {
589            if (isEdited(rel))
590                continue;
591            if (!EdMultipolygon.isUsableRelation(rel))
592                continue;
593            if (filter.evaluate(rel))
594                areas.add(this.useMultipolygon(rel));
595        }
596
597        // look for edited multipolygons
598        for (EdMultipolygon mp: this.searchEdMultipolygons(bbox)) {
599            if (areas.contains(mp))
600                continue;
601            if (filter.evaluate(mp))
602                areas.add(mp);
603        }
604
605        // look for edited ways
606        for (EdWay w: this.searchEdWays(bbox)) {
607            if (!filter.evaluate(w))
608                continue;
609
610            // Way is member of a multipolygon, so it was already implicitly included
611            // as a part of EdMultipolygon above, or it's a tagged member of
612            // multipolygon that doesn't match given filter. Which can be dangerous
613            // and we ignore it (for now). #### review this note
614            if (w.isMemberOfAnyMultipolygon())
615                continue;
616
617            areas.add(w);
618        }
619
620        // look for non-edited ways
621        for (Way w : getDataSet().searchWays(bbox)) {
622            if (!w.isUsable())
623                continue;
624            if (isEdited(w))
625                continue;
626            if (!filter.evaluate(w))
627                continue;
628
629            // Ignore member of a multipolygon, see notes above
630            List<Relation> relations = new ArrayList<>(Utils.filteredCollection(w.getReferrers(), Relation.class));
631            boolean is_member_of_any_multipolygon = false;
632            for (Relation rel: relations) {
633                if (rel.isMultipolygon()) {
634                    is_member_of_any_multipolygon = true;
635                    break;
636                }
637            }
638            if (is_member_of_any_multipolygon)
639                continue;
640
641            areas.add(useWay(w));
642        }
643        return areas;
644    }

646    class ResurrectableNodesPair implements Comparable<ResurrectableNodesPair> {
647        public final EdNode delete_node;
648        public final EdNode add_node;
649        public final double distance;
650
651        public ResurrectableNodesPair(EdNode an, EdNode dn, double dist) {
652            delete_node = dn;
653            add_node = an;
654            distance = dist;
655        }
656
657        @Override
658        public int compareTo(ResurrectableNodesPair t) {
659            return Double.compare(this.distance, t.distance);
660        }
661    }

664    private Set<EdWay> resurrectNodes(Set<EdNode> add_nodes, Set<EdNode> change_nodes, Set<EdNode> delete_nodes, double resurrect_dist) {
665        if (add_nodes.isEmpty() || delete_nodes.isEmpty())
666            return null;
667
668        PriorityQueue<ResurrectableNodesPair> queue = new PriorityQueue<>();
669
670        for (EdNode dn: delete_nodes) {
671
672            Node dnorig = dn.originalNode();
673
674            // don't resurrect explicitly deleted nodes and nodes that were originally tagged
675            if (dn.isDeleted() || dnorig.isTagged())
676                continue;
677
678            for (EdNode an: add_nodes) {
679                double dist = dnorig.getCoor().greatCircleDistance(an.getCoor());
680                if (dist > resurrect_dist)
681                    continue;
682                queue.add(new ResurrectableNodesPair (an, dn, dist));
683            }
684        }
685
686        Set<EdWay> affected_ways = new HashSet<>();
687
688        ResurrectableNodesPair rnp;
689        while ((rnp = queue.poll()) != null) {
690            if (!delete_nodes.contains(rnp.delete_node) || !add_nodes.contains(rnp.add_node))
691                continue;
692            System.out.println("Resurrecting node " + Long.toString(rnp.delete_node.getUniqueId()) + " <- " + Long.toString(rnp.add_node.getUniqueId()) + ", dist: " + Double.toString(rnp.distance));
693            add_nodes.remove(rnp.add_node);
694            delete_nodes.remove(rnp.delete_node);
695            change_nodes.add(rnp.add_node);
696            rnp.add_node.forgeOriginalNodeDangerous(rnp.delete_node.originalNode());
697            affected_ways.addAll(rnp.add_node.getEditorReferrers(EdWay.class));
698        }
699
700        return affected_ways;
701    }

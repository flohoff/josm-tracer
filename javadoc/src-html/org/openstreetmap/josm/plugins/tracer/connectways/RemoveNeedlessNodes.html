022import java.util.ArrayList;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Set;
026import org.openstreetmap.josm.data.Bounds;
027import org.openstreetmap.josm.data.osm.BBox;

029public class RemoveNeedlessNodes {
030    private final IEdAreaPredicate m_filter;
031    private final IEdAreaPredicate m_negatedFilter;
032
033    private Set<EdNode> m_requiredNodes = null;
034    private Set<EdNode> m_validBindingsCache = null;
035    private Set<EdNode> m_nodesToRemove = null;
036    private Set<EdWay> m_removeInWays = null;
037
038    private final GeomDeviation m_XteDeviation;
039    private final double m_MinimalVertexAngle;
040    private final BBox m_removeBBox; // can be null
041
042    public RemoveNeedlessNodes (IEdAreaPredicate filter, GeomDeviation xte_deviation, double minimal_vertex_angle, BBox remove_bbox) {
043        m_filter = filter;
044        m_negatedFilter = new NegatedAreaPredicate(filter);
045        m_XteDeviation = xte_deviation;
046        m_MinimalVertexAngle = minimal_vertex_angle;
047        m_removeBBox = remove_bbox;
048    }
049
050    public RemoveNeedlessNodes (IEdAreaPredicate filter, GeomDeviation xte_deviation, double minimal_vertex_angle) {
051        this (filter, xte_deviation, minimal_vertex_angle, null);
052    }
053
054
055    public void removeNeedlessNodes(Set<EdWay> input_ways) {
056
057        System.out.println("Removing needless nodes");
058
059        m_requiredNodes = new HashSet<>();
060        m_validBindingsCache = new HashSet<>();
061        m_nodesToRemove = new HashSet<>();
062        m_removeInWays = new HashSet<>();
063
064        Set<EdWay> ways = new HashSet<>();
065        for (EdWay way: input_ways) {
066            if (m_filter.evaluate(way)) {
067                ways.add(way);
068                m_removeInWays.add(way);
069            }
070        }
071
072        for (EdWay way: ways) {
073            collectRequiredNodes(way);
074        }
075
076        for (EdWay way: ways) {
077            selectNeedlessNodesInWay(way);
078        }
079
080        for (EdWay way: m_removeInWays) {
081            removeNeedlessNodesInWay(way);
082        }
083    }
084
085    private void removeNeedlessNodesInWay(EdWay way) {
086        List<EdNode> nodes = way.getNodes();
087        int ncount = nodes.size();
088        boolean closed = way.isClosed();
089
090        if (closed)
091            ncount--;
092
093        List<EdNode> result = new ArrayList<>();
094        boolean modified = false;
095
096        for (int i = 0; i < ncount; i++) {
097            EdNode n = nodes.get(i);
098            if (m_requiredNodes.contains(n) || !m_nodesToRemove.contains(n)) {
099                result.add(n);
100            }
101            else {
102                System.out.println(" + Removing needless node " + Long.toString(n.getUniqueId()) + " from way " + Long.toString(way.getUniqueId()));
103                modified = true;
104            }
105        }
106
107        if (!modified)
108            return;
109
110        if (closed)
111            result.add(result.get(0));
112        way.setNodes(result);
113    }
114
115    private void selectNeedlessNodesInWay(EdWay way) {
116
117        List<EdNode> nodes = way.getNodes();
118        int ncount = nodes.size();
119        boolean closed = way.isClosed();
120
121        if (closed)
122            ncount--;
123
124        // find first required node
125        int first = -1;
126        for (int i = 0; i < ncount; i++) {
127            if (m_requiredNodes.contains(nodes.get(i))) {
128                first = i;
129                break;
130            }
131        }
132
133        // no required nodes in way? #### maybe find node with the maximal vertex angle
134        if (first < 0)
135            first = 0;
136
137        int last = closed ? first : ncount - 1;
138        int start = first;
139        do {
140            // find next required node
141            int i = (start + 1) % ncount;
142            for (; i != last; i = (i + 1) % ncount) {
143                if (m_requiredNodes.contains(nodes.get(i)))
144                    break;
145            }
146
147            selectNeedlessNodesInSegment(nodes, start, i, ncount);
148            start = i;
149
150        } while (start != last);
151    }
152
153    private void selectNeedlessNodesInSegment(List<EdNode> nodes, int first, int last, int ncount) {
154
155        if (first == last)
156            return;
157        int i = (first + 1) % ncount;
158        if (i == last)
159            return;
160
161        EdNode n1 = nodes.get(first);
162        EdNode n2 = nodes.get(last);
163
164        int imaxd = -1;
165        double maxd = -1;
166        int imaxa = -1;
167        double maxa = -1;
168
169        for (; i != last; i = (i + 1) % ncount) {
170            EdNode p = nodes.get(i);
171            GeomDeviation dev = GeomUtils.pointDeviationFromSegment(p, n1, n2);
172            System.out.println(" - Xte distance: " + Double.toString(dev.distanceMeters()) + ", angle: " + Double.toString(Math.toDegrees(dev.angleRad())) + ", p: " + Long.toString(p.getUniqueId()) + ", n1: " + Long.toString(n1.getUniqueId()) + ", n2: " + Long.toString(n2.getUniqueId()));
173            if (!dev.inTolerance(m_XteDeviation)) {
174                if (imaxd < 0 || dev.distanceMeters() > maxd) {
175                    imaxd = i;
176                    maxd = dev.distanceMeters();
177                }
178                if (imaxa < 0 || dev.angleRad() > maxa) {
179                    imaxa = i;
180                    maxa = dev.angleRad();
181                }
182            }
183        }
184
185        if (imaxd >= 0 || imaxa >= 0) {
186            int imax = (imaxd >= 0) ? imaxd : imaxa;
187            m_requiredNodes.add(nodes.get(imax));
188            selectNeedlessNodesInSegment(nodes, first, imax, ncount);
189            selectNeedlessNodesInSegment(nodes, imax, last, ncount);
190            return;
191        }
192
193        for (i = (first + 1) % ncount; i != last; i = (i + 1) % ncount) {
194            m_nodesToRemove.add(nodes.get(i));
195        }
196    }

198    private void collectRequiredNodes(EdWay way) {
199
200        List<EdNode> nodes = way.getNodes();
201        int ncount = nodes.size();
202        boolean closed = way.isClosed();
203
204        if (closed)
205            ncount--;
206
207        if (ncount <= 2) {
208            m_requiredNodes.addAll(nodes);
209            return;
210        }
211
212        List<Bounds> bounds = way.getEditor().getDataSet().getDataSourceBounds();
213        Set<EdNode> seen_nodes = new HashSet<>();
214
215        for (int i = 0; i < ncount; i++) {
216            EdNode cur_node = nodes.get(i);
217
218            // already required?
219            if (m_requiredNodes.contains(cur_node))
220                continue;
221
222            // node outside bbox?
223            if (m_removeBBox != null && !m_removeBBox.bounds(cur_node.getCoor())) {
224                m_requiredNodes.add(cur_node);
225                continue;
226            }
227
228            // tagged, first/last node of non-closed way, 
229            // or node occurring more than once in the way?
230            if (cur_node.isTagged() ||
231                    (!closed && (i == 0 || i == ncount-1)) ||
232                    seen_nodes.contains(cur_node)) {
233                m_requiredNodes.add(cur_node);
234                continue;
235            }
236
237            seen_nodes.add(cur_node);
238
239            // global node tests, cache them to avoid repeated checks
240            if (!m_validBindingsCache.contains(cur_node)) {
241
242                // filter out nodes outside downloaded area
243                if (!cur_node.isInsideBounds(bounds, LatLonSize.Zero)) {
244                    m_requiredNodes.add(cur_node);
245                    System.out.println(" - Outside-bounds node " + Long.toString(cur_node.getUniqueId()));
246                    continue;
247                }
248
249                List<EdWay> referrers = cur_node.getAllAreaWayReferrers(m_filter);
250
251                // filter out non-matching and suspicious memberships
252                boolean othref = false;
253                for (EdWay refway: referrers) {
254                    if (!m_filter.evaluate(refway) || refway.hasMatchingReferrers(m_negatedFilter)) {
255                        othref = true;
256                        break;
257                    }
258                }
259                if (othref || cur_node.hasExternalReferrers()) {
260                    m_requiredNodes.add(cur_node);
261                    continue;
262                }
263
264                // node is a junction where different neighbors met
265                EdNode prev_node = nodes.get((i + ncount - 1) % ncount);
266                EdNode next_node = nodes.get((i + 1) % ncount);
267                if (!nodeHasSameNeighborsInAllWays(cur_node, way, referrers, prev_node, next_node)) {
268                    m_requiredNodes.add(cur_node);
269                    System.out.println(" - Junction node " + Long.toString(cur_node.getUniqueId()));
270                    continue;
271                }
272
273                // check minimal vertex angle
274                if (m_MinimalVertexAngle > 0) {
275                    double angle = GeomUtils.unorientedAngleBetween(prev_node, cur_node, next_node);
276                    if (angle < m_MinimalVertexAngle) {
277                        m_requiredNodes.add(cur_node);
278                        System.out.println(" - Angle node " + Long.toString(cur_node.getUniqueId()));
279                        continue;
280                    }
281                }
282
283                m_removeInWays.addAll(referrers);
284                m_validBindingsCache.add(cur_node);
285
286                System.out.println(" - Needless candidate " + Long.toString(cur_node.getUniqueId()));
287            }
288        }
289    }
290
291    private boolean nodeHasSameNeighborsInAllWays(EdNode node, EdWay skip_way, List<EdWay> ways, EdNode node1, EdNode node2) {
292        for (EdWay way: ways) {
293            if (way == skip_way)
294                continue;
295            if (!nodeHasSameNeighborsInWay(node, way, node1, node2))
296                return false;
297        }
298        return true;
299    }
300
301    private boolean nodeHasSameNeighborsInWay (EdNode node, EdWay way, EdNode node1, EdNode node2) {
302        List<EdNode> nodes = way.getNodes();
303        int ncount = nodes.size();
304        boolean closed = way.isClosed();
305
306        if (closed)
307            ncount--;
308
309        if (ncount <= 2)
310            return false;
311
312        for (int i = 0; i < ncount; i++) {
313            EdNode cur_node = nodes.get(i);
314            if (cur_node != node)
315                continue;
316
317            EdNode prev_node =
318                    (!closed && i == 0) ? null :
319                    nodes.get((i + ncount - 1) % ncount);
320            EdNode next_node =
321                    (!closed && i == ncount - 1) ? null :
322                    nodes.get((i + 1) % ncount);
323
324            boolean same_neighbors =
325                    (prev_node == node1) && (next_node == node2) ||
326                    (prev_node == node2) && (next_node == node1);
327            if (!same_neighbors)
328                return false;
329        }
330
331        return true;
332    }

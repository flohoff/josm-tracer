046    public void clipAreas(EdWay clip_way, AreaPredicate filter) {
047
048        Set<EdObject> areas = m_editor.useAllAreasInBBox(clip_way.getBBox(), filter);
049        for (EdObject obj : areas) {
050            if (obj.isMultipolygon()) {
051                EdMultipolygon subject_mp = (EdMultipolygon) obj;
052                if (subject_mp.containsWay(clip_way)) {
053                    continue;
054                }
055
056                clipSimpleMulti(clip_way, subject_mp);
057
058            } else if (obj.isWay()) {
059                EdWay subject_way = (EdWay) obj;
060                if (subject_way == clip_way) {
061                    continue;
062                }
063
064                clipSimpleSimple(clip_way, subject_way);
065            }
066        }
067    }

069    private boolean canSilentlyDiscard(EdWay way, double cutoffs_percent) {
070        IDiscardableCutoffPredicate pred = m_settings.discardablePredicate();
071        if (pred == null)
072            return false;
073        if (way.hasReferrers())
074            return false;
075        return pred.canSilentlyDiscard(way, cutoffs_percent);
076    }

078    private void clipSimpleSimple(EdWay clip_way, EdWay subject_way) {
079        // First, connect touching nodes of subject_way to clip_way. This is necessary because
080        // LPIS polygons series contain very small gaps that need to be elliminated before
081        // clipping is performed. Also, there are false joint points on LPIS polygons' edges
082        // where nodes must be added too.
083        subject_way.connectNonIncludedTouchingNodes(m_settings.connectSubjectToClipTolerance(), clip_way);
084
085        System.out.println("Computing difference: clip_way=" + Long.toString(clip_way.getUniqueId()) + ", subject_way=" + Long.toString(subject_way.getUniqueId()));
086
087        AngPolygonClipper clipper = new AngPolygonClipper(m_editor, m_settings.clipperWayCleanupsTolerance(), m_settings.discardCutoffsPercent());
088        clipper.polygonDifference(clip_way, subject_way);
089        if (clipper.changesOutsideDataBounds()) {
090            addPostTraceNotification(tr("Simple way {0} would be modified outside downloaded area, ignoring.", subject_way.getUniqueId()));
091            return;
092        }
093
094        List<List<EdNode>> outers = clipper.outerPolygons();
095        List<List<EdNode>> inners = clipper.innerPolygons();
096
097        System.out.println("- result: outers=" + Long.toString(outers.size()) + ", inners=" + Long.toString(inners.size()));
098
099        if (outers.isEmpty() && inners.isEmpty()) {
100            if (canSilentlyDiscard(subject_way, clipper.discardedPercent())) {
101                // #### fix EdObjects deletion support, instead of this hack
102                subject_way.setKeys(new HashMap<String, String>());
103                subject_way.setNodes(new ArrayList<EdNode>());
104            }
105            else {
106                addPostTraceNotification(tr("Simple way {0} would be completely removed, ignoring, please check.", subject_way.getUniqueId()));
107            }
108        } else if (outers.size() == 1 && inners.isEmpty()) {
109            handleSimpleSimpleSimple(clip_way, subject_way, outers.get(0));
110        } else if ((outers.size() + inners.size()) > 1) {
111            handleSimpleSimpleMulti(clip_way, subject_way, outers, inners);
112        } else {
113            throw new AssertionError(tr("PolygonClipper.polygonDifference returned nonsense!"));
114        }
115    }

117    private void clipSimpleMulti(EdWay clip_way, EdMultipolygon subject_mp) {
118
119        boolean subject_has_nonclosed_ways = subject_mp.containsNonClosedWays();
120
121        // #### add support for multipolygons with non-closed ways
122        if (subject_has_nonclosed_ways) {
123            // addPostTraceNotification(tr("Ignoring multipolygon {0}, it contains non-closed ways.", subject_mp.getUniqueId()));
124            System.out.println("Ignoring multipolygon " + subject_mp.getUniqueId() + ", it contains non-closed ways.");
125            return;
126        }
127
128        // First, connect all touching nodes of all subject ways to clip_way. This is necessary because
129        // LPIS polygons series contain very small gaps that need to be elliminated before
130        // clipping is performed. Also, there are false joint points on LPIS polygons' edges
131        // where nodes must be added too.
132        for (EdWay way : subject_mp.allWays()) {
133            way.connectNonIncludedTouchingNodes(m_settings.connectSubjectToClipTolerance(), clip_way);
134        }
135
136        System.out.println("Computing difference: clip_way=" + Long.toString(clip_way.getUniqueId()) + ", subject_relation=" + Long.toString(subject_mp.getUniqueId()));
137
138        AngPolygonClipper clipper = new AngPolygonClipper(m_editor, m_settings.clipperWayCleanupsTolerance(), m_settings.discardCutoffsPercent());
139        clipper.polygonDifference(clip_way, subject_mp);
140        if (clipper.changesOutsideDataBounds()) {
141            addPostTraceNotification(tr("Multipolygon {0} would be modified outside downloaded area, ignoring.", subject_mp.getUniqueId()));
142            return;
143        }
144
145        List<List<EdNode>> unmapped_new_outers = new ArrayList<>(clipper.outerPolygons());
146        List<List<EdNode>> unmapped_new_inners = new ArrayList<>(clipper.innerPolygons());
147
148        System.out.println("- result: outers=" + Long.toString(unmapped_new_outers.size()) + ", inners=" + Long.toString(unmapped_new_inners.size()));
149
150        // Whole multipolygon disappeared
151        if (unmapped_new_outers.isEmpty() && unmapped_new_inners.isEmpty()) {
152            // #### add handling of discarded cutoffs, but be careful
153            addPostTraceNotification(tr("Multipolygon {0} would be completely removed, ignoring, please check.", subject_mp.getUniqueId()));
154            return;
155        }
156
157        // Create bi-directional mapping of identical geometries
158        List<EdWay> unmapped_old_outers = new ArrayList<>(subject_mp.outerWays());
159        List<EdWay> unmapped_old_inners = new ArrayList<>(subject_mp.innerWays());
160        mapIdenticalWays(unmapped_old_outers, unmapped_new_outers);
161        mapIdenticalWays(unmapped_old_inners, unmapped_new_inners);
162
163        // All new ways were successfully mapped to old ways?
164        if (unmapped_old_outers.isEmpty() && unmapped_old_inners.isEmpty() && unmapped_new_outers.isEmpty() && unmapped_new_inners.isEmpty()) {
165            System.out.println(" o subject unchanged");
166            return;
167        }
168
169        System.out.println("- unmapped_outers: old=" + Long.toString(unmapped_old_outers.size()) + ", new=" + Long.toString(unmapped_new_outers.size()));
170        System.out.println("- unmapped_inners: old=" + Long.toString(unmapped_old_inners.size()) + ", new=" + Long.toString(unmapped_new_inners.size()));
171
172        // Handle the easiest and most common case, only one outer way of a multipolygon was clipped
173        // (Maybe, I should test that the old and new outer ways have non-empty intersection. Otherwise,
174        // it means that one brand new outer way was created and one old outer way was completely deleted.
175        // But as long as clip is a simple way, it should never happen, because new way can be created only
176        // by clipping another existing way.)
177        if (!subject_has_nonclosed_ways && unmapped_old_outers.size() == 1 && unmapped_new_outers.size() == 1
178                && unmapped_old_inners.isEmpty() && unmapped_new_inners.isEmpty()) {
179            EdWay old_outer_way = unmapped_old_outers.get(0);
180            List<EdNode> new_outer_way = unmapped_new_outers.get(0);
181            handleSimpleMultiOneOuterModified(clip_way, subject_mp, old_outer_way, new_outer_way);
182            return;
183        }
184
185        // If modified ways have no (interesting) tags and no other referrers, we can
186        // replace multipolygon's geometry quite agressively.
187        if (!subject_has_nonclosed_ways
188                && untaggedSingleReferrerWays(unmapped_old_outers, subject_mp)
189                && untaggedSingleReferrerWays(unmapped_old_inners, subject_mp)) {
190            handleSimpleMultiAgressiveUpdate(clip_way, subject_mp,
191                    unmapped_old_outers, unmapped_old_inners, unmapped_new_outers, unmapped_new_inners);
192            return;
193        }
194
195        addPostTraceNotification(tr("Multipolygon clipping result of {0} is too complex.", subject_mp.getUniqueId()));
196    }

198    /**
199     * Returns true if all ways in the list have no (interesting) tag,
200     * have the given multipolygon referrer and no other referrers.
201     * @param list list of ways to check
202     * @param referrer expected single referrer
203     * @return true if all ways satisfy the requirements
204     */
205    private static boolean untaggedSingleReferrerWays(List<EdWay> list, EdMultipolygon referrer) {
206        for (EdWay way: list) {
207            if (way.isTagged())
208                return false;
209            if (!way.hasSingleReferrer(referrer))
210                return false;
211        }
212        return true;
213    }

215    private static void mapIdenticalWays(List<EdWay> unmapped_old, List<List<EdNode>> unmapped_new) {
216        int iold = 0;
217        while (iold < unmapped_old.size()) {
218            EdWay old_way = unmapped_old.get(iold);
219            List<EdNode> new_way = null;
220            int inew = 0;
221            for (; inew < unmapped_new.size(); inew++) {
222                List<EdNode> test_way = unmapped_new.get(inew);
223                if (old_way.hasIdenticalEdNodeGeometry(test_way, true)) {
224                    new_way = test_way;
225                    break;
226                }
227            }
228            if (new_way != null) {
229                //mapped_old_new.put(old_way, new_way);
230                //mapped_new_old.put(new_way, old_way);
231                unmapped_old.remove(iold);
232                unmapped_new.remove(inew);
233            }
234            else {
235                iold++;
236            }
237        }
238    }

240    class SimilarWaysPair implements Comparable<SimilarWaysPair> {
241        public final EdWay src;
242        public final List<EdNode> dst;
243        public final double similarity;
244
245        public SimilarWaysPair(EdWay s, List<EdNode> d, double sim) {
246            src = s;
247            dst = d;
248            similarity = sim;
249        }
250
251        @Override
252        public int compareTo(SimilarWaysPair t) {
253            int x = -Double.compare(this.similarity, t.similarity);
254            if (x != 0)
255                return x;
256            return -Integer.compare(this.src.getNodesCount(), t.src.getNodesCount());
257        }
258    }

260    /**
261     * For every source EdWay, it tries to find the most similar dest way;
262     * for every dest way, it tries to find the most similar source EdWay.
263     * Ways with no suitable similar ways are excluded from the results.
264     * Forward mapping is guaranteed to be a bijection (suitable for
265     * way updates), reverse mapping can contain duplicate values (surjection).
266     *
267     * @param srcs list of source EdWays
268     * @param dsts list of dest ways
269     * @return a pair, first element is a map of source EdWays to most similar
270     * dest ways, second element is a map of dest ways to most similar source EdWay
271     */
272    private Pair<Map<EdWay, List<EdNode>>, Map<List<EdNode>, EdWay>> pairSimilarWays (List<EdWay> srcs, List<List<EdNode>> dsts) {
273        // Create Cartesian product of ways and sort pairs according to their similarity
274        // (For simplicity, we estimate similarity from percentage of shared nodes.
275        // It would be better to base the similarity on areas of polygon intersections.)
276        PriorityQueue<SimilarWaysPair> queue = new PriorityQueue<>();
277        for(EdWay w: srcs) {
278            for (List<EdNode> nw: dsts) {
279                int shared_nodes = w.getSharedNodesCount(nw);
280                if (shared_nodes > 0) {
281                    double similarity = (double)shared_nodes / w.getNodesCount();
282                    queue.add(new SimilarWaysPair(w, nw, similarity));
283                }
284            }
285        }
286
287        // Map ways...
288        Map<EdWay, List<EdNode>> res1 = new HashMap<>();
289        Map<List<EdNode>, EdWay> res2 = new HashMap<>();
290        Set<List<EdNode>> res1used = new HashSet<>();
291        SimilarWaysPair swp;
292        while ((swp = queue.poll()) != null) {
293
294            // forward mapping
295            if (!res1.containsKey(swp.src) && !res1used.contains(swp.dst)) {
296                res1.put(swp.src, swp.dst);
297                res1used.add(swp.dst);
298            }
299
300            // reverse mapping
301            if (!res2.containsKey(swp.dst)) {
302                res2.put(swp.dst, swp.src);
303            }
304        }
305
306        return new Pair<>(res1, res2);
307    }

309    private void handleSimpleSimpleSimple(EdWay clip_way, EdWay subject_way, List<EdNode> result) {
310        // ** Easiest case - simple way clipped by a simple way produced a single polygon **
311
312        System.out.println("Clip result: simple");
313
314        // Subject way unchanged?
315        if (subject_way.hasIdenticalEdNodeGeometry(result, true)) {
316            System.out.println(" o subject unchanged");
317            return;
318        }
319
320        System.out.println(" ! CLIPPING subject " + Long.toString(subject_way.getUniqueId()));
321
322        // Subject way changed, change its geometry
323        subject_way.setNodes(result);
324
325        // Connect clip_way to subject_way, this step guarantees that all newly created
326        // intersection nodes will be included in both ways.
327        clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), subject_way);
328    }

330    private void handleSimpleMultiOneOuterModified (EdWay clip_way, EdMultipolygon subject_mp, EdWay old_outer_way, List<EdNode> result) {
331        // ** Easy case - clip of a multipolygon modified exactly one outer way and nothing else **
332
333        System.out.println(" ! CLIPPING subject " + Long.toString(subject_mp.getUniqueId()) + ", outer way modified: " + Long.toString(old_outer_way.getUniqueId()));
334
335        // Change geometry of the changed outer way
336        old_outer_way.setNodes(result);
337
338        // Connect clip_way to subject_way, this step guarantees that all newly created
339        // intersection nodes will be included in both ways.
340        clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), old_outer_way);
341    }

343    private void handleSimpleSimpleMulti(EdWay clip_way, EdWay subject_way, List<List<EdNode>> outers, List<List<EdNode>> inners) {
344        // ** Simple way clipped by a simple way produced multiple polygons **
345
346        if (inners.isEmpty()) {
347            System.out.println("Clip result: multi outers");
348            handleSimpleSimpleMultiOuters(clip_way, subject_way, outers);
349        }
350        else {
351            System.out.println("Clip result: multi mixed");
352            // #### not completed
353            addPostTraceNotification(tr("Clipping changes simple way {0} to multipolygon, not supported yet.", subject_way.getUniqueId()));
354        }
355    }

357    private void handleSimpleSimpleMultiOuters(EdWay clip_way, EdWay subject_way, List<List<EdNode>> outers) {
358        // ** Simple subject clipped by a simple way produced multiple simple (outer) polygons **
359
360        // Don't clip subject which is a member of a (non-multipolygon) relation.
361        // It's questionable if all pieces should be added to the relation, or only some of them, etc...
362        if (subject_way.hasEditorReferrers() || subject_way.hasExternalReferrers()) {
363            addPostTraceNotification(tr("Clipped way {0} is a member of non-multipolygon relation, not supported yet.", subject_way.getUniqueId()));
364            return;
365        }
366
367        System.out.println(" ! CLIPPING subject " + Long.toString(subject_way.getUniqueId()) + " to multiple simple ways");
368
369        // #### Generally, it's better to create multiple simple ways than combine them to a new multipolygon.
370        // But in some cases, maybe it would make sense to create a multipolygon... E.g. named landuse areas??
371
372        // find the largest polygon, which will be used to update subject_way's geometry
373        List<EdNode> maxnodes = null;
374        double maxarea = Double.NEGATIVE_INFINITY;
375        for (List<EdNode> nodes: outers) {
376            double area = GeomUtils.getEastNorthArea(nodes);
377            if (area < maxarea)
378                continue;
379            maxnodes = nodes;
380            maxarea = area;
381        }
382
383        // update subject_way geometry and create new ways for the other pieces
384        for (List<EdNode> nodes: outers) {
385            if (nodes == maxnodes) {
386                subject_way.setNodes(nodes);
387                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), subject_way);
388            }
389            else {
390                EdWay new_way = m_editor.newWay(nodes);
391                new_way.setKeys(subject_way.getKeys());
392                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), new_way);
393            }
394        }
395    }

397    private void handleSimpleMultiAgressiveUpdate(EdWay clip_way, EdMultipolygon subject_mp, List<EdWay> unmapped_old_outers, List<EdWay> unmapped_old_inners, List<List<EdNode>> unmapped_new_outers, List<List<EdNode>> unmapped_new_inners) {
398        // ** Agressive update of a multipolygon clipped by a simple way
399
400        if (!unmapped_old_outers.isEmpty() || !unmapped_new_outers.isEmpty()) {
401
402            // Update geometry of outer ways that can be paired together as similar ones
403            Pair<Map<EdWay, List<EdNode>>, Map<List<EdNode>, EdWay>> outer_pair_maps = pairSimilarWays (unmapped_old_outers, unmapped_new_outers);
404            Map<EdWay, List<EdNode>> outer_pairs = outer_pair_maps.a;
405            Map<List<EdNode>, EdWay> outer_revs = outer_pair_maps.b;
406            for (Map.Entry<EdWay, List<EdNode>> pair: outer_pairs.entrySet()) {
407                EdWay old_way = pair.getKey();
408                List<EdNode> new_nodes = pair.getValue();
409                old_way.setNodes(new_nodes);
410                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), old_way);
411                unmapped_old_outers.remove(old_way);
412                unmapped_new_outers.remove(new_nodes);
413                System.out.println("Changing outer geometry " + Long.toString(old_way.getUniqueId()));
414            }
415
416            // Create new outer ways with tagging based on reverse similarity mapping
417            // (If no reverse mapping is available, leave way untagged)
418            for (List<EdNode> new_nodes: unmapped_new_outers) {
419                EdWay new_way = m_editor.newWay(new_nodes);
420                EdWay old_way = outer_revs.get(new_nodes);
421                if (old_way != null)
422                    new_way.setKeys(old_way.getKeys());
423                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), new_way);
424                subject_mp.addOuterWay(new_way);
425                System.out.println("Adding outer way " + Long.toString(new_way.getUniqueId()));
426            }
427
428            // Remove old outer ways that weren't mapped to new ways
429            // (We assume that the ways have no interesting tags and referrers,
430            // and will be automatically deleted by WayEditor.)
431            for (EdWay old_way: unmapped_old_outers) {
432                subject_mp.removeOuterWay(old_way);
433                System.out.println("Removing outer way " + Long.toString(old_way.getUniqueId()));
434            }
435        }
436
437        if (!unmapped_old_inners.isEmpty() || !unmapped_new_inners.isEmpty()) {
438
439            // Update geometry of inner ways that can be paired together as similar ones
440            Pair<Map<EdWay, List<EdNode>>, Map<List<EdNode>, EdWay>> inner_pair_maps = pairSimilarWays (unmapped_old_inners, unmapped_new_inners);
441            Map<EdWay, List<EdNode>> inner_pairs = inner_pair_maps.a;
442            Map<List<EdNode>, EdWay> inner_revs = inner_pair_maps.b;
443            for (Map.Entry<EdWay, List<EdNode>> pair: inner_pairs.entrySet()) {
444                EdWay old_way = pair.getKey();
445                List<EdNode> new_nodes = pair.getValue();
446                old_way.setNodes(new_nodes);
447                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), old_way);
448                unmapped_old_inners.remove(old_way);
449                unmapped_new_inners.remove(new_nodes);
450                System.out.println("Changing inner geometry " + Long.toString(old_way.getUniqueId()));
451            }
452
453            // Create new inner ways
454            for (List<EdNode> new_nodes: unmapped_new_inners) {
455                // use tagging of the most similar reverse mapped way, if any
456                EdWay new_way = m_editor.newWay(new_nodes);
457                EdWay old_way = inner_revs.get(new_nodes);
458                if (old_way != null)
459                    new_way.setKeys(old_way.getKeys());
460                clip_way.connectNonIncludedTouchingNodes(m_settings.reconnectIntersectionNodesTolerance(), new_way);
461                subject_mp.addInnerWay(new_way);
462                System.out.println("Adding inner way " + Long.toString(new_way.getUniqueId()));
463            }
464
465            // Remove old inner ways that weren't mapped to new ways
466            // (We assume that the ways have no interesting tags and referrers,
467            // and will be automatically deleted by WayEditor.)
468            for (EdWay old_way: unmapped_old_inners) {
469                subject_mp.removeInnerWay(old_way);
470                System.out.println("Removing inner way " + Long.toString(old_way.getUniqueId()));
471            }
472        }
473    }

020package org.openstreetmap.josm.plugins.tracer.connectways;
021
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.HashSet;
025import java.util.List;
026import java.util.Map;
027import java.util.Set;
028import org.openstreetmap.josm.data.osm.OsmPrimitive;
029import org.openstreetmap.josm.data.osm.Relation;
030import org.openstreetmap.josm.tools.Pair;
031
032public class MergeIdenticalWays {
033    private final WayEditor m_editor;
034    private final IEdAreaPredicate m_filter;
035    private final IEdAreaPredicate m_negatedFilter;
036
037    public MergeIdenticalWays (WayEditor editor, IEdAreaPredicate filter) {
038        m_editor = editor;
039        m_filter = filter;
040        m_negatedFilter = new NegatedAreaPredicate(filter);
041    }
042
043    public EdWay mergeWays(Set<EdWay> ways, boolean allow_inverted_orientation, EdWay watch_way) {
044        Set<EdWay> inserted = new HashSet<>();
045        List<List<EdWay>> bundles = new ArrayList<>(ways.size());
046
047        for (EdWay way: ways) {
048            // skip already included ways and ways that don't match given predicate
049            if (inserted.contains(way) || !m_filter.evaluate(way))
050                continue;
051
052            // get all matching referrers of a way's node (mergeable ways are expected
053            // to have EdNode-identical geometry, so all of them must obviously be referrers
054            // of any way's node).
055            EdNode node = way.getNode(0);
056            List<EdWay> referrers = node.getAllAreaWayReferrers(m_filter);
057
058            List<EdWay> bundle = new ArrayList<>();
059            for (EdWay refway: referrers) {
060                if (refway == way)
061                    continue;
062                // ignore ways that are also members of other relations than matching areas
063                if (refway.hasMatchingReferrers(m_negatedFilter))
064                    continue;
065                // ignore ways with non-identical geometry
066                if (!refway.hasIdenticalEdNodeGeometry(way.getNodes(), true))
067                    continue;
068                bundle.add(refway);
069                inserted.add(refway);
070            }
071            if (!bundle.isEmpty()) {
072                bundle.add(way);
073                inserted.add(way);
074                bundles.add(bundle);
075            }
076        }
077
078        // For every bundle of identical ways, try to merge them into minimal
079        // possible number of ways.
080        for (List<EdWay> bundle: bundles) {
081            boolean[] merged = new boolean[bundle.size()];
082            for (int i = 0; (i < bundle.size()) && !merged[i]; i++) {
083                for (int j = i+1; (j < bundle.size()) && !merged[j]; j++) {
084                    EdWay wi = bundle.get(i);
085                    EdWay wj = bundle.get(j);
086                    int m = mergeTwoIdenticalWays(wi, wj);
087                    if (m < 0) {
088                        merged[j] = true;
089                        if (watch_way != null && wj == watch_way)
090                            watch_way = wi;
091                    }
092                    else if (m > 0) {
093                        merged[i] = true;
094                        if (watch_way != null && wi == watch_way)
095                            watch_way = wj;
096                        break;
097                    }
098                }
099            }
100        }
101
102        return watch_way;
103    }
104
105    /**
106     * @param way1 first way to merge
107     * @param way2 second way to merge
108     * @return negative value if second way was merged into first,
109     * positive value if first way was merged into second,
110     * zero if merge attempt failed.
111     */
112    private int mergeTwoIdenticalWays(EdWay way1, EdWay way2) {
113        if (selectMergeDestWay(way1, way2))
114            return mergeIdenticalWayTo(way2, way1) ? -1 : 0;
115        else
116            return mergeIdenticalWayTo(way1, way2) ? 1 : 0;
117    }
118
119    /**
120     * Selects which of the given ways should be the destination way of merge.
121     * It prefers original ways over newly added ways and tagged ways over
122     * untagged.
123     * @param way1 first way
124     * @param way2 second way
125     * @return true if way1 should be used as destination way, false otherwise
126     */
127    private boolean selectMergeDestWay(EdWay way1, EdWay way2) {
128        if (way1.hasOriginal() && !way2.hasOriginal())
129            return true;
130        if (way2.hasOriginal() && !way1.hasOriginal())
131            return false;
132        return way1.isTagged();
133    }
134
135    private boolean mergeIdenticalWayTo(EdWay src, EdWay dst) {
136        // Merge all interesting tags into destination EdWay.
137        // If there's a collision, give up and don't merge ways.
138        Map<String, String> tags = new HashMap<>();
139        Map<String, String> srctags = src.getKeys();
140        Map<String, String> dsttags = dst.getKeys();
141        Set<String> keys = new HashSet<>(srctags.keySet());
142        keys.addAll(dsttags.keySet());
143        for (String key: keys) {
144            Pair<Boolean, String> tm = mergeTagValues(key, srctags, dsttags);
145            if (!tm.a)
146                return false;
147            tags.put(key, tm.b);
148        }
149        dst.setKeys(tags);
150
151        System.out.println("Merging identical ways: " + Long.toString(src.getUniqueId()) + " => " + Long.toString(dst.getUniqueId()));
152
153        // load all external multipolygons, make sure that all are usable
154        // (we've already checked that all referrers match the area filter)
155        List<Relation> rels = src.getExternalReferrers(Relation.class);
156        for (Relation rel: rels) {
157            if (!EdMultipolygon.isUsableRelation(rel))
158                return false;
159            m_editor.useMultipolygon(rel);
160        }
161
162        // replace all relation memberships of the source EdWay with destination EdWay memberships
163        List<EdMultipolygon> mps = src.getEditorReferrers(EdMultipolygon.class);
164        for (EdMultipolygon mp: mps)
165            mp.replaceWay (src, dst);
166
167        // remove all tags of the source EdWay
168        src.setKeys(new HashMap<String,String>());
169
170        return true;
171    }
172
173    private Pair<Boolean, String> mergeTagValues(String key, Map<String, String> set1, Map<String, String> set2) {
174        String val1 = set1.get(key);
175        String val2 = set2.get(key);
176        if (val2 == null)
177            return new Pair<>(true, val1);
178        if (val1 == null)
179            return new Pair<>(true, val2);
180        if (val1.equals(val2))
181            return new Pair<>(true, val1);
182        if (!OsmPrimitive.isUninterestingKey(key)) {
183            System.out.println("Cannot merge interesting tags: " + key + "=" + val1 + ", " + key + "=" + val2);
184            return new Pair<>(false, null);
185        }
186
187        if (key.equals("source") || key.equals("fixme"))
188            return new Pair<>(true, combineTagValues(val1, val2));
189
190        return new Pair<>(true, val1);
191    }
192
193    private String combineTagValues(String val1, String val2) {
194        String[] list1 = val1.split(";");
195        String[] list2 = val2.split(";");
196        Set<String> set = new HashSet<>();
197        for (String s: list1) {
198            s = s.trim();
199            if (s.isEmpty())
200                continue;
201            set.add(s);
202        }
203        for (String s: list2) {
204            s = s.trim();
205            if (s.isEmpty())
206                continue;
207            set.add(s);
208        }
209        StringBuilder sb = new StringBuilder();
210        for (String s: set) {
211            if (sb.length() > 0)
212                sb.append(";");
213            sb.append(s);
214        }
215        return sb.toString();
216    }
